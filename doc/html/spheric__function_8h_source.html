258 /// Inner product of two spherical functions.
259 template <function_domain_t domain_t, typename T>
260 T inner(Spheric_function<domain_t, T> const& f1, Spheric_function<domain_t, T> const& f2)
261 {
262  /* check radial grid */
263  if (f1.radial_grid().hash() != f2.radial_grid().hash()) {
264  TERMINATE("radial grids don't match");
265  }
266 
267  Spline<T> s(f1.radial_grid());
268 
269  if (domain_t == spectral) {
270  int lmmax = std::min(f1.angular_domain_size(), f2.angular_domain_size());
271  for (int ir = 0; ir < s.num_points(); ir++) {
272  for (int lm = 0; lm < lmmax; lm++) {
273  s[ir] += type_wrapper<T>::conjugate(f1(lm, ir)) * f2(lm, ir);
274  }
275  }
276  } else {
277  TERMINATE_NOT_IMPLEMENTED
278  }
279  return s.interpolate().integrate(2);
280 }

282 /// Compute Laplacian of the spheric function.
283 /** Laplacian in spherical coordinates has the following expression:
284  * \f[
285  * \Delta = \frac{1}{r^2}\frac{\partial}{\partial r}\Big( r^2 \frac{\partial}{\partial r} \Big) + \frac{1}{r^2}\Delta_{\theta, \phi}
286  * \f]
287  */
288 template <typename T>
289 Spheric_function<spectral, T> laplacian(Spheric_function<spectral, T> const& f__)
290 {
291  Spheric_function<spectral, T> g;
292  auto& rgrid = f__.radial_grid();
293  int lmmax = f__.angular_domain_size();
294  int lmax = Utils::lmax_by_lmmax(lmmax);
295  g = Spheric_function<spectral, T>(lmmax, rgrid);
296 
297  Spline<T> s1(rgrid);
298  for (int l = 0; l <= lmax; l++) {
299  int ll = l * (l + 1);
300  for (int m = -l; m <= l; m++) {
301  int lm = Utils::lm_by_l_m(l, m);
302  /* get lm component */
303  auto s = f__.component(lm);
304  /* compute 1st derivative */
305  for (int ir = 0; ir < s.num_points(); ir++) {
306  s1[ir] = s.deriv(1, ir);
307  }
308  s1.interpolate();
309 
310  for (int ir = 0; ir < s.num_points(); ir++) {
311  g(lm, ir) = 2 * s1[ir] * rgrid.x_inv(ir) + s1.deriv(1, ir) - s[ir] * ll / std::pow(rgrid[ir], 2);
312  }
313  }
314  }
315 
316  return std::move(g);
317 }

319 /// Convert from Ylm to Rlm representation.
320 inline Spheric_function<spectral, double> convert(Spheric_function<spectral, double_complex> const& f__)
321 {
322  int lmax = Utils::lmax_by_lmmax(f__.angular_domain_size());
323 
324  /* cache transformation arrays */
325  std::vector<double_complex> tpp(f__.angular_domain_size());
326  std::vector<double_complex> tpm(f__.angular_domain_size());
327  for (int l = 0; l <= lmax; l++) {
328  for (int m = -l; m <= l; m++) {
329  int lm = Utils::lm_by_l_m(l, m);
330  tpp[lm] = SHT::rlm_dot_ylm(l, m, m);
331  tpm[lm] = SHT::rlm_dot_ylm(l, m, -m);
332  }
333  }
334 
335  Spheric_function<spectral, double> g(f__.angular_domain_size(), f__.radial_grid());
336 
337  for (int ir = 0; ir < f__.radial_grid().num_points(); ir++) {
338  int lm = 0;
339  for (int l = 0; l <= lmax; l++) {
340  for (int m = -l; m <= l; m++) {
341  if (m == 0) {
342  g(lm, ir) = std::real(f__(lm, ir));
343  } else {
344  int lm1 = Utils::lm_by_l_m(l, -m);
345  g(lm, ir) = std::real(tpp[lm] * f__(lm, ir) + tpm[lm] * f__(lm1, ir));
346  }
347  lm++;
348  }
349  }
350  }
351 
352  return std::move(g);
353 }

355 /// Convert from Rlm to Ylm representation.
356 inline Spheric_function<spectral, double_complex> convert(Spheric_function<spectral, double> const& f__)
357 {
358  int lmax = Utils::lmax_by_lmmax(f__.angular_domain_size());
359 
360  /* cache transformation arrays */
361  std::vector<double_complex> tpp(f__.angular_domain_size());
362  std::vector<double_complex> tpm(f__.angular_domain_size());
363  for (int l = 0; l <= lmax; l++) {
364  for (int m = -l; m <= l; m++) {
365  int lm = Utils::lm_by_l_m(l, m);
366  tpp[lm] = SHT::ylm_dot_rlm(l, m, m);
367  tpm[lm] = SHT::ylm_dot_rlm(l, m, -m);
368  }
369  }
370 
371  Spheric_function<spectral, double_complex> g(f__.angular_domain_size(), f__.radial_grid());
372 
373  for (int ir = 0; ir < f__.radial_grid().num_points(); ir++) {
374  int lm = 0;
375  for (int l = 0; l <= lmax; l++) {
376  for (int m = -l; m <= l; m++) {
377  if (m == 0) {
378  g(lm, ir) = f__(lm, ir);
379  } else {
380  int lm1 = Utils::lm_by_l_m(l, -m);
381  g(lm, ir) = tpp[lm] * f__(lm, ir) + tpm[lm] * f__(lm1, ir);
382  }
383  lm++;
384  }
385  }
386  }
387 
388  return std::move(g);
389 }

457 /// Gradient of the function in complex spherical harmonics.
458 inline Spheric_function_gradient<spectral, double_complex> gradient(Spheric_function<spectral, double_complex>& f)
459 {
460  Spheric_function_gradient<spectral, double_complex> g(f.angular_domain_size(), f.radial_grid());
461  for (int i = 0; i < 3; i++)
462  {
463  g[i] = Spheric_function<spectral, double_complex>(f.angular_domain_size(), f.radial_grid());
464  g[i].zero();
465  }
466 
467  int lmax = Utils::lmax_by_lmmax(f.angular_domain_size());
468 
469  //Spline<double_complex> s(f.radial_grid());
470 
471  for (int l = 0; l <= lmax; l++)
472  {
473  double d1 = sqrt(double(l + 1) / double(2 * l + 3));
474  double d2 = sqrt(double(l) / double(2 * l - 1));
475 
476  for (int m = -l; m <= l; m++)
477  {
478  int lm = Utils::lm_by_l_m(l, m);
479  auto s = f.component(lm);
480 
481  for (int mu = -1; mu <= 1; mu++)
482  {
483  int j = (mu + 2) % 3; // map -1,0,1 to 1,2,0
484 
485  if ((l + 1) <= lmax && abs(m + mu) <= l + 1)
486  {
487  int lm1 = Utils::lm_by_l_m(l + 1, m + mu);
488  double d = d1 * SHT::clebsch_gordan(l, 1, l + 1, m, mu, m + mu);
489  for (int ir = 0; ir < f.radial_grid().num_points(); ir++)
490  g[j](lm1, ir) += (s.deriv(1, ir) - f(lm, ir) * f.radial_grid().x_inv(ir) * double(l)) * d;
491  }
492  if ((l - 1) >= 0 && abs(m + mu) <= l - 1)
493  {
494  int lm1 = Utils::lm_by_l_m(l - 1, m + mu);
495  double d = d2 * SHT::clebsch_gordan(l, 1, l - 1, m, mu, m + mu);
496  for (int ir = 0; ir < f.radial_grid().num_points(); ir++)
497  g[j](lm1, ir) -= (s.deriv(1, ir) + f(lm, ir) * f.radial_grid().x_inv(ir) * double(l + 1)) * d;
498  }
499  }
500  }
501  }
502 
503  double_complex d1(1.0 / sqrt(2.0), 0);
504  double_complex d2(0, 1.0 / sqrt(2.0));
505 
506  for (int ir = 0; ir < f.radial_grid().num_points(); ir++)
507  {
508  for (int lm = 0; lm < f.angular_domain_size(); lm++)
509  {
510  double_complex g_p = g[0](lm, ir);
511  double_complex g_m = g[1](lm, ir);
512  g[0](lm, ir) = d1 * (g_m - g_p);
513  g[1](lm, ir) = d2 * (g_m + g_p);
514  }
515  }
516 
517  return g;
518 }
