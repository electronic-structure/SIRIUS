168 /// Gradient of the function in the plane-wave domain.
169 inline Smooth_periodic_function_gradient<double> gradient(Smooth_periodic_function<double>& f__)
170 {
171  Smooth_periodic_function_gradient<double> g(f__);
172 
173  #pragma omp parallel for
174  for (int igloc = 0; igloc < f__.gvec().partition().gvec_count_fft(); igloc++)
175  {
176  int ig = f__.gvec().partition().gvec_offset_fft() + igloc;
177 
178  auto G = f__.gvec().gvec_cart(ig);
179  for (int x: {0, 1, 2}) g[x].f_pw_local(igloc) = f__.f_pw_local(igloc) * double_complex(0, G[x]);
180  }
181  return std::move(g);
182 }

184 /// Laplacian of the function in the plane-wave domain.
185 inline Smooth_periodic_function<double> laplacian(Smooth_periodic_function<double>& f__)
186 {
187  Smooth_periodic_function<double> g(f__.fft(), f__.gvec());
188 
189  #pragma omp parallel for
190  for (int igloc = 0; igloc < f__.gvec().partition().gvec_count_fft(); igloc++)
191  {
192  int ig = f__.gvec().partition().gvec_offset_fft() + igloc;
193 
194  auto G = f__.gvec().gvec_cart(ig);
195  g.f_pw_local(igloc) = f__.f_pw_local(igloc) * double_complex(-std::pow(G.length(), 2), 0);
196  }
197 
198  return std::move(g);
199 }

201 template <typename T>
202 Smooth_periodic_function<T> operator*(Smooth_periodic_function_gradient<T>& grad_f__,
203  Smooth_periodic_function_gradient<T>& grad_g__)
204 
205 {
206  assert(&grad_f__.f().fft() == &grad_g__.f().fft());
207  assert(&grad_f__.f().gvec() == &grad_g__.f().gvec());
208 
209  Smooth_periodic_function<T> result(grad_f__.f().fft());
210 
211  #pragma omp parallel for
212  for (int ir = 0; ir < grad_f__.f().fft().local_size(); ir++)
213  {
214  double d = 0;
215  for (int x: {0, 1, 2})
216  {
217  d += grad_f__[x].f_rg(ir) * grad_g__[x].f_rg(ir);
218  }
219  result.f_rg(ir) = d;
220  }
221 
222  return std::move(result);
223 }

228 /// Representation of a smooth (Fourier-transformable) periodic function.
229 /** The class is designed to handle periodic functions such as density or potential, defined on a regular FFT grid.
230  * The following functionality is expected:
231  * - access to real-space values
232  * - access to plane-wave coefficients
233  * - distribution of plane-wave coefficients over entire communicator
234  * - Fourier transformation using FFT communicator
235  * - gather PW coefficients into global array
236  * In some cases the PW coefficients are not necessary and only the real-space values are stored.
237  */
238 template <typename T>
239 class Smooth_periodic_function
240 {
241  protected:
242 
243  /// FFT driver.
244  FFT3D* fft_{nullptr};
245 
246  /// Distribution of G-vectors.
247  Gvec const* gvec_{nullptr};
248 
249  /// Total communicator which is used to distribute plane-wave coefficients.
250  /** The size of this communicator should be exactly the same as in the fine-grained distribution of G-vectors. */
251  Communicator const* comm_{nullptr};
252 
253  /// Communicator which is orthogonal to FFT communicator.
254  /** This communicator is used to collect G-vectors for the FFT driver. */
255  Communicator const* comm_local_{nullptr};
256 
257  /// Function on the regular real-space grid.
258  mdarray<T, 1> f_rg_;
259 
260  /// Local set of plane-wave expansion coefficients.
261  mdarray<double_complex, 1> f_pw_local_;
262 
263  /// Storage of the PW coefficients for the FFT transformation.
264  mdarray<double_complex, 1> f_pw_fft_;
265 
266  /// Distribution of G-vectors inside FFT slab.
267  block_data_descriptor gvec_fft_slab_;

271  /* assigment is not allowed */

274  /// Gather plane-wave coefficients for the subsequent FFT call.
275  inline void gather_f_pw_fft()
276  {
277  int rank = fft_->comm().rank() * comm_local_->size() + comm_local_->rank();
278  /* collect scattered PW coefficients */
279  comm_local_->allgather(f_pw_local_.at<CPU>(),
280  gvec_->gvec_count(rank),
281  f_pw_fft_.at<CPU>(),
282  gvec_fft_slab_.counts.data(),
283  gvec_fft_slab_.offsets.data());
284  }

293  Smooth_periodic_function(FFT3D& fft__, Gvec const& gvec__)
294  : fft_(&fft__)
295  , gvec_(&gvec__)
296  , comm_(&gvec__.comm())
297  , comm_local_(&gvec__.comm_ortho_fft())
298  {
299  f_rg_ = mdarray<T, 1>(fft_->local_size());
300  f_pw_fft_ = mdarray<double_complex, 1>(gvec_->partition().gvec_count_fft());
301  f_pw_local_ = mdarray<double_complex, 1>(gvec_->gvec_count(comm_->rank()));
302 
303  int rank = fft_->comm().rank() * comm_local_->size() + comm_local_->rank();
304  if (rank != comm_->rank()) {
305  TERMINATE("wrong order of MPI ranks");
306  }
307 
308  gvec_fft_slab_ = block_data_descriptor(comm_local_->size());
309  for (int i = 0; i < comm_local_->size(); i++) {
310  gvec_fft_slab_.counts[i] = gvec_->gvec_count(fft_->comm().rank() * comm_local_->size() + i);
311  }
312  gvec_fft_slab_.calc_offsets();
313  }

358  void fft_transform(int direction__)
359  {
360  PROFILE("sirius::Smooth_periodic_function::fft_transform");
361 
362  assert(gvec_ != nullptr);
363 
364  switch (direction__) {
365  case 1: {

368  fft_->output(f_rg_.template at<CPU>());

373  fft_->transform<-1>(gvec_->partition(), f_pw_fft_.at<CPU>());

375  int offset = gvec_fft_slab_.offsets[comm_local_->rank()];
376  std::memcpy(f_pw_local_.at<CPU>(), f_pw_fft_.at<CPU>(offset), count * sizeof(double_complex));
377  break;
378  }
379  default: {
380  TERMINATE("wrong fft direction");
381  }
382  }
383  }

388
