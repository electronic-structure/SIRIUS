360  inline void bcast(std::string& str__, int root__) const
361  {
362  int sz;
363  if (rank() == root__) {
364  sz = static_cast<int>(str__.size());
365  }
366  bcast(&sz, 1, root__);
367  char* buf = new char[sz + 1];
368  if (rank() == root__) {
369  std::copy(str__.c_str(), str__.c_str() + sz + 1, buf);
370  }
371  bcast(buf, sz + 1, root__);
372  str__ = std::string(buf);
373  delete[] buf;
374  }
