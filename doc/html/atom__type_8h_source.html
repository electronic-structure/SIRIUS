96 
97  for (size_t order = 0; order < aw_descriptors[l].size(); order++)
98  {
99  radial_function_index_descriptors_.push_back(radial_function_index_descriptor(l, num_rf_[l]));
100  num_rf_[l]++;
101  }
102  }
103 
104  for (int idxlo = 0; idxlo < (int)lo_descriptors.size(); idxlo++)
105  {
106  int l = lo_descriptors[idxlo].l;
107  radial_function_index_descriptors_.push_back(radial_function_index_descriptor(l, num_rf_[l], idxlo));
108  num_rf_[l]++;
109  num_lo_[l]++;
110  }

112  for (int l = 0; l <= lmax_; l++) max_num_rf_ = std::max(max_num_rf_, num_rf_[l]);
113 
114  index_by_l_order_ = mdarray<int, 2>(lmax_ + 1, max_num_rf_);
115 
116  if (lo_descriptors.size())
117  {
118  index_by_idxlo_ = mdarray<int, 1>(lo_descriptors.size());
119  }
120
