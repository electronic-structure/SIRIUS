132  zone,

134  halm_col.at<CPU>(), halm_col.ld(),

274 inline void Band::set_fv_h_o_apw_lo(K_point* kp,
275  Atom_type const& type,
276  Atom const& atom,
277  int ia,
278  mdarray<double_complex, 2>& alm_row, // alm_row comes conjugated
279  mdarray<double_complex, 2>& alm_col,
280  mdarray<double_complex, 2>& h,
281  mdarray<double_complex, 2>& o) const
282 {
283  /* apw-lo block */
284  for (int i = 0; i < kp->num_atom_lo_cols(ia); i++) {
285  int icol = kp->lo_col(ia, i);
286  /* local orbital indices */
287  int l = kp->lo_basis_descriptor_col(icol).l;
288  int lm = kp->lo_basis_descriptor_col(icol).lm;
289  int idxrf = kp->lo_basis_descriptor_col(icol).idxrf;
290  int order = kp->lo_basis_descriptor_col(icol).order;
291  /* loop over apw components */
292  for (int j1 = 0; j1 < type.mt_aw_basis_size(); j1++) {
293  int lm1 = type.indexb(j1).lm;
294  int idxrf1 = type.indexb(j1).idxrf;
295 
296  double_complex zsum = atom.radial_integrals_sum_L3<spin_block_t::nm>(idxrf, idxrf1, gaunt_coefs_->gaunt_vector(lm1, lm));
297 
298  if (std::abs(zsum) > 1e-14) {
299  for (int igkloc = 0; igkloc < kp->num_gkvec_row(); igkloc++) {
300  h(igkloc, kp->num_gkvec_col() + icol) += zsum * alm_row(igkloc, j1);
301  }
302  }
303  }
304 
305  for (int order1 = 0; order1 < type.aw_order(l); order1++) {
306  int xi1 = type.indexb().index_by_lm_order(lm, order1);
307  for (int igkloc = 0; igkloc < kp->num_gkvec_row(); igkloc++) {
308  o(igkloc, kp->num_gkvec_col() + icol) += atom.symmetry_class().o_radial_integral(l, order1, order) *
309  alm_row(igkloc, xi1);
310  }
311  if (ctx_.valence_relativity() == relativity_t::iora) {
312  int idxrf1 = type.indexr().index_by_l_order(l, order1);
313  for (int igkloc = 0; igkloc < kp->num_gkvec_row(); igkloc++) {
314  o(igkloc, kp->num_gkvec_col() + icol) += atom.symmetry_class().o1_radial_integral(idxrf1, idxrf) *
315  alm_row(igkloc, xi1);
316  }
317  }
318  }
319  }
320 
321  std::vector<double_complex> ztmp(kp->num_gkvec_col());
322  /* lo-apw block */
323  for (int i = 0; i < kp->num_atom_lo_rows(ia); i++) {
324  int irow = kp->lo_row(ia, i);
325  /* local orbital indices */
326  int l = kp->lo_basis_descriptor_row(irow).l;
327  int lm = kp->lo_basis_descriptor_row(irow).lm;
328  int idxrf = kp->lo_basis_descriptor_row(irow).idxrf;
329  int order = kp->lo_basis_descriptor_row(irow).order;
330 
331  std::memset(&ztmp[0], 0, kp->num_gkvec_col() * sizeof(double_complex));
332  /* loop over apw components */
333  for (int j1 = 0; j1 < type.mt_aw_basis_size(); j1++) {
334  int lm1 = type.indexb(j1).lm;
335  int idxrf1 = type.indexb(j1).idxrf;
336 
337  double_complex zsum = atom.radial_integrals_sum_L3<spin_block_t::nm>(idxrf1, idxrf, gaunt_coefs_->gaunt_vector(lm, lm1));
338 
339  if (std::abs(zsum) > 1e-14) {
340  for (int igkloc = 0; igkloc < kp->num_gkvec_col(); igkloc++) {
341  ztmp[igkloc] += zsum * alm_col(igkloc, j1);
342  }
343  }
344  }
345 
346  for (int igkloc = 0; igkloc < kp->num_gkvec_col(); igkloc++) {
347  h(irow + kp->num_gkvec_row(), igkloc) += ztmp[igkloc];
348  }
349 
350  for (int order1 = 0; order1 < type.aw_order(l); order1++) {
351  int xi1 = type.indexb().index_by_lm_order(lm, order1);
352  for (int igkloc = 0; igkloc < kp->num_gkvec_col(); igkloc++) {
353  o(irow + kp->num_gkvec_row(), igkloc) += atom.symmetry_class().o_radial_integral(l, order, order1) *
354  alm_col(igkloc, xi1);
355  }
356  if (ctx_.valence_relativity() == relativity_t::iora) {
357  int idxrf1 = type.indexr().index_by_l_order(l, order1);
358  for (int igkloc = 0; igkloc < kp->num_gkvec_col(); igkloc++) {
359  o(irow + kp->num_gkvec_row(), igkloc) += atom.symmetry_class().o1_radial_integral(idxrf, idxrf1) *
360  alm_col(igkloc, xi1);
361  }
362  }
363  }
364  }

367 inline void Band::set_fv_h_o_it(K_point* kp,
368  Potential const& potential__,
369  mdarray<double_complex, 2>& h,
370  mdarray<double_complex, 2>& o) const
371 {
372  PROFILE("sirius::Band::set_fv_h_o_it");
373 
374  //#ifdef __PRINT_OBJECT_CHECKSUM
375  //double_complex z1 = mdarray<double_complex, 1>(&effective_potential->f_pw(0), ctx_.gvec().num_gvec()).checksum();
376  //DUMP("checksum(veff_pw): %18.10f %18.10f", std::real(z1), std::imag(z1));
377  //#endif
378 
379  double sq_alpha_half = 0.5 * std::pow(speed_of_light, -2);
380 
381  #pragma omp parallel for default(shared)
382  for (int igk_col = 0; igk_col < kp->num_gkvec_col(); igk_col++) {
383  int ig_col = kp->igk_col(igk_col);
384  auto gvec_col = kp->gkvec().gvec(ig_col);
385  auto gkvec_col_cart = kp->gkvec().gkvec_cart(ig_col);
386  for (int igk_row = 0; igk_row < kp->num_gkvec_row(); igk_row++) {
387  int ig_row = kp->igk_row(igk_row);
388  auto gvec_row = kp->gkvec().gvec(ig_row);
389  auto gkvec_row_cart = kp->gkvec().gkvec_cart(ig_row);
390  int ig12 = ctx_.gvec().index_g12(gvec_row, gvec_col);
391  /* pw kinetic energy */
392  double t1 = 0.5 * (gkvec_row_cart * gkvec_col_cart);
393 
394  h(igk_row, igk_col) += potential__.veff_pw(ig12);
395  o(igk_row, igk_col) += ctx_.step_function().theta_pw(ig12);
396 
397  if (ctx_.valence_relativity() == relativity_t::none) {
398  h(igk_row, igk_col) += t1 * ctx_.step_function().theta_pw(ig12);
399  } else {
400  h(igk_row, igk_col) += t1 * potential__.rm_inv_pw(ig12);
401  }
402  if (ctx_.valence_relativity() == relativity_t::iora) {
403  o(igk_row, igk_col) += t1 * sq_alpha_half * potential__.rm2_inv_pw(ig12);
404  }
405  }
406  }
407 }

409 inline void Band::set_fv_h_o_lo_lo(K_point* kp,
410  mdarray<double_complex, 2>& h,
411  mdarray<double_complex, 2>& o) const
412 {
413  PROFILE("sirius::Band::set_fv_h_o_lo_lo");
414 
415  /* lo-lo block */
416  #pragma omp parallel for default(shared)
417  for (int icol = 0; icol < kp->num_lo_col(); icol++) {
418  int ia = kp->lo_basis_descriptor_col(icol).ia;
419  int lm2 = kp->lo_basis_descriptor_col(icol).lm;
420  int idxrf2 = kp->lo_basis_descriptor_col(icol).idxrf;
421 
422  for (int irow = 0; irow < kp->num_lo_row(); irow++) {
423  /* lo-lo block is diagonal in atom index */
424  if (ia == kp->lo_basis_descriptor_row(irow).ia) {
425  auto& atom = unit_cell_.atom(ia);
426  int lm1 = kp->lo_basis_descriptor_row(irow).lm;
427  int idxrf1 = kp->lo_basis_descriptor_row(irow).idxrf;
428 
429  h(kp->num_gkvec_row() + irow, kp->num_gkvec_col() + icol) +=
430  atom.radial_integrals_sum_L3<spin_block_t::nm>(idxrf1, idxrf2, gaunt_coefs_->gaunt_vector(lm1, lm2));
431 
432  if (lm1 == lm2) {

434  int order1 = kp->lo_basis_descriptor_row(irow).order;

562 template <spin_block_t sblock>
563 void Band::set_h_it(K_point* kp, Periodic_function<double>* effective_potential,
564  Periodic_function<double>* effective_magnetic_field[3], mdarray<double_complex, 2>& h) const
565 {
566  PROFILE("sirius::Band::set_h_it");
567 
568  STOP(); // effective potential is now stored in the veff_pw_ auxiliary array. Fix this.
569 
570  #pragma omp parallel for default(shared)
571  for (int igk_col = 0; igk_col < kp->num_gkvec_col(); igk_col++) {
572  auto gkvec_col_cart = kp->gkvec().gkvec_cart(kp->igk_col(igk_col));
573  auto gvec_col = kp->gkvec().gvec(kp->igk_col(igk_col));
574  for (int igk_row = 0; igk_row < kp->num_gkvec_row(); igk_row++) {
575  auto gkvec_row_cart = kp->gkvec().gkvec_cart(kp->igk_row(igk_row));
576  auto gvec_row = kp->gkvec().gvec(kp->igk_row(igk_row));
577 
578  int ig12 = ctx_.gvec().index_g12(gvec_row, gvec_col);
579 
580  /* pw kinetic energy */
581  double t1 = 0.5 * (gkvec_row_cart * gkvec_col_cart);
582 
583  switch (sblock) {
584  case spin_block_t::nm: {
585  h(igk_row, igk_col) += (effective_potential->f_pw(ig12) + t1 * ctx_.step_function().theta_pw(ig12));
586  break;
587  }
588  case spin_block_t::uu: {
589  h(igk_row, igk_col) += (effective_potential->f_pw(ig12) + effective_magnetic_field[0]->f_pw(ig12) +
590  t1 * ctx_.step_function().theta_pw(ig12));
591  break;
592  }
593  case spin_block_t::dd: {
594  h(igk_row, igk_col) += (effective_potential->f_pw(ig12) - effective_magnetic_field[0]->f_pw(ig12) +
595  t1 * ctx_.step_function().theta_pw(ig12));
596  break;
597  }
598  case spin_block_t::ud: {
599  h(igk_row, igk_col) += (effective_magnetic_field[1]->f_pw(ig12) -
600  double_complex(0, 1) * effective_magnetic_field[2]->f_pw(ig12));
601  break;
602  }
603  case spin_block_t::du: {
604  h(igk_row, igk_col) += (effective_magnetic_field[1]->f_pw(ig12) +
605  double_complex(0, 1) * effective_magnetic_field[2]->f_pw(ig12));
606  break;
607  }
608  }
609  }
610  }
