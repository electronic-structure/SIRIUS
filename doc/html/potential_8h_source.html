168  /// Solve Poisson equation for a single atom.
169  template <bool free_atom>
170  inline std::vector<double>
171  poisson_vmt(Atom const& atom__,
172  Spheric_function<function_domain_t::spectral, double> const& rho_mt__,
173  Spheric_function<function_domain_t::spectral, double>& vha_mt__) const
174  {
175  int lmmax_rho = rho_mt__.angular_domain_size();
176  int lmmax_pot = vha_mt__.angular_domain_size();
177  assert((int)l_by_lm_.size() >= lmmax_rho);
178  if (lmmax_rho > ctx_.lmmax_rho()) {
179  std::stringstream s;
180  s << "wrong angular size of rho_mt for atom of " << atom__.type().symbol() << std::endl
181  << " lmmax_rho: " << lmmax_rho << std::endl
182  << " ctx.lmmax_rho(): " << ctx_.lmmax_rho();
183  TERMINATE(s);
184  }
185  std::vector<double> qmt(ctx_.lmmax_rho(), 0);
186 
187  double R = atom__.mt_radius();
188  int nmtp = atom__.num_mt_points();
189 
190  #pragma omp parallel
191  {
192  std::vector<double> g1;
193  std::vector<double> g2;
194 
195  #pragma omp for
196  for (int lm = 0; lm < lmmax_rho; lm++) {
197  int l = l_by_lm_[lm];
198 
199  auto rholm = rho_mt__.component(lm);
200 
201  /* save multipole moment */
202  qmt[lm] = rholm.integrate(g1, l + 2);
203 
204  if (lm < lmmax_pot) {
205 
206  rholm.integrate(g2, 1 - l);
207 
208  double d2 = fourpi / double(2 * l + 1);
209  double vlm;
210 
211  for (int ir = 0; ir < nmtp; ir++) {
212  double r = atom__.radial_grid(ir);
213 
214  if (free_atom) {
215  vlm = g1[ir] / std::pow(r, l + 1) + (g2.back() - g2[ir]) * std::pow(r, l);
216  } else {
217  double d1 = 1.0 / std::pow(R, 2 * l + 1);
218  vlm = (1.0 - std::pow(r / R, 2 * l + 1)) * g1[ir] / std::pow(r, l + 1) +
219  (g2.back() - g2[ir]) * std::pow(r, l) - (g1.back() - g1[ir]) * std::pow(r, l) * d1;
220  }
221  vha_mt__(lm, ir) = vlm * d2;
222  }
223  }
224  }
225  }
226  if (!free_atom) {
227  /* constant part of nuclear potential -z*(1/r - 1/R) */
228  for (int ir = 0; ir < nmtp; ir++) {
229  vha_mt__(0, ir) += atom__.zn() / R / y00;
230  }
231  }
232  /* nuclear multipole moment */
233  qmt[0] -= atom__.zn() * y00;
234 
235  return std::move(qmt);
236  }

268  /// Generate local part of pseudo potential.
269  /** Total local potential is a lattice sum:
270  * \f[
271  * V({\bf r}) = \sum_{{\bf T},\alpha} V_{\alpha}({\bf r} - {\bf T} - {\bf \tau}_{\alpha})
272  * \f]
273  * We want to compute it's plane-wave expansion coefficients:
274  * \f[
275  * V({\bf G}) = \frac{1}{V} \int e^{-i{\bf Gr}} V({\bf r}) d{\bf r} =
276  * \frac{1}{V} \sum_{{\bf T},\alpha} \int e^{-i{\bf Gr}}V_{\alpha}({\bf r} - {\bf T} - {\bf \tau}_{\alpha})d{\bf r}
277  * \f]
278  * Standard change of variables: \f$ {\bf r}' = {\bf r} - {\bf T} - {\bf \tau}_{\alpha},\; {\bf r} = {\bf r}' + {\bf T} + {\bf \tau}_{\alpha} \f$
279  * leads to:
280  * \f[
281  * V({\bf G}) = \frac{1}{V} \sum_{{\bf T},\alpha} \int e^{-i{\bf G}({\bf r}' + {\bf T} + {\bf \tau}_{\alpha})}V_{\alpha}({\bf r}')d{\bf r'} =
282  * \frac{N}{V} \sum_{\alpha} \int e^{-i{\bf G}({\bf r}' + {\bf \tau}_{\alpha})}V_{\alpha}({\bf r}')d{\bf r'} =
283  * \frac{1}{\Omega} \sum_{\alpha} e^{-i {\bf G} {\bf \tau}_{\alpha} } \int e^{-i{\bf G}{\bf r}}V_{\alpha}({\bf r})d{\bf r}
284  * \f]
285  * Using the well-known expansion of a plane wave in terms of spherical Bessel functions:
286  * \f[
287  * e^{i{\bf G}{\bf r}}=4\pi \sum_{\ell m} i^\ell j_{\ell}(Gr)Y_{\ell m}^{*}({\bf \hat G})Y_{\ell m}({\bf \hat r})
288  * \f]
289  * and remembering that for \f$ \ell = 0 \f$ (potential is sphericla) \f$ j_{0}(x) = \sin(x) / x \f$ we have:
290  * \f[
291  * V_{\alpha}({\bf G}) = \int V_{\alpha}(r) 4\pi \frac{\sin(Gr)}{Gr} Y^{*}_{00} Y_{00} r^2 \sin(\theta) dr d \phi d\theta =
292  * 4\pi \int V_{\alpha}(r) \frac{\sin(Gr)}{Gr} r^2 dr
293  * \f]
294  * The tricky part comes next: \f$ V_{\alpha}({\bf r}) \f$ is a long-range potential -- it decays slowly as
295  * \f$ -Z_{\alpha}^{p}/r \f$ and the straightforward integration with sperical Bessel function is numerically
296  * unstable. For \f$ {\bf G} = 0 \f$ an extra term \f$ Z_{\alpha}^p/r \f$, corresponding to the potential of
297  * pseudo-ion, is added to and removed from the local part of the atomic pseudopotential \f$ V_{\alpha}({\bf r}) \f$:
298  * \f[
299  * V_{\alpha}({\bf G} = 0) = \int V_{\alpha}({\bf r})d{\bf r} \Rightarrow
300  * 4\pi \int \Big( V_{\alpha}(r) + \frac{Z_{\alpha}^p}{r} \Big) r^2 dr -
301  * 4\pi \int \Big( \frac{Z_{\alpha}^p}{r} \Big) r^2 dr
302  * \f]
303  * Second term corresponds to the average electrostatic potential of ions and it is ignored
304  * (like the \f$ {\bf G} = 0 \f$ term in the Hartree potential of electrons).
305  * For \f$ G \ne 0 \f$ the following trick is done: \f$ Z_{\alpha}^p {\rm erf}(r) / r \f$ is added to and
306  * removed from \f$ V_{\alpha}(r) \f$. The idea is to make potential decay quickly and then take the extra
307  * contribution analytically. We have:
308  * \f[
309  * V_{\alpha}({\bf G}) = 4\pi \int \Big(V_{\alpha}(r) + Z_{\alpha}^p \frac{{\rm erf}(r)} {r} -
310  * Z_{\alpha}^p \frac{{\rm erf}(r)}{r}\Big) \frac{\sin(Gr)}{Gr} r^2 dr
311  * \f]
312  * Analytical contribution from the error function is computed using the 1D Fourier transform in complex plane:
313  * \f[
314  * \frac{1}{\sqrt{2 \pi}} \int_{-\infty}^{\infty} {\rm erf}(t) e^{i\omega t} dt =
315  * \frac{i e^{-\frac{\omega ^2}{4}} \sqrt{\frac{2}{\pi }}}{\omega }
316  * \f]
317  * from which we immediately get
318  * \f[
319  * \int_{0}^{\infty} \frac{{\rm erf}(r)}{r} \frac{\sin(Gr)}{Gr} r^2 dr = \frac{e^{-\frac{G^2}{4}}}{G^2}
320  * \f]
321  * The final expression for the local potential radial integrals for \f$ G \ne 0 \f$ take the following form:
322  * \f[
323  * 4\pi \int \Big(V_{\alpha}(r) r + Z_{\alpha}^p {\rm erf}(r) \Big) \frac{\sin(Gr)}{G} dr - \frac{e^{-\frac{G^2}{4}}}{G^2}
324  * \f]
325  */

547  /// Poisson solver.
548  /** Detailed explanation is available in:
549  * - Weinert, M. (1981). Solution of Poisson's equation: beyond Ewald-type methods.
550  * Journal of Mathematical Physics, 22(11), 24332439. doi:10.1063/1.524800
551  * - Classical Electrodynamics Third Edition by J. D. Jackson.
552  *
553  * Solution of Poisson's equation for the muffin-tin geometry is carried out in several steps:
554  * - True multipole moments \f$ q_{\ell m}^{\alpha} \f$ of the muffin-tin charge density are computed.
555  * - Pseudocharge density is introduced. Pseudocharge density coincides with the true charge density
556  * in the interstitial region and it's multipole moments inside muffin-tin spheres coincide with the
557  * true multipole moments.
558  * - Poisson's equation for the pseudocharge density is solved in the plane-wave domain. It gives the
559  * correct interstitial potential and correct muffin-tin boundary values.
560  * - Finally, muffin-tin part of potential is found by solving Poisson's equation in spherical coordinates
561  * with Dirichlet boundary conditions.
562  *
563  * We start by computing true multipole moments of the charge density inside the muffin-tin spheres:
564  * \f[
565  * q_{\ell m}^{\alpha} = \int Y_{\ell m}^{*}(\hat {\bf r}) r^{\ell} \rho({\bf r}) d {\bf r} =
566  * \int \rho_{\ell m}^{\alpha}(r) r^{\ell + 2} dr
567  * \f]
568  * and for the nucleus with charge density \f$ \rho(r, \theta, \phi) = -\frac{Z \delta(r)}{4 \pi r^2} \f$:
569  * \f[
570  * q_{00}^{\alpha} = \int Y_{0 0} \frac{-Z_{\alpha} \delta(r)}{4 \pi r^2} r^2 \sin \theta dr d\phi d\theta =
571  * -Z_{\alpha} Y_{00}
572  * \f]
573  *
574  * Now we need to get the multipole moments of the interstitial charge density \f$ \rho^{I}({\bf r}) \f$ inside
575  * muffin-tin spheres. We need this in order to estimate the amount of pseudocharge to be added to
576  * \f$ \rho^{I}({\bf r}) \f$ to get the pseudocharge multipole moments equal to the true multipole moments.
577  * We want to compute
578  * \f[
579  * q_{\ell m}^{I,\alpha} = \int Y_{\ell m}^{*}(\hat {\bf r}) r^{\ell} \rho^{I}({\bf r}) d {\bf r}
580  * \f]
581  * where
582  * \f[
583  * \rho^{I}({\bf r}) = \sum_{\bf G}e^{i{\bf Gr}} \rho({\bf G})
584  * \f]
585  *
586  * Recall the spherical plane wave expansion:
587  * \f[
588  * e^{i{\bf G r}}=4\pi e^{i{\bf G r}_{\alpha}} \sum_{\ell m} i^\ell
589  * j_{\ell}(G|{\bf r}-{\bf r}_{\alpha}|)
590  * Y_{\ell m}^{*}({\bf \hat G}) Y_{\ell m}(\widehat{{\bf r}-{\bf r}_{\alpha}})
591  * \f]
592  * Multipole moments of each plane-wave are computed as:
593  * \f[
594  * q_{\ell m}^{\alpha}({\bf G}) = 4 \pi e^{i{\bf G r}_{\alpha}} Y_{\ell m}^{*}({\bf \hat G}) i^{\ell}
595  * \int_{0}^{R} j_{\ell}(Gr) r^{\ell + 2} dr = 4 \pi e^{i{\bf G r}_{\alpha}} Y_{\ell m}^{*}({\bf \hat G}) i^{\ell}
596  * \left\{\begin{array}{ll} \frac{R^{\ell + 2} j_{\ell + 1}(GR)}{G} & G \ne 0 \\
597  * \frac{R^3}{3} \delta_{\ell 0} & G = 0 \end{array} \right.
598  * \f]
599  *
600  * Final expression for the muffin-tin multipole moments of the interstitial charge denisty:
601  * \f[
602  * q_{\ell m}^{I,\alpha} = \sum_{\bf G}\rho({\bf G}) q_{\ell m}^{\alpha}({\bf G})
603  * \f]
604  *
605  * Now we are going to modify interstitial charge density inside the muffin-tin region in order to
606  * get the true multipole moments. We will add a pseudodensity of the form:
607  * \f[
608  * P({\bf r}) = \sum_{\ell m} p_{\ell m}^{\alpha} Y_{\ell m}(\hat {\bf r}) r^{\ell} \left(1-\frac{r^2}{R^2}\right)^n
609  * \f]
610  * Radial functions of the pseudodensity are chosen in a special way. First, they produce a confined and
611  * smooth functions inside muffin-tins and second (most important) plane-wave coefficients of the
612  * pseudodensity can be computed analytically. Let's find the relation between \f$ p_{\ell m}^{\alpha} \f$
613  * coefficients and true and interstitial multipole moments first. We are searching for the pseudodensity which restores
614  * the true multipole moments:
615  * \f[
616  * \int Y_{\ell m}^{*}(\hat {\bf r}) r^{\ell} \Big(\rho^{I}({\bf r}) + P({\bf r})\Big) d {\bf r} = q_{\ell m}^{\alpha}
617  * \f]
618  * Then
619  * \f[
620  * p_{\ell m}^{\alpha} = \frac{q_{\ell m}^{\alpha} - q_{\ell m}^{I,\alpha}}
621  * {\int r^{2 \ell + 2} \left(1-\frac{r^2}{R^2}\right)^n dr} =
622  * (q_{\ell m}^{\alpha} - q_{\ell m}^{I,\alpha}) \frac{2 \Gamma(5/2 + \ell + n)}{R^{2\ell + 3}\Gamma(3/2 + \ell) \Gamma(n + 1)}
623  * \f]
624  *
625  * Now let's find the plane-wave coefficients of \f$ P({\bf r}) \f$ inside each muffin-tin:
626  * \f[
627  * P^{\alpha}({\bf G}) = \frac{4\pi e^{-i{\bf G r}_{\alpha}}}{\Omega} \sum_{\ell m} (-i)^{\ell} Y_{\ell m}({\bf \hat G})
628  * p_{\ell m}^{\alpha} \int_{0}^{R} j_{\ell}(G r) r^{\ell} \left(1-\frac{r^2}{R^2}\right)^n r^2 dr
629  * \f]
630  *
631  * Integral of the spherical Bessel function with the radial pseudodensity component is taken analytically:
632  * \f[
633  * \int_{0}^{R} j_{\ell}(G r) r^{\ell} \left(1-\frac{r^2}{R^2}\right)^n r^2 dr =
634  * 2^n R^{\ell + 3} (GR)^{-n - 1} \Gamma(n + 1) j_{n + \ell + 1}(GR)
635  * \f]
636  *
637  * The final expression for the pseudodensity plane-wave component is:
638  * \f[
639  * P^{\alpha}({\bf G}) = \frac{4\pi e^{-i{\bf G r}_{\alpha}}}{\Omega} \sum_{\ell m} (-i)^{\ell} Y_{\ell m}({\bf \hat G})
640  * (q_{\ell m}^{\alpha} - q_{\ell m}^{I,\alpha}) \Big( \frac{2}{GR} \Big)^{n+1}
641  * \frac{ \Gamma(5/2 + n + \ell) } {R^{\ell} \Gamma(3/2+\ell)}
642  * \f]
643  *
644  * For \f$ G=0 \f$ only \f$ \ell = 0 \f$ contribution survives:
645  * \f[
646  * P^{\alpha}({\bf G}=0) = \frac{4\pi}{\Omega} Y_{00} (q_{00}^{\alpha} - q_{00}^{I,\alpha})
647  * \f]
648  *
649  * We can now sum the contributions from all muffin-tin spheres and obtain a modified charge density,
650  * which is equal to the exact charge density in the interstitial region and which has correct multipole
651  * moments inside muffin-tin spheres:
652  * \f[
653  * \tilde \rho({\bf G}) = \rho({\bf G}) + \sum_{\alpha} P^{\alpha}({\bf G})
654  * \f]
655  * This density is used to solve the Poisson's equation in the plane-wave domain:
656  * \f[
657  * V_{H}({\bf G}) = \frac{4 \pi \tilde \rho({\bf G})}{G^2}
658  * \f]
659  * The potential is correct in the interstitial region and also on the muffin-tin surface. We will use
660  * it to find the boundary conditions for the potential inside the muffin-tins. Using spherical
661  * plane-wave expansion we get:
662  * \f[
663  * V^{\alpha}_{\ell m}(R) = \sum_{\bf G} V_{H}({\bf G})
664  * 4\pi e^{i{\bf G r}_{\alpha}} i^\ell
665  * j_{\ell}^{\alpha}(GR) Y_{\ell m}^{*}({\bf \hat G})
666  * \f]
667  *
668  * As soon as the muffin-tin boundary conditions for the potential are known, we can find the potential
669  * inside spheres using Dirichlet Green's function:
670  * \f[
671  * V({\bf x}) = \int \rho({\bf x'})G_D({\bf x},{\bf x'}) d{\bf x'} - \frac{1}{4 \pi} \int_{S} V({\bf x'})
672  * \frac{\partial G_D}{\partial n'} d{\bf S'}
673  * \f]
674  * where Dirichlet Green's function for the sphere is defined as:
675  * \f[
676  * G_D({\bf x},{\bf x'}) = 4\pi \sum_{\ell m} \frac{Y_{\ell m}^{*}({\bf \hat x'})
677  * Y_{\ell m}(\hat {\bf x})}{2\ell + 1}
678  * \frac{r_{<}^{\ell}}{r_{>}^{\ell+1}}\Biggl(1 - \Big( \frac{r_{>}}{R} \Big)^{2\ell + 1} \Biggr)
679  * \f]
680  * and it's normal derivative at the surface is equal to:
681  * \f[
682  * \frac{\partial G_D}{\partial n'} = -\frac{4 \pi}{R^2} \sum_{\ell m} \Big( \frac{r}{R} \Big)^{\ell}
683  * Y_{\ell m}^{*}({\bf \hat x'}) Y_{\ell m}(\hat {\bf x})
684  * \f]
685  */
686  inline void poisson(Periodic_function<double>* rho, Periodic_function<double>* vh);

688  /// Generate XC potential and energy density
689  /** In case of spin-unpolarized GGA the XC potential has the following expression:
690  * \f[
691  * V_{XC}({\bf r}) = \frac{\partial}{\partial \rho} \varepsilon_{xc}(\rho, \nabla \rho) -
692  * \nabla \frac{\partial}{\partial (\nabla \rho)} \varepsilon_{xc}(\rho, \nabla \rho)
693  * \f]
694  * LibXC packs the gradient information into the so-called \a sigma array:
695  * \f[
696  * \sigma = \nabla \rho \nabla \rho
697  * \f]
698  * Changing variables in \f$ V_{XC} \f$ expression gives:
699  * \f{eqnarray*}{
700  * V_{XC}({\bf r}) &=& \frac{\partial}{\partial \rho} \varepsilon_{xc}(\rho, \sigma) -
701  * \nabla \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma}
702  * \frac{\partial \sigma}{ \partial (\nabla \rho)} \\
703  * &=& \frac{\partial}{\partial \rho} \varepsilon_{xc}(\rho, \sigma) -
704  * 2 \nabla \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma} \nabla \rho -
705  * 2 \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma} \nabla^2 \rho
706  * \f}
707  * The following sequence of functions must be computed:
708  * - density on the real space grid
709  * - gradient of density (in spectral representation)
710  * - gradient of density on the real space grid
711  * - laplacian of density (in spectral representation)
712  * - laplacian of density on the real space grid
713  * - \a sigma array
714  * - a call to Libxc must be performed \a sigma derivatives must be obtained
715  * - \f$ \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma} \f$ in spectral representation
716  * - gradient of \f$ \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma} \f$ in spectral representation
717  * - gradient of \f$ \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma} \f$ on the real space grid
718  *
719  * Expression for spin-polarized potential has a bit more complicated form:
720  * \f{eqnarray*}
721  * V_{XC}^{\gamma} &=& \frac{\partial \varepsilon_{xc}}{\partial \rho_{\gamma}} - \nabla
722  * \Big( 2 \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \gamma}} \nabla \rho_{\gamma} +
723  * \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \delta}} \nabla \rho_{\delta} \Big) \\
724  * &=& \frac{\partial \varepsilon_{xc}}{\partial \rho_{\gamma}}
725  * -2 \nabla \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \gamma}} \nabla \rho_{\gamma}
726  * -2 \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \gamma}} \nabla^2 \rho_{\gamma}
727  * - \nabla \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \delta}} \nabla \rho_{\delta}
728  * - \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \delta}} \nabla^2 \rho_{\delta}
729  * \f}
730  * In magnetic case the "up" and "dn" density and potential decomposition is used. Using the fact that the
731  * effective magnetic field is parallel to magnetization at each point in space, we can write the coupling
732  * of density and magnetization with XC potential and XC magentic field as:
733  * \f[
734  * V_{xc}({\bf r}) \rho({\bf r}) + {\bf B}_{xc}({\bf r}){\bf m}({\bf r}) =
735  * V_{xc}({\bf r}) \rho({\bf r}) + {\rm B}_{xc}({\bf r}) {\rm m}({\bf r}) =
736  * V^{\uparrow}({\bf r})\rho^{\uparrow}({\bf r}) + V^{\downarrow}({\bf r})\rho^{\downarrow}({\bf r})
737  * \f]
738  * where
739  * \f{eqnarray*}{
740  * \rho^{\uparrow}({\bf r}) &=& \frac{1}{2}\Big( \rho({\bf r}) + {\rm m}({\bf r}) \Big) \\
741  * \rho^{\downarrow}({\bf r}) &=& \frac{1}{2}\Big( \rho({\bf r}) - {\rm m}({\bf r}) \Big)
742  * \f}
743  * and
744  * \f{eqnarray*}{
745  * V^{\uparrow}({\bf r}) &=& V_{xc}({\bf r}) + {\rm B}_{xc}({\bf r}) \\
746  * V^{\downarrow}({\bf r}) &=& V_{xc}({\bf r}) - {\rm B}_{xc}({\bf r})
747  * \f}
748  */
749  void xc(Periodic_function<double>* rho,

751  Periodic_function<double>* vxc,
752  Periodic_function<double>* bxc[3],
753  Periodic_function<double>* exc);

870  /// Calculate D operator from potential and augmentation charge.
871  /** The following real symmetric matrix is computed:
872  * \f[
873  * D_{\xi \xi'}^{\alpha} = \int V({\bf r}) Q_{\xi \xi'}^{\alpha}({\bf r}) d{\bf r}
874  * \f]
875  * In the plane-wave domain this integrals transform into sum over Fourier components:
876  * \f[
877  * D_{\xi \xi'}^{\alpha} = \sum_{\bf G} \langle V |{\bf G}\rangle \langle{\bf G}|Q_{\xi \xi'}^{\alpha} \rangle =
878  * \sum_{\bf G} V^{*}({\bf G}) e^{-i{\bf r}_{\alpha}{\bf G}} Q_{\xi \xi'}^{A}({\bf G}) =
879  * \sum_{\bf G} Q_{\xi \xi'}^{A}({\bf G}) \tilde V_{\alpha}^{*}({\bf G})
880  * \f]
881  * where \f$ \alpha \f$ is the atom, \f$ A \f$ is the atom type and
882  * \f[
883  * \tilde V_{\alpha}({\bf G}) = e^{i{\bf r}_{\alpha}{\bf G}} V({\bf G})
884  * \f]
885  * Both \f$ V({\bf r}) \f$ and \f$ Q({\bf r}) \f$ functions are real and the following condition is fulfilled:
886  * \f[
887  * \tilde V_{\alpha}({\bf G}) = \tilde V_{\alpha}^{*}(-{\bf G})
888  * \f]
889  * \f[
890  * Q_{\xi \xi'}({\bf G}) = Q_{\xi \xi'}^{*}(-{\bf G})
891  * \f]
892  * In the sum over plane-wave coefficients the \f$ {\bf G} \f$ and \f$ -{\bf G} \f$ contributions will give:
893  * \f[
894  * Q_{\xi \xi'}^{A}({\bf G}) \tilde V_{\alpha}^{*}({\bf G}) + Q_{\xi \xi'}^{A}(-{\bf G}) \tilde V_{\alpha}^{*}(-{\bf G}) =
895  * 2 \Re \Big( Q_{\xi \xi'}^{A}({\bf G}) \Big) \Re \Big( \tilde V_{\alpha}^{*}({\bf G}) \Big) +
896  * 2 \Im \Big( Q_{\xi \xi'}^{A}({\bf G}) \Big) \Im \Big( \tilde V_{\alpha}^{*}({\bf G}) \Big)
897  * \f]
898  * This allows the use of a <b>dgemm</b> instead of a <b>zgemm</b> when \f$ D_{\xi \xi'}^{\alpha} \f$ matrix
899  * is calculated for all atoms of the same type.
900  */
