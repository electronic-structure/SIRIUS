99  /// Generate rotation matrix from three Euler angles
100  /** Euler angles \f$ \alpha, \beta, \gamma \f$ define the general rotation as three consecutive rotations:
101  * - about \f$ \hat e_z \f$ through the angle \f$ \gamma \f$ (\f$ 0 \le \gamma < 2\pi \f$)
102  * - about \f$ \hat e_y \f$ through the angle \f$ \beta \f$ (\f$ 0 \le \beta \le \pi \f$)
103  * - about \f$ \hat e_z \f$ through the angle \f$ \alpha \f$ (\f$ 0 \le \gamma < 2\pi \f$)
104  *
105  * The total rotation matrix is defined as a product of three rotation matrices:
106  * \f[
107  * R(\alpha, \beta, \gamma) =
108  * \left( \begin{array}{ccc} \cos(\alpha) & -\sin(\alpha) & 0 \\
109  * \sin(\alpha) & \cos(\alpha) & 0 \\
110  * 0 & 0 & 1 \end{array} \right)
111  * \left( \begin{array}{ccc} \cos(\beta) & 0 & \sin(\beta) \\
112  * 0 & 1 & 0 \\
113  * -\sin(\beta) & 0 & \cos(\beta) \end{array} \right)
114  * \left( \begin{array}{ccc} \cos(\gamma) & -\sin(\gamma) & 0 \\
115  * \sin(\gamma) & \cos(\gamma) & 0 \\
116  * 0 & 0 & 1 \end{array} \right) =
117  * \left( \begin{array}{ccc} \cos(\alpha) \cos(\beta) \cos(\gamma) - \sin(\alpha) \sin(\gamma) &
118  * -\sin(\alpha) \cos(\gamma) - \cos(\alpha) \cos(\beta) \sin(\gamma) &
119  * \cos(\alpha) \sin(\beta) \\
120  * \sin(\alpha) \cos(\beta) \cos(\gamma) + \cos(\alpha) \sin(\gamma) &
121  * \cos(\alpha) \cos(\gamma) - \sin(\alpha) \cos(\beta) \sin(\gamma) &
122  * \sin(\alpha) \sin(\beta) \\
123  * -\sin(\beta) \cos(\gamma) &
124  * \sin(\beta) \sin(\gamma) &
125  * \cos(\beta) \end{array} \right)
126  * \f]
127  */
128  matrix3d<double> rot_mtrx_cart(vector3d<double> euler_angles__) const;

204  /// Symmetrize scalar function.
205  /** The following operation is performed:
206  * \f[
207  * f({\bf x}) = \frac{1}{N_{sym}} \sum_{{\bf \hat P}} f({\bf \hat P x})
208  * \f]
209  * For the function expanded in plane-waves we have:
210  * \f[
211  * f({\bf x}) = \frac{1}{N_{sym}} \sum_{{\bf \hat P}} \sum_{\bf G} e^{i{\bf G \hat P x}} f({\bf G})
212  * = \frac{1}{N_{sym}} \sum_{{\bf \hat P}} \sum_{\bf G} e^{i{\bf G (Rx + t)}} f({\bf G})
213  * = \frac{1}{N_{sym}} \sum_{{\bf \hat P}} \sum_{\bf G} e^{i{\bf G t}} e^{i{\bf G Rx}} f({\bf G})
214  * \f]
215  * Now we do a mapping \f$ {\bf GR} \rightarrow \tilde {\bf G} \f$ and find expansion coefficients of the
216  * symmetry transformed function:
217  * \f[
218  * f(\tilde{\bf G}) = e^{i{\bf G t}} f({\bf G})
219  * \f]
220  */

262 inline Symmetry::Symmetry(matrix3d<double>& lattice_vectors__,
263  int num_atoms__,
264  mdarray<double, 2>& positions__,
265  mdarray<double, 2>& spins__,
266  std::vector<int>& types__,
267  double tolerance__)
268  : lattice_vectors_(lattice_vectors__)
269  , num_atoms_(num_atoms__)
270  , types_(types__)
271  , tolerance_(tolerance__)
272 {
273  PROFILE("sirius::Symmetry::Symmetry");
274 
275  if (lattice_vectors__.det() < 0) {
276  std::stringstream s;
277  s << "spglib requires positive determinant for a matrix of lattice vectors";
278  TERMINATE(s);
279  }
280 
281  double lattice[3][3];
282  for (int i: {0, 1, 2}) {
283  for (int j: {0, 1, 2}) {
284  lattice[i][j] = lattice_vectors_(i, j);
285  }
286  }
287  positions_ = mdarray<double, 2>(3, num_atoms_);
288  for (int ia = 0; ia < num_atoms_; ia++) {
289  for (int x: {0, 1, 2}) {
290  positions_(x, ia) = positions__(x, ia);
291  }
292  }
293 
294  spg_dataset_ = spg_get_dataset(lattice, (double(*)[3])&positions_(0, 0), &types_[0], num_atoms_, tolerance_);
295  if (spg_dataset_ == NULL) {
296  TERMINATE("spg_get_dataset() returned NULL");
297  }
298 
299  if (spg_dataset_->spacegroup_number == 0) {
300  TERMINATE("spg_get_dataset() returned 0 for the space group");
301  }
302 
303  if (spg_dataset_->n_atoms != num_atoms__) {
304  std::stringstream s;
305  s << "spg_get_dataset() returned wrong number of atoms (" << spg_dataset_->n_atoms << ")" << std::endl
306  << "expected number of atoms is " << num_atoms__;
307  TERMINATE(s);
308  }
309 
310  inverse_lattice_vectors_ = inverse(lattice_vectors_);
311 
312  for (int isym = 0; isym < spg_dataset_->n_operations; isym++) {
313  space_group_symmetry_descriptor sym_op;
314 
315  sym_op.R = matrix3d<int>(spg_dataset_->rotations[isym]);
316  sym_op.t = vector3d<double>(spg_dataset_->translations[isym][0],
317  spg_dataset_->translations[isym][1],
318  spg_dataset_->translations[isym][2]);
319  int p = sym_op.R.det();
320  if (!(p == 1 || p == -1)) TERMINATE("wrong rotation matrix");
321  sym_op.proper = p;
322  sym_op.rotation = lattice_vectors_ * matrix3d<double>(sym_op.R * p) * inverse_lattice_vectors_;
323  sym_op.euler_angles = euler_angles(sym_op.rotation);
324 
325  space_group_symmetry_.push_back(sym_op);
326  }
327 
328  sym_table_ = mdarray<int, 2>(num_atoms_, num_spg_sym());
329  /* loop over spatial symmetries */
330  for (int isym = 0; isym < num_spg_sym(); isym++) {
331  for (int ia = 0; ia < num_atoms_; ia++) {
332  auto R = space_group_symmetry(isym).R;
333  auto t = space_group_symmetry(isym).t;
334  /* spatial transform */
335  vector3d<double> pos(positions__(0, ia), positions__(1, ia), positions__(2, ia));
336  auto v = reduce_coordinates(R * pos + t);
337 
338  int ja = -1;
339  /* check for equivalent atom */
340  for (int k = 0; k < num_atoms_; k++) {
341  vector3d<double> pos1(positions__(0, k), positions__(1, k), positions__(2, k));
342  if ((v.first - pos1).length() < tolerance_) {
343  ja = k;
344  break;
345  }
346  }
347 
348  if (ja == -1) {
349  TERMINATE("equivalent atom was not found");
350  }
351  sym_table_(ia, isym) = ja;
352  }
353  }
354 
355  /* loop over spatial symmetries */
356  for (int isym = 0; isym < num_spg_sym(); isym++) {
357  /* loop over spin symmetries */
358  for (int jsym = 0; jsym < num_spg_sym(); jsym++) {
359  /* take proper part of rotation matrix */
360  auto Rspin = space_group_symmetry(jsym).rotation;
361 
362  int n{0};
363  /* check if all atoms transfrom under spatial and spin symmetries */
364  for (int ia = 0; ia < num_atoms_; ia++) {
365  int ja = sym_table_(ia, isym);
366 
367  /* now check tha vector filed transforms from atom ia to atom ja */
368  /* vector field of atom is expected to be in Cartesian coordinates */
369  auto vd = Rspin * vector3d<double>(spins__(0, ia), spins__(1, ia), spins__(2, ia)) -
370  vector3d<double>(spins__(0, ja), spins__(1, ja), spins__(2, ja));
371 
372  if (vd.length() < 1e-10) {
373  n++;
374  }
375  }
376  /* if all atoms transform under spin rotaion, add it to a list */
377  if (n == num_atoms_) {
378  magnetic_group_symmetry_descriptor mag_op;
379  mag_op.spg_op = space_group_symmetry(isym);
380  mag_op.isym = isym;
381  mag_op.spin_rotation = Rspin;
382  magnetic_group_symmetry_.push_back(mag_op);
383  break;
384  }
385  }
386  }
387 }

389 inline matrix3d<double> Symmetry::rot_mtrx_cart(vector3d<double> euler_angles) const
390 {
391  double alpha = euler_angles[0];
392  double beta = euler_angles[1];
393  double gamma = euler_angles[2];
394 
395  matrix3d<double> rm;
396  rm(0, 0) = std::cos(alpha) * std::cos(beta) * std::cos(gamma) - std::sin(alpha) * std::sin(gamma);
397  rm(0, 1) = -std::cos(gamma) * std::sin(alpha) - std::cos(alpha) * std::cos(beta) * std::sin(gamma);
398  rm(0, 2) = std::cos(alpha) * std::sin(beta);
399  rm(1, 0) = std::cos(beta) * std::cos(gamma) * std::sin(alpha) + std::cos(alpha) * std::sin(gamma);
400  rm(1, 1) = std::cos(alpha) * std::cos(gamma) - std::cos(beta) * std::sin(alpha) * std::sin(gamma);
401  rm(1, 2) = std::sin(alpha) * std::sin(beta);
402  rm(2, 0) = -std::cos(gamma) * std::sin(beta);
403  rm(2, 1) = std::sin(beta) * std::sin(gamma);
404  rm(2, 2) = std::cos(beta);
405 
406  return rm;
407 }

409 inline vector3d<double> Symmetry::euler_angles(matrix3d<double> const& rot__) const
410 {
411  vector3d<double> angles(0, 0, 0);
412 
413  if (std::abs(rot__.det() - 1) > 1e-10)
414  {
415  std::stringstream s;
416  s << "determinant of rotation matrix is " << rot__.det();
417  TERMINATE(s);
418  }
419 
420  if (std::abs(rot__(2, 2) - 1.0) < 1e-10) // cos(beta) == 1, beta = 0
421  {
422  angles[0] = Utils::phi_by_sin_cos(rot__(1, 0), rot__(0, 0));
423  }
424  else if (std::abs(rot__(2, 2) + 1.0) < 1e-10) // cos(beta) == -1, beta = Pi
425  {
426  angles[0] = Utils::phi_by_sin_cos(-rot__(0, 1), rot__(1, 1));
427  angles[1] = pi;
428  }
429  else
430  {
431  double beta = std::acos(rot__(2, 2));
432  angles[0] = Utils::phi_by_sin_cos(rot__(1, 2) / std::sin(beta), rot__(0, 2) / std::sin(beta));
433  angles[1] = beta;
434  angles[2] = Utils::phi_by_sin_cos(rot__(2, 1) / std::sin(beta), -rot__(2, 0) / std::sin(beta));
435  }
436 
437  auto rm1 = rot_mtrx_cart(angles);
438 
439  for (int i = 0; i < 3; i++)
440  {
441  for (int j = 0; j < 3; j++)
442  {
443  if (std::abs(rot__(i, j) - rm1(i, j)) > 1e-10)
444  {
445  std::stringstream s;
446  s << "matrices don't match" << std::endl
447  << "initial symmetry matrix: " << std::endl
448  << rot__(0, 0) << " " << rot__(0, 1) << " " << rot__(0, 2) << std::endl
449  << rot__(1, 0) << " " << rot__(1, 1) << " " << rot__(1, 2) << std::endl
450  << rot__(2, 0) << " " << rot__(2, 1) << " " << rot__(2, 2) << std::endl
451  << "euler angles : " << angles[0] / pi << " " << angles[1] / pi << " " << angles[2] / pi << std::endl
452  << "computed symmetry matrix : " << std::endl
453  << rm1(0, 0) << " " << rm1(0, 1) << " " << rm1(0, 2) << std::endl
454  << rm1(1, 0) << " " << rm1(1, 1) << " " << rm1(1, 2) << std::endl
455  << rm1(2, 0) << " " << rm1(2, 1) << " " << rm1(2, 2) << std::endl;
456  TERMINATE(s);
457  }
458  }
459  }
460 
461  return angles;
462 }

464 inline int Symmetry::get_irreducible_reciprocal_mesh(vector3d<int> k_mesh__,
465  vector3d<int> is_shift__,
466  mdarray<double, 2>& kp__,
467  std::vector<double>& wk__) const
468 {
469  int nktot = k_mesh__[0] * k_mesh__[1] * k_mesh__[2];
470 
471  mdarray<int, 2> grid_address(3, nktot);
472  std::vector<int> ikmap(nktot);
473 
474  double lattice[3][3];
475  for (int i = 0; i < 3; i++)
476  {
477  for (int j = 0; j < 3; j++) lattice[i][j] = lattice_vectors_(i, j);
478  }
479 
480  int nknr = spg_get_ir_reciprocal_mesh((int(*)[3])&grid_address(0, 0),
481  &ikmap[0],
482  &k_mesh__[0],
483  &is_shift__[0],
484  1,
485  lattice,
486  (double(*)[3])&positions_(0, 0),
487  &types_[0],
488  num_atoms_,
489  tolerance_);
490 
491  std::map<int, int> wknr;
492  for (int ik = 0; ik < nktot; ik++)
493  {
494  if (wknr.count(ikmap[ik]) == 0) wknr[ikmap[ik]] = 0;
495  wknr[ikmap[ik]] += 1;
496  }
497 
498  wk__ = std::vector<double>(nknr);
499  kp__ = mdarray<double, 2>(3, nknr);
500 
501  int n = 0;
502  for (auto it = wknr.begin(); it != wknr.end(); it++) {
503  wk__[n] = double(it->second) / nktot;
504  for (int x = 0; x < 3; x++) {
505  kp__(x, n) = double(grid_address(x, it->first) + is_shift__[x] / 2.0) / k_mesh__[x];
506  }
507  n++;
508  }
509 
510  return nknr;
511 }

576  Gvec const& gvec__,
577  Communicator const& comm__) const
578 {
579  PROFILE("sirius::Symmetry::symmetrize_function_pw");
580 
581  int gvec_count = gvec__.gvec_count(comm__.rank());
582  int gvec_offset = gvec__.gvec_offset(comm__.rank());
583 
584  mdarray<double_complex, 1> sym_f_pw(gvec__.num_gvec());
585  sym_f_pw.zero();
586 
587  double* ptr = (double*)&sym_f_pw(0);
588 
589  #pragma omp parallel for
590  for (int i = 0; i < num_mag_sym(); i++) {
591  /* full space-group symmetry operation is {R|t} */
592  auto R = magnetic_group_symmetry(i).spg_op.R;
593  auto t = magnetic_group_symmetry(i).spg_op.t;
594 
595  for (int igloc = 0; igloc < gvec_count; igloc++) {
596  int ig = gvec_offset + igloc;
597 
598  double_complex z = f_pw__[ig] * std::exp(double_complex(0, twopi * (gvec__.gvec(ig) * t)));
599 
600  /* apply symmetry operation to the G-vector;
601  * remember that we move R from acting on x to acting on G: G(Rx) = (GR)x;
602  * GR is a vector-matrix multiplication [G][.....]
603  * [..R..]
604  * [.....]
605  * which can also be written as matrix^{T}-vector operation
606  */
607  auto gv_rot = transpose(R) * gvec__.gvec(ig);
608 
609  /* index of a rotated G-vector */
610  int ig_rot = gvec__.index_by_gvec(gv_rot);
611 
612  if (gvec__.reduced() && ig_rot == -1) {
613  gv_rot = gv_rot * (-1);
614  int ig_rot = gvec__.index_by_gvec(gv_rot);
615 
616  #pragma omp atomic update
617  ptr[2 * ig_rot] += z.real();
618 
619  #pragma omp atomic update
620  ptr[2 * ig_rot + 1] -= z.imag();
621  } else {
622  assert(ig_rot >= 0 && ig_rot < gvec__.num_gvec());
623 
624  #pragma omp atomic update
625  ptr[2 * ig_rot] += z.real();
626 
627  #pragma omp atomic update
628  ptr[2 * ig_rot + 1] += z.imag();
629  }
630  }
631  }
632  comm__.allreduce(&sym_f_pw(0), gvec__.num_gvec());
633 
634  double nrm = 1 / double(num_mag_sym());
635  #pragma omp parallel for
636  for (int ig = 0; ig < gvec__.num_gvec(); ig++) {
637  f_pw__[ig] = sym_f_pw(ig) * nrm;
638  }
639 }

936 /** \page sym Symmetry
937  * \section section1 Definition of symmetry operation
938  *
939  * SIRIUS uses Spglib to find the spacial symmetry operations. Spglib defines symmetry operation in fractional
940  * coordinates:
941  * \f[
942  * {\bf x'} = \{ {\bf R} | {\bf t} \} {\bf x} \equiv {\bf R}{\bf x} + {\bf t}
943  * \f]
944  * where \b R is the proper or improper rotation matrix with elements equal to -1,0,1 and determinant of 1
945  * (pure rotation) or -1 (rotoreflection) and \b t is the fractional translation, associated with the symmetry
946  * operation. The inverse of the symmetry operation is:
947  * \f[
948  * {\bf x} = \{ {\bf R} | {\bf t} \}^{-1} {\bf x'} = {\bf R}^{-1} ({\bf x'} - {\bf t}) =
949  * {\bf R}^{-1} {\bf x'} - {\bf R}^{-1} {\bf t}
950  * \f]
951  *
952  * We will always use an \a active transformation (transformation of vectors or functions) and never a passive
953  * transformation (transformation of coordinate system). However one should remember definition of the function
954  * transformation:
955  * \f[
956  * \hat {\bf P} f({\bf r}) \equiv f(\hat {\bf P}^{-1} {\bf r})
957  * \f]
958  *
959  * It is straightforward to get the rotation matrix in Cartesian coordinates. We know how the vector in Cartesian
960  * coordinates is obtained from the vector in fractional coordinates:
961  * \f[
962  * {\bf v} = {\bf L} {\bf x}
963  * \f]
964  * where \b L is the 3x3 matrix which clomuns are three lattice vectors. The backward transformation is simply
965  * \f[
966  * {\bf x} = {\bf L}^{-1} {\bf v}
967  * \f]
968  * Now we write rotation operation in fractional coordinates and apply the backward transformation to Cartesian
969  * coordinates:
970  * \f[
971  * {\bf x'} = {\bf R}{\bf x} \rightarrow {\bf L}^{-1} {\bf v'} = {\bf R} {\bf L}^{-1} {\bf v}
972  * \f]
973  * from which we derive the rotation operation in Cartesian coordinates:
974  * \f[
975  * {\bf v'} = {\bf L} {\bf R} {\bf L}^{-1} {\bf v}
976  * \f]
977  */
