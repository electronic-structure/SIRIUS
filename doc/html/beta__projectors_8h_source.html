234  void generate_beta_gk_t_lat_deriv(Simulation_context const& ctx__)
235  {
236  PROFILE("sirius::Beta_projectors::generate_beta_gk_t_lat_deriv");
237 
238  int mu = 0;
239  int nu = 0;
240 
241  /* allocate array */
242  matrix<double_complex> beta_gk_t_lat_deriv(gkvec_.gvec_count(comm_.rank()), num_beta_t_);
243 
244  /* compute dG_tau / da_{mu,nu} */
245  auto dG_da = [this](vector3d<double>& gvc, int tau, int mu, int nu)
246  {
247  return -unit_cell_.inverse_lattice_vectors()(nu, tau) * gvc[mu];
248  };
249 
250  /* compute derivative of theta angle with respect to lattice vectors d theta / da_{mu,nu} */
251  auto dtheta_da = [this, dG_da](vector3d<double>& gvc, vector3d<double>& gvs, int mu, int nu)
252  {
253  double g = gvs[0];
254  double theta = gvs[1];
255  double phi = gvs[2];
256 
257  double result = std::cos(theta) * std::cos(phi) * dG_da(gvc, 0, mu, nu) +
258  std::cos(theta) * std::sin(phi) * dG_da(gvc, 1, mu, nu) -
259  std::sin(theta) * dG_da(gvc, 2, mu, nu);
260  return result / g;
261  };
262 
263  auto dphi_da = [this, dG_da](vector3d<double>& gvc, vector3d<double>& gvs, int mu, int nu)
264  {
265  double g = gvs[0];
266  double phi = gvs[2];
267 
268  double result = -std::sin(phi) * dG_da(gvc, 0, mu, nu) + std::cos(phi) * dG_da(gvc, 0, mu, nu);
269  return result / g;
270  };
271 
272  auto dRlm_da = [this, dtheta_da, dphi_da](int lm, vector3d<double>& gvc, vector3d<double>& gvs, int mu, int nu)
273  {
274  double theta = gvs[1];
275  double phi = gvs[2];
276 
277  return SHT::dRlm_dtheta(lm, theta, phi) * dtheta_da(gvc, gvs, mu, nu) +
278  SHT::dRlm_dphi_sin_theta(lm, theta, phi) * dphi_da(gvc, gvs, mu, nu);
279  };
280 
281  auto djl_da = [this, dG_da](vector3d<double>& gvc, vector3d<double>& gvs, int mu, int nu)
282  {
283  double theta = gvs[1];
284  double phi = gvs[2];
285 
286  return std::sin(theta) * std::cos(phi) * dG_da(gvc, 0, mu, nu) +
287  std::sin(theta) * std::sin(phi) * dG_da(gvc, 1, mu, nu) +
288  std::cos(theta) * dG_da(gvc, 2, mu, nu);
289  };
290 
291  /* compute d <G+k|beta> / a_{mu, nu} */
292  #pragma omp parallel for
293  for (int igkloc = 0; igkloc < gkvec_.gvec_count(comm_.rank()); igkloc++) {
294  int igk = gkvec_.gvec_offset(comm_.rank()) + igkloc;
295  auto gvc = gkvec_.gkvec_cart(igk);
296  /* vs = {r, theta, phi} */
297  auto gvs = SHT::spherical_coordinates(gvc);
298 
299  if (gvs[0] < 1e-10) {
300  for (int i = 0; i < num_beta_t_; i++) {
301  beta_gk_t_lat_deriv(igkloc, i) = 0;
302  }
303  continue;
304  }
305 
306  /* compute real spherical harmonics for G+k vector */
307  std::vector<double> gkvec_rlm(Utils::lmmax(lmax_beta_));
308  std::vector<double> gkvec_drlm(Utils::lmmax(lmax_beta_));
309 
310  SHT::spherical_harmonics(lmax_beta_, gvs[1], gvs[2], &gkvec_rlm[0]);
311 
312  for (int lm = 0; lm < Utils::lmmax(lmax_beta_); lm++) {
313  gkvec_drlm[lm] = dRlm_da(lm, gvc, gvs, mu, nu);
314  }
315 
316  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
317  auto& atom_type = unit_cell_.atom_type(iat);
318  for (int xi = 0; xi < atom_type.mt_basis_size(); xi++) {
319  int l = atom_type.indexb(xi).l;
320  int lm = atom_type.indexb(xi).lm;
321  int idxrf = atom_type.indexb(xi).idxrf;
322 
323  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
324 
325  auto d1 = ctx__.radial_integrals().beta_radial_integral(idxrf, iat, gvs[0]) *
326  (gkvec_drlm[lm] - 0.5 * gkvec_rlm[lm] * unit_cell_.inverse_lattice_vectors()(nu, mu));
327 
328  auto d2 = gkvec_rlm[lm] * djl_da(gvc, gvs, mu, nu) * ctx__.radial_integrals().beta_djldq_radial_integral(idxrf, iat, gvs[0]);
329 
330  beta_gk_t_lat_deriv(igkloc, atom_type.offset_lo() + xi) = z * (d1 + d2);
331  }
332  }
333  }
334 
335  //if (unit_cell_.parameters().control().print_checksum_) {
336  // auto c1 = beta_gk_t_.checksum();
337  // comm_.allreduce(&c1, 1);
338  // if (comm_.rank() == 0) {
339  // DUMP("checksum(beta_gk_t) : %18.10f %18.10f", c1.real(), c1.imag())
340  // }
341  //}
342  }

345 inline Beta_projectors::Beta_projectors(Simulation_context const& ctx__,
346  Communicator const& comm__,
347  Gvec const& gkvec__)
348  : comm_(comm__)
349  , unit_cell_(ctx__.unit_cell())
350  , gkvec_(gkvec__)
351  , lmax_beta_(unit_cell_.lmax())
352  , pu_(ctx__.processing_unit())
353 {
354  PROFILE("sirius::Beta_projectors::Beta_projectors");
355 
356  num_gkvec_loc_ = gkvec_.gvec_count(comm_.rank());
357 
358  split_in_chunks();
359 
360  generate_beta_gk_t(ctx__);
361 
362  if (pu_ == GPU) {
363  gkvec_coord_ = mdarray<double, 2>(3, num_gkvec_loc_, ctx__.dual_memory_t());
364  /* copy G+k vectors */
365  for (int igk_loc = 0; igk_loc < num_gkvec_loc_; igk_loc++) {
366  int igk = gkvec_.gvec_offset(comm_.rank()) + igk_loc;
367  auto vgk = gkvec_.gkvec(igk);
368  for (auto x: {0, 1, 2}) {
369  gkvec_coord_(x, igk_loc) = vgk[x];
370  }
371  }
372  gkvec_coord_.copy<memory_t::host, memory_t::device>();
373 
374  beta_gk_t_.allocate(memory_t::device);
375  beta_gk_t_.copy<memory_t::host, memory_t::device>();
376  }
377  beta_gk_gpu_ = matrix<double_complex>(num_gkvec_loc_, max_num_beta_, memory_t::none);
378 
379  beta_gk_a_ = matrix<double_complex>(num_gkvec_loc_, unit_cell_.mt_lo_basis_size());
380 
381  #pragma omp for
382  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
383  double phase = twopi * (gkvec_.vk() * unit_cell_.atom(ia).position());
384  double_complex phase_k = std::exp(double_complex(0.0, phase));
385 
386  std::vector<double_complex> phase_gk(num_gkvec_loc_);
387  for (int igk_loc = 0; igk_loc < num_gkvec_loc_; igk_loc++) {
388  int igk = gkvec_.gvec_offset(comm_.rank()) + igk_loc;
389  auto G = gkvec_.gvec(igk);
390  phase_gk[igk_loc] = std::conj(ctx__.gvec_phase_factor(G, ia) * phase_k);
391  }
392 
393  for (int xi = 0; xi < unit_cell_.atom(ia).mt_lo_basis_size(); xi++) {
394  for (int igk_loc = 0; igk_loc < num_gkvec_loc_; igk_loc++) {
395  beta_gk_a_(igk_loc, unit_cell_.atom(ia).offset_lo() + xi) =
396  beta_gk_t_(igk_loc, unit_cell_.atom(ia).type().offset_lo() + xi) * phase_gk[igk_loc];
397  }
398  }
399  }

402 inline void Beta_projectors::generate_beta_gk_t(Simulation_context const& ctx__)
403 {
404  PROFILE("sirius::Beta_projectors::generate_beta_gk_t");
405 
406  if (!num_beta_t_) {
407  return;
408  }
409 
410  /* allocate array */
411  beta_gk_t_ = matrix<double_complex>(gkvec_.gvec_count(comm_.rank()), num_beta_t_);
412 
413  /* compute <G+k|beta> */
414  #pragma omp parallel for
415  for (int igkloc = 0; igkloc < gkvec_.gvec_count(comm_.rank()); igkloc++) {
416  int igk = gkvec_.gvec_offset(comm_.rank()) + igkloc;
417  double gk = gkvec_.gvec_len(igk);
418  /* vs = {r, theta, phi} */
419  auto vs = SHT::spherical_coordinates(gkvec_.gkvec_cart(igk));
420  /* compute real spherical harmonics for G+k vector */
421  std::vector<double> gkvec_rlm(Utils::lmmax(lmax_beta_));
422  SHT::spherical_harmonics(lmax_beta_, vs[1], vs[2], &gkvec_rlm[0]);
423  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
424  auto& atom_type = unit_cell_.atom_type(iat);
425  for (int xi = 0; xi < atom_type.mt_basis_size(); xi++) {
426  int l = atom_type.indexb(xi).l;
427  int lm = atom_type.indexb(xi).lm;
428  int idxrf = atom_type.indexb(xi).idxrf;
429 
430  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
431  beta_gk_t_(igkloc, atom_type.offset_lo() + xi) = z * gkvec_rlm[lm] * ctx__.radial_integrals().beta_radial_integral(idxrf, iat, gk);
432  }
433  }
434  }
435 
436  if (unit_cell_.parameters().control().print_checksum_) {
437  auto c1 = beta_gk_t_.checksum();
438  comm_.allreduce(&c1, 1);
439  if (comm_.rank() == 0) {
440  DUMP("checksum(beta_gk_t) : %18.10f %18.10f", c1.real(), c1.imag())
441  }
442  }

445 inline void Beta_projectors::split_in_chunks()
446 {
447  /* split beta-projectors into chunks */
448  int num_atoms_in_chunk = (comm_.size() == 1) ? unit_cell_.num_atoms() : std::min(unit_cell_.num_atoms(), 256);
449  int num_beta_chunks = unit_cell_.num_atoms() / num_atoms_in_chunk + std::min(1, unit_cell_.num_atoms() % num_atoms_in_chunk);
450  splindex<block> spl_beta_chunks(unit_cell_.num_atoms(), num_beta_chunks, 0);
451 
452  beta_chunks_ = mdarray<beta_chunk_t, 1>(num_beta_chunks);
453 
454  int offset_in_beta_gk = 0;
455 
456  for (int ib = 0; ib < num_beta_chunks; ib++) {
457  /* number of atoms in chunk */
458  int na = spl_beta_chunks.local_size(ib);
459  beta_chunks_(ib).num_atoms_ = na;
460  beta_chunks_(ib).desc_ = mdarray<int, 2>(4, na);
461  beta_chunks_(ib).atom_pos_ = mdarray<double, 2>(3, na);
462 
463  int num_beta{0};
464 
465  for (int i = 0; i < na; i++) {
466  /* global index of atom by local index and chunk */
467  int ia = spl_beta_chunks.global_index(i, ib);
468  auto pos = unit_cell_.atom(ia).position();
469  auto& type = unit_cell_.atom(ia).type();
470  /* atom fractional coordinates */
471  for (int x: {0, 1, 2}) {
472  beta_chunks_(ib).atom_pos_(x, i) = pos[x];
473  }
474  /* number of beta functions for atom */
475  beta_chunks_(ib).desc_(beta_desc_idx::nbf, i) = type.mt_basis_size();
476  /* offset in beta_gk*/
477  beta_chunks_(ib).desc_(beta_desc_idx::offset, i) = num_beta;
478  /* offset in beta_gk_t */
479  beta_chunks_(ib).desc_(beta_desc_idx::offset_t, i) = type.offset_lo();
480  /* global index of atom */
481  beta_chunks_(ib).desc_(beta_desc_idx::ia, i) = ia;
482 
483  num_beta += type.mt_basis_size();
484  }
485  /* number of beta-projectors in this chunk */
486  beta_chunks_(ib).num_beta_ = num_beta;
487  beta_chunks_(ib).offset_ = offset_in_beta_gk;
488  offset_in_beta_gk += num_beta;
489 
490  if (pu_ == GPU) {
491  beta_chunks_[ib].desc_.allocate(memory_t::device);
492  beta_chunks_[ib].desc_.copy<memory_t::host, memory_t::device>();
493 
494  beta_chunks_[ib].atom_pos_.allocate(memory_t::device);
495  beta_chunks_[ib].atom_pos_.copy<memory_t::host, memory_t::device>();
496  }
497  }
498 
499  max_num_beta_ = 0;
500  for (int ib = 0; ib < num_beta_chunks; ib++) {
501  max_num_beta_ = std::max(max_num_beta_, beta_chunks_(ib).num_beta_);
502  }
503 
504  num_beta_t_ = 0;
505  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
506  num_beta_t_ += unit_cell_.atom_type(iat).mt_lo_basis_size();
507  }
508 }

535 template<>
536 inline void Beta_projectors::inner<double_complex>(int chunk__,
537  wave_functions& phi__,
538  int idx0__,
539  int n__,
540  mdarray<double_complex, 2> &beta_gk,
541  mdarray<double, 1> &beta_phi)
542 {
543  PROFILE("sirius::Beta_projectors::inner");
544 
545  assert(num_gkvec_loc_ == phi__.pw_coeffs().num_rows_loc());
546 
547  int nbeta = beta_chunk(chunk__).num_beta_;
548 
549  if (static_cast<size_t>(nbeta * n__) > beta_phi.size()) {
550  beta_phi = mdarray<double, 1>(2 * nbeta * n__);
551  if (pu_ == GPU) {
552  beta_phi.allocate(memory_t::device);
553  }
554  }
555 
556  switch (pu_) {
557  case CPU: {
558  /* compute <beta|phi> */
559  linalg<CPU>::gemm(2, 0, nbeta, n__, num_gkvec_loc_,
560  beta_gk.at<CPU>(), num_gkvec_loc_,
561  phi__.pw_coeffs().prime().at<CPU>(0, idx0__), phi__.pw_coeffs().prime().ld(),
562  (double_complex*)beta_phi.at<CPU>(), nbeta);
563  break;
564  }
565  case GPU: {
566  #ifdef __GPU
567  linalg<GPU>::gemm(2, 0, nbeta, n__, num_gkvec_loc_, beta_gk.at<GPU>(), num_gkvec_loc_,
568  phi__.pw_coeffs().prime().at<GPU>(0, idx0__), phi__.pw_coeffs().prime().ld(),
569  (double_complex*)beta_phi.at<GPU>(), nbeta);
570  beta_phi.copy_to_host(2 * nbeta * n__);
571  #else
572  TERMINATE_NO_GPU
573  #endif
574  break;
575  }
576  }
577 
578  comm_.allreduce(beta_phi.at<CPU>(), 2 * nbeta * n__);
579 
580  if (pu_ == GPU) {
