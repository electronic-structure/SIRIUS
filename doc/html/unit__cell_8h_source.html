643  auto v = inp__.coordinates_[iat][ia];

648  }

660 inline void Unit_cell::initialize()
661 {
662  PROFILE("sirius::Unit_cell::initialize");
663 
664  /* split number of atom between all MPI ranks */
665  spl_num_atoms_ = splindex<block>(num_atoms(), comm_.size(), comm_.rank());
666 
667  /* initialize atom types */
668  int offs_lo{0};
669  for (int iat = 0; iat < num_atom_types(); iat++) {
670  atom_type(iat).init(offs_lo);
671  max_num_mt_points_ = std::max(max_num_mt_points_, atom_type(iat).num_mt_points());
672  max_mt_basis_size_ = std::max(max_mt_basis_size_, atom_type(iat).mt_basis_size());
673  max_mt_radial_basis_size_ = std::max(max_mt_radial_basis_size_, atom_type(iat).mt_radial_basis_size());
674  max_mt_aw_basis_size_ = std::max(max_mt_aw_basis_size_, atom_type(iat).mt_aw_basis_size());
675  max_mt_lo_basis_size_ = std::max(max_mt_lo_basis_size_, atom_type(iat).mt_lo_basis_size());
676  lmax_ = std::max(lmax_, atom_type(iat).indexr().lmax());
677  offs_lo += atom_type(iat).mt_lo_basis_size();
678  }
679 
680  /* find the charges */
681  for (int i = 0; i < num_atoms(); i++) {
682  total_nuclear_charge_ += atom(i).zn();
683  num_core_electrons_ += atom(i).type().num_core_electrons();
684  num_valence_electrons_ += atom(i).type().num_valence_electrons();
685  }
686  num_electrons_ = num_core_electrons_ + num_valence_electrons_;
687 
688  /* initialize atoms */
689  for (int ia = 0; ia < num_atoms(); ia++) {
690  atom(ia).init(mt_aw_basis_size_, mt_lo_basis_size_, mt_basis_size_);
691  mt_aw_basis_size_ += atom(ia).mt_aw_basis_size();
692  mt_lo_basis_size_ += atom(ia).mt_lo_basis_size();
693  mt_basis_size_ += atom(ia).mt_basis_size();

833 inline std::vector<double> Unit_cell::find_mt_radii()
834 {
835  if (nearest_neighbours_.size() == 0) {
836  TERMINATE("array of nearest neighbours is empty");
837  }
838 
839  std::vector<double> Rmt(num_atom_types(), 1e10);
840 
841  if (parameters_.auto_rmt() == 1) {
842  for (int ia = 0; ia < num_atoms(); ia++) {
843  int id1 = atom(ia).type_id();
844  if (nearest_neighbours_[ia].size() > 1) {
845  int ja = nearest_neighbours_[ia][1].atom_id;
846  int id2 = atom(ja).type_id();
847  /* don't allow spheres to touch: take a smaller value than half a distance */
848  double R = std::min(parameters_.rmt_max(), 0.95 * nearest_neighbours_[ia][1].distance / 2);
849  /* take minimal R for the given atom type */
850  Rmt[id1] = std::min(R, Rmt[id1]);
851  Rmt[id2] = std::min(R, Rmt[id2]);
852  } else {
853  Rmt[id1] = parameters_.rmt_max();
854  }
855  }
856  }
857 
858  if (parameters_.auto_rmt() == 2) {
859  std::vector<double> scale(num_atom_types(), 1e10);
860 
861  for (int ia = 0; ia < num_atoms(); ia++) {
862  int id1 = atom(ia).type_id();
863  if (nearest_neighbours_[ia].size() > 1) {
864  int ja = nearest_neighbours_[ia][1].atom_id;
865  int id2 = atom(ja).type_id();
866 
867  double d = nearest_neighbours_[ia][1].distance;
868  double s = 0.95 * d / (atom_type(id1).mt_radius() + atom_type(id2).mt_radius());
869  scale[id1] = std::min(s, scale[id1]);
870  scale[id2] = std::min(s, scale[id2]);
871  } else {
872  scale[id1] = parameters_.rmt_max() / atom_type(id1).mt_radius();
873  }
874  }
875 
876  for (int iat = 0; iat < num_atom_types(); iat++) {
877  Rmt[iat] = std::min(parameters_.rmt_max(), atom_type(iat).mt_radius() * scale[iat]);
878  }
879  }
880 
881  /* Suppose we have 3 different atoms. First we determint Rmt between 1st and 2nd atom,
882  * then we determine Rmt between (let's say) 2nd and 3rd atom and at this point we reduce
883  * the Rmt of the 2nd atom. This means that the 1st atom gets a possibility to expand if
884  * it is far from the 3rd atom. */
885  bool inflate = true;
886 
887  if (inflate) {
888  std::vector<bool> scale_Rmt(num_atom_types(), true);
889  for (int ia = 0; ia < num_atoms(); ia++) {
890  int id1 = atom(ia).type_id();
891 
892  if (nearest_neighbours_[ia].size() > 1) {
893  int ja = nearest_neighbours_[ia][1].atom_id;
894  int id2 = atom(ja).type_id();
895  double dist = nearest_neighbours_[ia][1].distance;
896 
897  if (Rmt[id1] + Rmt[id2] > dist * 0.94) {
898  scale_Rmt[id1] = false;
899  scale_Rmt[id2] = false;
900  }
901  }
902  }
903 
904  for (int ia = 0; ia < num_atoms(); ia++) {
905  int id1 = atom(ia).type_id();
906  if (nearest_neighbours_[ia].size() > 1) {
907  int ja = nearest_neighbours_[ia][1].atom_id;
908  int id2 = atom(ja).type_id();
909  double dist = nearest_neighbours_[ia][1].distance;
910 
911  if (scale_Rmt[id1]) {
912  Rmt[id1] = std::min(parameters_.rmt_max(), 0.95 * (dist - Rmt[id2]));
913  }
914  }
915  }
916  }
917 
918  for (int i = 0; i < num_atom_types(); i++) {
919  if (Rmt[i] < 0.3) {
920  TERMINATE("Muffin-tin radius is too small");
921  }
922  }
923 
924  return Rmt;
925 }
