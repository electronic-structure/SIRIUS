60  BLACS_grid(Communicator const& comm__, int num_ranks_row__, int num_ranks_col__)
61  : comm_(comm__)
62  , num_ranks_row_(num_ranks_row__)
63  , num_ranks_col_(num_ranks_col__)
64  {
65  PROFILE("sddk::BLACS_grid::BLACS_grid");
66 
67  mpi_grid_ = std::unique_ptr<MPI_grid>(new MPI_grid({num_ranks_row__, num_ranks_col__}, comm_));
68 
69  rank_row_ = mpi_grid_->coordinate(0);
70  rank_col_ = mpi_grid_->coordinate(1);
71 
72  #ifdef __SCALAPACK
73  /* create handler first */
74  blacs_handler_ = linalg_base::create_blacs_handler(mpi_grid_->communicator().mpi_comm());
75 
76  std::vector<int> map_ranks(num_ranks_row__ * num_ranks_col__);
77  for (int j = 0; j < num_ranks_col__; j++) {
78  for (int i = 0; i < num_ranks_row__; i++) {
79  map_ranks[i + j * num_ranks_row__] = mpi_grid_->communicator().cart_rank({i, j});
80  }
81  }
82 
83  /* create context */
84  blacs_context_ = blacs_handler_;
85  linalg_base::gridmap(&blacs_context_, &map_ranks[0], num_ranks_row__, num_ranks_row__, num_ranks_col__);
86 
87  /* check the grid */
88  int nrow1, ncol1, irow1, icol1;
89  linalg_base::gridinfo(blacs_context_, &nrow1, &ncol1, &irow1, &icol1);
90 
91  if (rank_row_ != irow1 || rank_col_ != icol1 || num_ranks_row__ != nrow1 || num_ranks_col__ != ncol1) {
92  std::stringstream s;
93  s << "wrong grid" << std::endl
94  << " row | col | nrow | ncol " << std::endl
95  << " mpi_grid " << rank_row_ << " " << rank_col_ << " " << num_ranks_row__ << " " << num_ranks_col__
96  << std::endl
97  << " blacs " << irow1 << " " << icol1 << " " << nrow1 << " " << ncol1;
98  TERMINATE(s);
99  }
100  #endif
101  }
