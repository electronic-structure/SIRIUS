46 /// Simulation context is a set of parameters and objects describing a single simulation.

347  inline void generate_phase_factors(int iat__, mdarray<double_complex, 2>& phase_factors__) const
348  {
349  int na = unit_cell_.atom_type(iat__).num_atoms();
350  switch (processing_unit_) {
351  case CPU: {
352  #pragma omp parallel for
353  for (int igloc = 0; igloc < gvec_count(); igloc++) {
354  int ig = gvec_offset() + igloc;
355  for (int i = 0; i < na; i++) {
356  int ia = unit_cell_.atom_type(iat__).atom_id(i);
357  phase_factors__(igloc, i) = gvec_phase_factor(ig, ia);
358  }
359  }
360  break;
361  }
362  case GPU: {
363  #ifdef __GPU
364  generate_phase_factors_gpu(gvec_count(), na, gvec_coord().at<GPU>(), atom_coord(iat__).at<GPU>(),
365  phase_factors__.at<GPU>());
366  #else
367  TERMINATE_NO_GPU
368  #endif
369  break;
370  }
371  }
372  }

389 inline void Simulation_context::init_fft()
390 {
391  auto rlv = unit_cell_.reciprocal_lattice_vectors();
392 
393  int npr = control_input_section_.mpi_grid_dims_[0];
394  int npc = control_input_section_.mpi_grid_dims_[1];
395 
396  if (!(control_input_section_.fft_mode_ == "serial" || control_input_section_.fft_mode_ == "parallel")) {
397  TERMINATE("wrong FFT mode");
398  }
399 
400  /* create FFT driver for dense mesh (density and potential) */
401  fft_ = std::unique_ptr<FFT3D>(new FFT3D(find_translations(pw_cutoff(), rlv), comm_fft(), processing_unit()));
402 
403  /* create a list of G-vectors for dense FFT grid; G-vectors are divided between all available MPI ranks.*/
404  gvec_ = Gvec(rlv, pw_cutoff(), comm(), comm_fft(), control_input_section_.reduce_gvec_);
405 
406  /* prepare fine-grained FFT driver for the entire simulation */
407  fft_->prepare(gvec_.partition());
408 
409  /* create FFT driver for coarse mesh */
410  fft_coarse_ = std::unique_ptr<FFT3D>(new FFT3D(find_translations(2 * gk_cutoff(), rlv), comm_fft_coarse(),
411  processing_unit()));
412 
413  /* create a list of G-vectors for corase FFT grid */
414  gvec_coarse_ = Gvec(rlv, gk_cutoff() * 2, comm(), comm_fft_coarse(), control_input_section_.reduce_gvec_);
415 }

417 inline void Simulation_context::initialize()
418 {
419  PROFILE("sirius::Simulation_context::initialize");
420 
421  /* can't initialize twice */
422  if (initialized_) {
423  TERMINATE("Simulation context is already initialized.");
424  }
425 
426  /* get processing unit */
427  std::string pu = control_input_section_.processing_unit_;
428  if (pu == "") {
429  #ifdef __GPU
430  pu = "gpu";
431  #else
432  pu = "cpu";
433  #endif
434  }
435  if (pu == "cpu") {
436  processing_unit_ = CPU;
437  } else if (pu == "gpu") {
438  processing_unit_ = GPU;
439  } else {
440  TERMINATE("wrong processing unit");
441  }
442 
443  /* check if we can use a GPU device */
444  if (processing_unit() == GPU) {
445  #ifndef __GPU
446  TERMINATE_NO_GPU
447  #endif
448  }
449 
450  /* check MPI grid dimensions and set a default grid if needed */
451  if (!control_input_section_.mpi_grid_dims_.size()) {
452  control_input_section_.mpi_grid_dims_ = {1, 1};
453  }
454  if (control_input_section_.mpi_grid_dims_.size() != 2) {
455  TERMINATE("wrong MPI grid");
456  }
457 
458  int npr = control_input_section_.mpi_grid_dims_[0];
459  int npc = control_input_section_.mpi_grid_dims_[1];
460  int npb = npr * npc;
461  int npk = comm_.size() / npb;
462  if (npk * npb != comm_.size()) {
463  std::stringstream s;
464  s << "Can't divide " << comm_.size() << " ranks into groups of size " << npb;
465  TERMINATE(s);
466  }
467 
468  /* setup MPI grid */
469  mpi_grid_ = std::unique_ptr<MPI_grid>(new MPI_grid({npr, npc, npk}, comm_));
470 
471  /* setup BLACS grid */
472  blacs_grid_ = std::unique_ptr<BLACS_grid>(new BLACS_grid(comm_band(), npr, npc));
473 
474  /* can't use reduced G-vectors in LAPW code */
475  if (full_potential()) {
476  control_input_section_.reduce_gvec_ = false;
477  }
478 
479  if (!iterative_solver_input_section_.type_.size()) {
480  if (full_potential()) {
481  iterative_solver_input_section_.type_ = "exact";
482  } else {
483  iterative_solver_input_section_.type_ = "davidson";
484  }
485  }
486 
487  /* initialize variables, related to the unit cell */
488  unit_cell_.initialize();
489 
490  /* find the cutoff for G+k vectors (derived from rgkmax (aw_cutoff here) and minimum MT radius) */
491  if (full_potential()) {
492  set_gk_cutoff(aw_cutoff() / unit_cell_.min_mt_radius());
493  }
494 
495  if (esm_type() == electronic_structure_method_t::pseudopotential) {
496  lmax_rho_ = unit_cell_.lmax() * 2;
497  lmax_pot_ = unit_cell_.lmax() * 2;
498  }
499 
500  /* initialize FFT interface */
501  init_fft();
502 
503  if (comm_.rank() == 0 && control().print_memory_usage_) {
504  MEMORY_USAGE_INFO();
505  }
506 
507  if (unit_cell_.num_atoms() != 0) {
508  unit_cell_.symmetry().check_gvec_symmetry(gvec_, comm_);
509  if (!full_potential()) {
510  unit_cell_.symmetry().check_gvec_symmetry(gvec_coarse_, comm_);
511  }
512  }
513 
514  auto& fft_grid = fft().grid();
515  std::pair<int, int> limits(0, 0);
516  for (int x: {0, 1, 2}) {
517  limits.first = std::min(limits.first, fft_grid.limits(x).first);
518  limits.second = std::max(limits.second, fft_grid.limits(x).second);
519  }
520 
521  phase_factors_ = mdarray<double_complex, 3>(3, limits, unit_cell().num_atoms());
522 
523  #pragma omp parallel for
524  for (int i = limits.first; i <= limits.second; i++) {
525  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
526  auto pos = unit_cell_.atom(ia).position();
527  for (int x: {0, 1, 2}) {
528  phase_factors_(x, i, ia) = std::exp(double_complex(0.0, twopi * (i * pos[x])));
529  }
530  }
531  }
532 
533  if (full_potential()) {
534  step_function_ = std::unique_ptr<Step_function>(new Step_function(unit_cell_, fft_.get(), gvec_, comm_));
535  }
536 
537  /* take 10% of empty non-magnetic states */
538  if (num_fv_states_ < 0) {
539  num_fv_states_ = static_cast<int>(1e-8 + unit_cell_.num_valence_electrons() / 2.0) +
540  std::max(10, static_cast<int>(0.1 * unit_cell_.num_valence_electrons()));
541  }
542 
543  if (num_fv_states() < static_cast<int>(unit_cell_.num_valence_electrons() / 2.0)) {
544  std::stringstream s;
545  s << "not enough first-variational states : " << num_fv_states();
546  TERMINATE(s);
547  }
548 
549  /* setup the cyclic block size */
550  if (cyclic_block_size() < 0) {
551  double a = std::min(std::log2(double(num_fv_states_) / blacs_grid_->num_ranks_col()),
552  std::log2(double(num_fv_states_) / blacs_grid_->num_ranks_row()));
553  if (a < 1) {
554  control_input_section_.cyclic_block_size_ = 2;
555  } else {
556  control_input_section_.cyclic_block_size_ = static_cast<int>(std::min(128.0, std::pow(2.0, static_cast<int>(a))) + 1e-12);
557  }
558  }
559 
560  std::string evsn[] = {std_evp_solver_name(), gen_evp_solver_name()};
561 
562  if (comm_band().size() == 1) {
563  if (evsn[0] == "") {
564  #if defined(__GPU) && defined(__MAGMA)
565  evsn[0] = "magma";
566  #else
567  evsn[0] = "lapack";
568  #endif
569  }
570  if (evsn[1] == "") {
571  #if defined(__GPU) && defined(__MAGMA)
572  evsn[1] = "magma";
573  #else
574  evsn[1] = "lapack";
575  #endif
576  }
577  } else {
578  if (evsn[0] == "") {
579  #ifdef __SCALAPACK
580  evsn[0] = "scalapack";
581  #endif
582  #ifdef __ELPA
583  evsn[0] = "elpa1";
584  #endif
585  }
586  if (evsn[1] == "") {
587  #ifdef __SCALAPACK
588  evsn[1] = "scalapack";
589  #endif
590  #ifdef __ELPA
591  evsn[1] = "elpa1";
592  #endif
593  }
594  }
595 
596  ev_solver_t* evst[] = {&std_evp_solver_type_, &gen_evp_solver_type_};
597 
598  std::map<std::string, ev_solver_t> str_to_ev_solver_t;
599 
600  str_to_ev_solver_t["lapack"] = ev_lapack;
601  str_to_ev_solver_t["scalapack"] = ev_scalapack;
602  str_to_ev_solver_t["elpa1"] = ev_elpa1;
603  str_to_ev_solver_t["elpa2"] = ev_elpa2;
604  str_to_ev_solver_t["magma"] = ev_magma;
605  str_to_ev_solver_t["plasma"] = ev_plasma;
606  str_to_ev_solver_t["rs_cpu"] = ev_rs_cpu;
607  str_to_ev_solver_t["rs_gpu"] = ev_rs_gpu;
608 
609  for (int i: {0, 1}) {
610  auto name = evsn[i];
611 
612  if (str_to_ev_solver_t.count(name) == 0) {
613  std::stringstream s;
614  s << "wrong eigen value solver " << name;
615  TERMINATE(s);
616  }
617  *evst[i] = str_to_ev_solver_t[name];
618  }
619 
620  if (control().verbosity_ > 0 && comm_.rank() == 0) {
621  print_info();
622  }
623 
624  radial_integrals_ = std::unique_ptr<Radial_integrals>(new Radial_integrals(*this, unit_cell()));
625 
626  if (esm_type() == electronic_structure_method_t::pseudopotential) {
627  /* create augmentation operator Q_{xi,xi'}(G) here */
628  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
629  augmentation_op_.push_back(std::move(Augmentation_operator(comm_, unit_cell_.atom_type(iat), gvec_,
630  unit_cell_.omega(), *radial_integrals_)));
631  }
632  }
633 
634  if (processing_unit() == GPU) {
635  #ifdef __GPU
636  gvec_coord_ = mdarray<int, 2>(gvec_count(), 3, memory_t::host | memory_t::device, "gvec_coord_");
637  for (int igloc = 0; igloc < gvec_count(); igloc++) {
638  int ig = gvec_offset() + igloc;
639  auto G = gvec_.gvec(ig);
640  for (int x: {0, 1, 2}) {
641  gvec_coord_(igloc, x) = G[x];
642  }
643  }
644  gvec_coord_.copy_to_device();
645 
646  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
647  int nat = unit_cell_.atom_type(iat).num_atoms();
648  atom_coord_.push_back(std::move(mdarray<double, 2>(nat, 3, memory_t::host | memory_t::device)));
649  for (int i = 0; i < nat; i++) {
650  int ia = unit_cell_.atom_type(iat).atom_id(i);
651  for (int x: {0, 1, 2}) {
652  atom_coord_.back()(i, x) = unit_cell_.atom(ia).position()[x];
653  }
654  }
655  atom_coord_.back().copy_to_device();
656  }
657  #endif
658  }
659 
660  //time_active_ = -runtime::wtime();
661 
662  initialized_ = true;
663 }
