72 inline void Potential::poisson_add_pseudo_pw(mdarray<double_complex, 2>& qmt,
73  mdarray<double_complex, 2>& qit,
74  double_complex* rho_pw)
75 {
76  PROFILE("sirius::Potential::poisson_add_pseudo_pw");
77 
78  /* The following term is added to the plane-wave coefficients of the charge density:
79  * Integrate[SphericalBesselJ[l,a*x]*p[x,R]*x^2,{x,0,R},Assumptions->{l>=0,n>=0,R>0,a>0}] /
80  * Integrate[p[x,R]*x^(2+l),{x,0,R},Assumptions->{h>=0,n>=0,R>0}]
81  * i.e. contributon from pseudodensity to l-th channel of plane wave expansion multiplied by
82  * the difference bethween true and interstitial-in-the-mt multipole moments and divided by the
83  * moment of the pseudodensity.
84  */
85  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
86  int iat = unit_cell_.atom(ia).type_id();
87 
88  double R = unit_cell_.atom(ia).mt_radius();
89 
90  /* compute G-vector independent prefactor */
91  std::vector<double_complex> zp(ctx_.lmmax_rho());
92  for (int l = 0, lm = 0; l <= ctx_.lmax_rho(); l++) {
93  for (int m = -l; m <= l; m++, lm++) {
94  zp[lm] = (qmt(lm, ia) - qit(lm, ia)) * std::conj(zil_[l]) * gamma_factors_R_(l, iat);
95  }
96  }
97 
98  /* add pseudo_density to interstitial charge density so that rho(G) has the correct
99  * multipole moments in the muffin-tins */
100  #pragma omp parallel for schedule(static)
101  for (int igloc = 0; igloc < ctx_.gvec_count(); igloc++) {
102  int ig = ctx_.gvec_offset() + igloc;
103 
104  double gR = ctx_.gvec().gvec_len(ig) * R;
105 
106  double_complex zt = fourpi * std::conj(ctx_.gvec_phase_factor(ig, ia)) / unit_cell_.omega();
107 
108  if (ig) {
109  double_complex zt2(0, 0);
110  for (int l = 0, lm = 0; l <= ctx_.lmax_rho(); l++) {
111  double_complex zt1(0, 0);
112  for (int m = -l; m <= l; m++, lm++) {
113  zt1 += gvec_ylm_(lm, igloc) * zp[lm];
114  }
115  zt2 += zt1 * sbessel_mt_(l + pseudo_density_order_ + 1, iat, ctx_.gvec().shell(ig));
116  }
117  rho_pw[ig] += zt * zt2 * std::pow(2.0 / gR, pseudo_density_order_ + 1);
118  } else { /* for |G|=0 */
119  rho_pw[ig] += zt * y00 * (qmt(0, ia) - qit(0, ia));
120  }
121  }
122  }
123 
124  ctx_.comm().allgather(&rho_pw[0], ctx_.gvec_offset(), ctx_.gvec_count());
125 }
