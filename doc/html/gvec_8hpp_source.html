230  inline void find_z_columns(double Gmax__, FFT3D_grid const& fft_box__)
231  {
232  mdarray<int, 2> non_zero_columns(fft_box__.limits(0), fft_box__.limits(1));
233  non_zero_columns.zero();
234 
235  num_gvec_ = 0;
236  for (int i = fft_box__.limits(0).first; i <= fft_box__.limits(0).second; i++) {
237  for (int j = fft_box__.limits(1).first; j <= fft_box__.limits(1).second; j++) {
238  std::vector<int> zcol;
239 
240  /* in general case take z in [0, Nz) */
241  int zmax = fft_box__.size(2) - 1;
242  /* in case of G-vector reduction take z in [0, Nz/2] for {x=0,y=0} stick */
243  if (reduce_gvec_ && !i && !j) {
244  zmax = fft_box__.limits(2).second;
245  }
246  /* loop over z-coordinates of FFT grid */
247  for (int iz = 0; iz <= zmax; iz++) {
248  /* get z-coordinate of G-vector */
249  int k = fft_box__.gvec_by_coord(iz, 2);
250  /* take G+k */
251  auto vgk = lattice_vectors_ * (vector3d<double>(i, j, k) + vk_);
252  /* add z-coordinate of G-vector to the list */
253  if (vgk.length() <= Gmax__) {
254  zcol.push_back(k);
255  }
256  }
257 
258  if (zcol.size() && !non_zero_columns(i, j)) {
259  z_columns_.push_back(z_column_descriptor(i, j, zcol));
260  num_gvec_ += static_cast<int>(zcol.size());
261 
262  non_zero_columns(i, j) = 1;
263  if (reduce_gvec_) {
264  non_zero_columns(-i, -j) = 1;
265  }
266  }
267  }
268  }
269 
270  /* put column with {x, y} = {0, 0} to the beginning */
271  for (size_t i = 0; i < z_columns_.size(); i++) {
272  if (z_columns_[i].x == 0 && z_columns_[i].y == 0) {
273  std::swap(z_columns_[i], z_columns_[0]);
274  break;
275  }
276  }
277  }

279  inline void distribute_z_columns()
280  {
281  /* sort z-columns starting from the second */
282  std::sort(z_columns_.begin() + 1, z_columns_.end(),
283  [](z_column_descriptor const& a, z_column_descriptor const& b) { return a.z.size() > b.z.size(); });
284 
285  /* distribute z-columns between N ranks */
286  gvec_distr_ = block_data_descriptor(num_ranks_);
287  zcol_distr_ = block_data_descriptor(num_ranks_);
288  /* local number of z-columns for each rank */
289  std::vector<std::vector<z_column_descriptor>> zcols_local(num_ranks_);
290 
291  std::vector<int> ranks;
292  for (size_t i = 0; i < z_columns_.size(); i++) {
293  /* initialize the list of ranks to 0,1,2,... */
294  if (ranks.empty()) {
295  ranks.resize(num_ranks_);
296  std::iota(ranks.begin(), ranks.end(), 0);
297  }
298  /* find rank with minimum number of G-vectors */
299  auto rank_with_min_gvec = std::min_element(ranks.begin(), ranks.end(), [this](const int& a, const int& b) {
300  return gvec_distr_.counts[a] < gvec_distr_.counts[b];
301  });
302 
303  /* assign column to the found rank */
304  zcols_local[*rank_with_min_gvec].push_back(z_columns_[i]);
305  zcol_distr_.counts[*rank_with_min_gvec] += 1;
306  /* count local number of G-vectors */
307  gvec_distr_.counts[*rank_with_min_gvec] += static_cast<int>(z_columns_[i].z.size());
308  /* exclude this rank from the search */
309  ranks.erase(rank_with_min_gvec);
310  }
311  gvec_distr_.calc_offsets();
312  zcol_distr_.calc_offsets();
313 
314  /* save new ordering of z-columns */
315  z_columns_.clear();
316  for (int rank = 0; rank < num_ranks_; rank++) {
317  z_columns_.insert(z_columns_.end(), zcols_local[rank].begin(), zcols_local[rank].end());
318  }
319  }

321  inline void find_gvec_shells()
322  {
323  /* find G-shells */
324  std::map<size_t, std::vector<int>> gsh;
325  for (int ig = 0; ig < num_gvec_; ig++) {
326  /* take G+k */
327  auto gk = gkvec_cart(ig);
328  /* make some reasonable roundoff */
329  size_t len = size_t(gk.length() * 1e10);
330 
331  if (!gsh.count(len)) {
332  gsh[len] = std::vector<int>();
333  }
334  gsh[len].push_back(ig);
335  }
336  num_gvec_shells_ = static_cast<int>(gsh.size());
337  gvec_shell_ = mdarray<int, 1>(num_gvec_);
338  gvec_shell_len_ = mdarray<double, 1>(num_gvec_shells_);
339 
340  int n{0};
341  for (auto it = gsh.begin(); it != gsh.end(); it++) {
342  gvec_shell_len_(n) = static_cast<double>(it->first) * 1e-10;
343  for (int ig : it->second) {
344  gvec_shell_(ig) = n;
345  }
346  n++;
347  }
348  }

350  void init()
351  {
352  PROFILE("sddk::Gvec::init");
353 
354  auto fft_grid = FFT3D_grid(find_translations(Gmax_, lattice_vectors_) + vector3d<int>({2, 2, 2}));
355 
356  find_z_columns(Gmax_, fft_grid);
357 
358  distribute_z_columns();
359 
360  gvec_index_by_xy_ =
361  mdarray<int, 3>(2, fft_grid.limits(0), fft_grid.limits(1), memory_t::host, "Gvec.gvec_index_by_xy_");
362  std::fill(gvec_index_by_xy_.at<CPU>(), gvec_index_by_xy_.at<CPU>() + gvec_index_by_xy_.size(), -1);
363 
364  /* build the full G-vector index and reverse mapping */
365  gvec_full_index_ = mdarray<int, 1>(num_gvec_);
366  int ig{0};
367  for (size_t i = 0; i < z_columns_.size(); i++) {
368  /* starting G-vector index for a z-stick */
369  gvec_index_by_xy_(0, z_columns_[i].x, z_columns_[i].y) = ig;
370  /* size of a z-stick */
371  gvec_index_by_xy_(1, z_columns_[i].x, z_columns_[i].y) =
372  static_cast<int>((z_columns_[i].z.size() << 20) + i);
373  for (size_t j = 0; j < z_columns_[i].z.size(); j++) {
374  gvec_full_index_[ig++] = static_cast<int>((i << 12) + j);
375  }
376  }
377  if (ig != num_gvec_) {
378  TERMINATE("wrong G-vector count");
379  }
380  for (int ig = 0; ig < num_gvec_; ig++) {
381  auto gv = gvec(ig);
382  if (index_by_gvec(gv) != ig) {
383  std::stringstream s;
384  s << "wrong G-vector index: ig=" << ig << " gv=" << gv << " index_by_gvec(gv)=" << index_by_gvec(gv);
385  TERMINATE(s);
386  }
387  }
388 
389  /* first G-vector must be (0, 0, 0); never reomove this check!!! */
390  auto g0 = gvec_by_full_index(gvec_full_index_(0));
391  if (g0[0] || g0[1] || g0[2]) {
392  TERMINATE("first G-vector is not zero");
393  }
394 
395  find_gvec_shells();
396 
397  /* create default partition for G-vectors */
398  gvec_partition_ = std::unique_ptr<Gvec_partition>(new Gvec_partition(*this, *comm_fft_));
399  }

560  inline int index_by_gvec(vector3d<int> const& G__) const
561  {
562  if (reduced() && G__[0] == 0 && G__[1] == 0 && G__[2] < 0) {
563  return -1;
564  }
565  int ig0 = gvec_index_by_xy_(0, G__[0], G__[1]);
566  if (ig0 == -1) {
567  return -1;

570  int col_size = gvec_index_by_xy_(1, G__[0], G__[1]) >> 20;
571  int z0 = G__[2] - z_columns_[icol].z[0];
572  int offs = (z0 >= 0) ? z0 : z0 + col_size;
573  int ig = ig0 + offs;
574  assert(ig < num_gvec());
575  return ig;
576  }

614 inline void Gvec_partition::build_fft_distr()
615 {
616  /* calculate distribution of G-vectors and z-columns for the FFT communicator */
617  gvec_distr_fft_ = block_data_descriptor(fft_comm_->size());
618  zcol_distr_fft_ = block_data_descriptor(fft_comm_->size());
619 
620  int nrc = gvec_->num_ranks_ / fft_comm_->size();
621 
622  if (gvec_->num_ranks_ != nrc * fft_comm_->size()) {
623  TERMINATE("wrong number of MPI ranks");
624  }
625 
626  for (int rank = 0; rank < fft_comm_->size(); rank++) {
627  for (int i = 0; i < nrc; i++) {
628  /* fine-grained rank */
629  int r = rank * nrc + i;
630  gvec_distr_fft_.counts[rank] += gvec_->gvec_distr_.counts[r];
631  zcol_distr_fft_.counts[rank] += gvec_->zcol_distr_.counts[r];
632  }
633  }
634  /* get offsets of z-columns */
635  zcol_distr_fft_.calc_offsets();
636  /* get offsets of G-vectors */
637  gvec_distr_fft_.calc_offsets();
638 }

640 inline void Gvec_partition::calc_offsets()
641 {
642  zcol_offs_ = mdarray<int, 1>(gvec_->num_zcol(), memory_t::host, "Gvec_partition.zcol_offs_");
643  for (int rank = 0; rank < fft_comm_->size(); rank++) {
644  int offs{0};
645  /* loop over local number of z-columns */
646  for (int i = 0; i < zcol_distr_fft_.counts[rank]; i++) {
647  /* global index of z-column */
648  int icol = zcol_distr_fft_.offsets[rank] + i;
649  zcol_offs_[icol] = offs;
650  offs += static_cast<int>(gvec_->z_columns_[icol].z.size());
651  }
652  assert(offs == gvec_distr_fft_.counts[rank]);
653  }
654 }

656 inline void Gvec_partition::pile_gvec()
657 {
658  /* build a table of {offset, count} values for G-vectors in the swapped wfs;
659  * we are preparing to swap wave-functions from a default slab distribution to a FFT-friendly distribution
660  * +==============+ +----+----+----+
661  * | : : | I I I I
662  * +==============+ I....I....I....I
663  * | : : | -> I I I I
664  * +==============+ I....I....I....I
665  * | : : | I I I I
666  * +==============+ +----+----+----+
667  *
668  * i.e. we will make G-vector slabs more fat (pile-of-slabs) and at the same time reshulffle wave-functions
669  * between columns of the 2D MPI grid */
670  int rank_row = fft_comm_->rank();
671 
672  int nrc = gvec_->num_ranks_ / fft_comm_->size();
673  if (gvec_->num_ranks_ != nrc * fft_comm_->size()) {
674  TERMINATE("wrong number of MPI ranks");
675  }
676 
677  gvec_fft_slab_ = block_data_descriptor(nrc);
678  for (int i = 0; i < nrc; i++) {
679  gvec_fft_slab_.counts[i] = gvec_->gvec_count(rank_row * nrc + i);
680  }
681  gvec_fft_slab_.calc_offsets();
682 
683  assert(gvec_fft_slab_.offsets.back() + gvec_fft_slab_.counts.back() == gvec_distr_fft_.counts[rank_row]);
684 }
