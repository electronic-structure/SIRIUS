108  /// Map effective potential and magnetic field to a coarse FFT mesh in case of PP-PW.
109  /** \param [in] gvec_coarse G-vectors of the coarse FFT grid.
110  * \param [in] num_mag_dims Number of magnetic dimensions.
111  * \param [in] effective_potential \f$ V_{eff}({\bf r}) \f$ on the fine grid FFT grid.
112  * \param [in] effective_magnetic_field \f$ {\bf B}_{eff}({\bf r}) \f$ on the fine FFT grid.
113  *
114  * This function should be called prior to the band diagonalziation. In case of GPU execution all
115  * effective fields on the coarse grid will be copied to the device and will remain there until the
116  * dismiss() method is called after band diagonalization.
117  */
118  inline void prepare(Gvec const& gvec_coarse__,
119  int num_mag_dims__,
120  Periodic_function<double>* effective_potential__,
121  Periodic_function<double>* effective_magnetic_field__[3])
122  {
123  PROFILE("sirius::Local_operator::prepare");
124 
125  /* group effective fields into single vector */
126  std::vector<Periodic_function<double>*> veff_vec(num_mag_dims__ + 1);
127  veff_vec[0] = effective_potential__;
128  for (int j = 0; j < num_mag_dims__; j++) {
129  veff_vec[1 + j] = effective_magnetic_field__[j];
130  }
131 
132  /* allocate only once */
133  if (!veff_vec_.size()) {
134  veff_vec_ = mdarray<double, 2>(fft_coarse_.local_size(), num_mag_dims__ + 1, memory_t::host, "Local_operator::veff_vec_");
135  }
136 
137  /* low-frequency part of PW coefficients */
138  std::vector<double_complex> v_pw_coarse(gvec_coarse__.partition().gvec_count_fft());
139  /* prepare FFT for transformation */
140  fft_coarse_.prepare(gvec_coarse__.partition());
141  /* map components of effective potential to a corase grid */
142  for (int j = 0; j < num_mag_dims__ + 1; j++) {
143  /* loop over low-frequency G-vectors */
144  for (int ig = 0; ig < gvec_coarse__.partition().gvec_count_fft(); ig++) {
145  /* G-vector in fractional coordinates */
146  auto G = gvec_coarse__.gvec(ig + gvec_coarse__.partition().gvec_offset_fft());
147  v_pw_coarse[ig] = veff_vec[j]->f_pw(G);
148  }
149  /* transform to real space */
150  fft_coarse_.transform<1>(gvec_coarse__.partition(), &v_pw_coarse[0]);
151  /* save V(r) */
152  fft_coarse_.output(&veff_vec_(0, j));
153  }
154  fft_coarse_.dismiss();
155 
156  if (num_mag_dims__) {
157  for (int ir = 0; ir < fft_coarse_.local_size(); ir++) {
158  double v0 = veff_vec_(ir, 0);
159  double v1 = veff_vec_(ir, 1);
160  veff_vec_(ir, 0) = v0 + v1; // v + Bz
161  veff_vec_(ir, 1) = v0 - v1; // v - Bz
162  }
163  }
164 
165  if (num_mag_dims__ == 0) {
166  v0_[0] = veff_vec[0]->f_pw(0).real();
167  } else {
168  v0_[0] = veff_vec[0]->f_pw(0).real() + veff_vec[1]->f_pw(0).real();
169  v0_[1] = veff_vec[0]->f_pw(0).real() - veff_vec[1]->f_pw(0).real();
170  }
171 
172  /* copy veff to device */
173  #ifdef __GPU
174  if (fft_coarse_.pu() == GPU) {
175  veff_vec_.allocate(memory_t::device);
176  veff_vec_.copy_to_device();
177  }
178  #endif
179  }
180 
181  /// Map effective potential and magnetic field to a coarse FFT mesh in case of FP-LAPW.
182  /** \param [in] gvec_coarse G-vectors of the coarse FFT grid.
183  * \param [in] num_mag_dims Number of magnetic dimensions.
184  * \param [in] effective_potential \f$ V_{eff}({\bf r}) \f$ on the fine grid FFT grid.
185  * \param [in] effective_magnetic_field \f$ {\bf B}_{eff}({\bf r}) \f$ on the fine FFT grid.
186  * \param [in] step_function Unit step function of the LAPW method.
187  */
188  inline void prepare(Gvec const& gvec_coarse__,
189  int num_mag_dims__,
190  Periodic_function<double>* effective_potential__,
191  Periodic_function<double>* effective_magnetic_field__[3],
192  Step_function const& step_function__)
193  {
194  PROFILE("sirius::Local_operator::prepare");
195 
196  /* group effective fields into single vector */
197  std::vector<Periodic_function<double>*> veff_vec(num_mag_dims__ + 1);
198  veff_vec[0] = effective_potential__;
199  for (int j = 0; j < num_mag_dims__; j++) {
200  veff_vec[1 + j] = effective_magnetic_field__[j];
201  }
202 
203  /* allocate only once */
204  if (!veff_vec_.size()) {
205  veff_vec_ = mdarray<double, 2>(fft_coarse_.local_size(), num_mag_dims__ + 1, memory_t::host, "Local_operator::veff_vec_");
206  }
207 
208  if (!theta_.size()) {
209  theta_ = mdarray<double, 1>(fft_coarse_.local_size(), memory_t::host, "Local_operator::theta_");
210  }
211 
212  if (!buf_rg_.size()) {
213  buf_rg_ = mdarray<double_complex, 1>(fft_coarse_.local_size(), memory_t::host, "Local_operator::buf_rg_");
214  }
215 
216  auto& fft_dense = effective_potential__->fft();
217  auto& gvec_dense = effective_potential__->gvec();
218 
219  mdarray<double_complex, 1> v_pw_fine(gvec_dense.num_gvec());
220  /* low-frequency part of PW coefficients */
221  std::vector<double_complex> v_pw_coarse(gvec_coarse__.partition().gvec_count_fft());
222  /* prepare coarse-grained FFT for transformation */
223  fft_coarse_.prepare(gvec_coarse__.partition());
224  /* map components of effective potential to a corase grid */
225  for (int j = 0; j < num_mag_dims__ + 1; j++) {
226  for (int ir = 0; ir < fft_dense.local_size(); ir++) {
227  fft_dense.buffer(ir) = veff_vec[j]->f_rg(ir) * step_function__.theta_r(ir);
228  }
229  if (fft_dense.pu() == GPU) {
230  fft_dense.buffer().copy<memory_t::host, memory_t::device>();
231  }
232  fft_dense.transform<-1>(gvec_dense.partition(), &v_pw_fine[gvec_dense.partition().gvec_offset_fft()]);
233  fft_dense.comm().allgather(&v_pw_fine[0], gvec_dense.partition().gvec_offset_fft(),
234  gvec_dense.partition().gvec_count_fft());
235  if (j == 0) {
236  v0_[0] = v_pw_fine[0].real();
237  }
238  /* loop over low-frequency G-vectors */
239  for (int ig = 0; ig < gvec_coarse__.partition().gvec_count_fft(); ig++) {
240  /* G-vector in fractional coordinates */
241  auto G = gvec_coarse__.gvec(ig + gvec_coarse__.partition().gvec_offset_fft());
242  v_pw_coarse[ig] = v_pw_fine[gvec_dense.index_by_gvec(G)];
243  }
244 
245  fft_coarse_.transform<1>(gvec_coarse__.partition(), &v_pw_coarse[0]);
246  fft_coarse_.output(&veff_vec_(0, j));
247  }
248 
249  /* map unit-step function */
250  for (int ig = 0; ig < gvec_coarse__.partition().gvec_count_fft(); ig++) {
251  /* G-vector in fractional coordinates */
252  auto G = gvec_coarse__.gvec(ig + gvec_coarse__.partition().gvec_offset_fft());
253  v_pw_coarse[ig] = step_function__.theta_pw(gvec_dense.index_by_gvec(G));
254  }
255  fft_coarse_.transform<1>(gvec_coarse__.partition(), &v_pw_coarse[0]);
256  fft_coarse_.output(&theta_(0));
257  /* release FFT driver */
258  fft_coarse_.dismiss();
259 
260  if (fft_coarse_.pu() == GPU) {
261  veff_vec_.allocate(memory_t::device);
262  veff_vec_.copy<memory_t::host, memory_t::device>();
263  theta_.allocate(memory_t::device);
264  theta_.copy<memory_t::host, memory_t::device>();
265  buf_rg_.allocate(memory_t::device);
266  }
267 
268  if (param_->control().print_checksum_) {
269  auto cs = veff_vec_.checksum();
270  DUMP("checksum(veff_vec): %18.10f", cs);
271  auto cs1 = theta_.checksum();
272  DUMP("checksum(theta): %18.10f", cs1);
273  }
274  }

278  {
279  PROFILE("sirius::Local_operator::prepare");
280 
281  int ngv_fft = gkvec__.partition().gvec_count_fft();
282 
283  /* cache kinteic energy of plane-waves */
284  if (static_cast<int>(pw_ekin_.size()) < ngv_fft) {
285  pw_ekin_ = mdarray<double, 1>(ngv_fft, memory_t::host, "Local_operator::pw_ekin");
286  }
287  for (int ig_loc = 0; ig_loc < ngv_fft; ig_loc++) {
288  /* global index of G-vector */
289  int ig = gkvec__.partition().gvec_offset_fft() + ig_loc;
290  /* get G+k in Cartesian coordinates */
291  auto gv = gkvec__.gkvec_cart(ig);
292  pw_ekin_[ig_loc] = 0.5 * (gv * gv);
293  }
294 
295  if (static_cast<int>(vphi1_.size()) < ngv_fft) {
296  vphi1_ = mdarray<double_complex, 1>(ngv_fft, memory_t::host, "Local_operator::vphi1");
297  }
298  if (gkvec__.reduced() && static_cast<int>(vphi2_.size()) < ngv_fft) {
299  vphi2_ = mdarray<double_complex, 1>(ngv_fft, memory_t::host, "Local_operator::vphi2");
300  }
301 
302  #ifdef __GPU
303  if (fft_coarse_.pu() == GPU) {
304  pw_ekin_.allocate(memory_t::device);
305  pw_ekin_.copy_to_device();
306  vphi1_.allocate(memory_t::device);
307  if (gkvec__.reduced()) {
308  vphi2_.allocate(memory_t::device);

310  }

326  template <device_t data_ptr_type>
327  void apply_h(int ispn__, wave_functions& hphi__, int idx0__, int n__)
328  {
329  PROFILE("sirius::Local_operator::apply_h");
330 
331  auto& gkp = hphi__.gkvec().partition();
332  auto& comm_col = hphi__.gkvec().comm_ortho_fft();
333 
334  switch (data_ptr_type) {
335  case CPU: {
336  hphi__.pw_coeffs().remap_forward(gkp.gvec_fft_slab(), comm_col, n__, idx0__);
337  break;
338  }
339  case GPU: {
340  hphi__.pw_coeffs().remap_forward<memory_t::host | memory_t::device>(gkp.gvec_fft_slab(), comm_col, n__, idx0__);
341  break;
342  }
343  }
344 
345  int first{0};
346  /* if G-vectors are reduced, wave-functions are real and
347  * we can transform two of them at once */
348  if (gkp.reduced()) {
349  int npairs = hphi__.pw_coeffs().spl_num_col().local_size() / 2;
350 
351  for (int i = 0; i < npairs; i++) {
352  /* phi(G) -> phi(r) */
353  fft_coarse_.transform<1, data_ptr_type>(gkp,
354  hphi__.pw_coeffs().extra().at<data_ptr_type>(0, 2 * i),
355  hphi__.pw_coeffs().extra().at<data_ptr_type>(0, 2 * i + 1));
356 
357  /* multiply by effective potential */
358  if (fft_coarse_.pu() == GPU) {
359  #ifdef __GPU
360  scale_matrix_rows_gpu(fft_coarse_.local_size(), 1, fft_coarse_.buffer().at<GPU>(), veff_vec_.at<GPU>(0, ispn__));
361  #else
362  TERMINATE_NO_GPU
363  #endif
364  } else {
365  #pragma omp parallel for schedule(static)
366  for (int ir = 0; ir < fft_coarse_.local_size(); ir++) {
367  fft_coarse_.buffer(ir) *= veff_vec_(ir, ispn__);
368  }
369  }
370  /* V(r)phi(r) -> [V*phi](G) */
371  fft_coarse_.transform<-1, data_ptr_type>(gkp, vphi1_.at<data_ptr_type>(), vphi2_.at<data_ptr_type>());
372 
373  /* add kinetic energy */
374  switch (data_ptr_type) {
375  case CPU: {
376  #pragma omp parallel for schedule(static)
377  for (int ig = 0; ig < gkp.gvec_count_fft(); ig++) {
378  hphi__.pw_coeffs().extra()(ig, 2 * i) = hphi__.pw_coeffs().extra()(ig, 2 * i) * pw_ekin_[ig] + vphi1_[ig];
379  hphi__.pw_coeffs().extra()(ig, 2 * i + 1) = hphi__.pw_coeffs().extra()(ig, 2 * i + 1) * pw_ekin_[ig] + vphi2_[ig];
380  }
381  break;
382  }
383  case GPU: {
384  #ifdef __GPU
385  add_pw_ekin_gpu(gkp.gvec_count_fft(), pw_ekin_.at<GPU>(), vphi1_.at<GPU>(),
386  hphi__.pw_coeffs().extra().at<GPU>(0, 2 * i));
387  add_pw_ekin_gpu(gkp.gvec_count_fft(), pw_ekin_.at<GPU>(), vphi2_.at<GPU>(),
388  hphi__.pw_coeffs().extra().at<GPU>(0, 2 * i + 1));
389  #else
390  TERMINATE_NO_GPU
391  #endif
392  break;
393  }
394  }
395  }
396 
397  /* check if we have to do last wave-function which had no pair */
398  first = (hphi__.pw_coeffs().spl_num_col().local_size() % 2) ? hphi__.pw_coeffs().spl_num_col().local_size() - 1
399  : hphi__.pw_coeffs().spl_num_col().local_size();
400  }
401 
402  /* if we don't have G-vector reductions, first = 0 and we start a normal loop */
403  for (int i = first; i < hphi__.pw_coeffs().spl_num_col().local_size(); i++) {
404  /* phi(G) -> phi(r) */
405  fft_coarse_.transform<1, data_ptr_type>(gkp, hphi__.pw_coeffs().extra().at<data_ptr_type>(0, i));
406 
407  /* multiply by effective potential */
408  switch (fft_coarse_.pu()) {
409  case CPU: {
410  #pragma omp parallel for schedule(static)
411  for (int ir = 0; ir < fft_coarse_.local_size(); ir++) {
412  fft_coarse_.buffer(ir) *= veff_vec_(ir, ispn__);
413  }
414  break;
415  }
416  case GPU: {
417  #ifdef __GPU
418  scale_matrix_rows_gpu(fft_coarse_.local_size(), 1, fft_coarse_.buffer().at<GPU>(), veff_vec_.at<GPU>(0, ispn__));
419  #else
420  TERMINATE_NO_GPU
421  #endif
422  break;
423  }
424  }
425 
426  /* V(r)phi(r) -> [V*phi](G) */
427  fft_coarse_.transform<-1, data_ptr_type>(gkp, vphi1_.at<data_ptr_type>());
428 
429  /* add kinetic energy */
430  switch (data_ptr_type) {
431  case CPU: {
432  #pragma omp parallel for schedule(static)
433  for (int ig = 0; ig < gkp.gvec_count_fft(); ig++) {
434  hphi__.pw_coeffs().extra()(ig, i) = hphi__.pw_coeffs().extra()(ig, i) * pw_ekin_[ig] + vphi1_[ig];
435  }
436  break;
437  }
438  case GPU: {
439  #ifdef __GPU
440  add_pw_ekin_gpu(gkp.gvec_count_fft(), pw_ekin_.at<GPU>(), vphi1_.at<GPU>(),
441  hphi__.pw_coeffs().extra().at<GPU>(0, i));
442  #else
443  TERMINATE_NO_GPU
444  #endif
445  break;
446  }
447  }
448  }
449 
450  switch (data_ptr_type) {
451  case CPU: {
452  hphi__.pw_coeffs().remap_backward(gkp.gvec_fft_slab(), comm_col, n__, idx0__);
453  break;
454  }
455  case GPU: {
456  hphi__.pw_coeffs().remap_backward<memory_t::host | memory_t::device>(gkp.gvec_fft_slab(), comm_col, n__, idx0__);
457  break;
458  }
459  }
460  }

462  void apply_h_o(Gvec_partition const& gkvec_par__,
463  int N__,
464  int n__,
465  wave_functions& phi__,
466  wave_functions& hphi__,
467  wave_functions& ophi__)
468  {
469  PROFILE("sirius::Local_operator::apply_h_o");
470 
471  auto& comm_col = gkvec_par__.gvec().comm_ortho_fft();
472 
473  fft_coarse_.prepare(gkvec_par__);
474 
475  mdarray<double_complex, 1> buf_pw(gkvec_par__.gvec_count_fft());
476 
477  #ifdef __GPU
478  if (fft_coarse_.pu() == GPU) {
479  phi__.pw_coeffs().copy_to_host(N__, n__);
480  }
481  #endif
482 
483  if (param_->control().print_checksum_) {
484  auto cs = phi__.checksum_pw(N__, n__);
485  if (phi__.comm().rank() == 0) {
486  DUMP("checksum(phi_pw): %18.10f %18.10f", cs.real(), cs.imag());
487  }
488  }
489 
490  phi__.pw_coeffs().remap_forward(gkvec_par__.gvec_fft_slab(), comm_col, n__, N__);
491  hphi__.pw_coeffs().set_num_extra(gkvec_par__.gvec_count_fft(), comm_col, n__, N__);
492  ophi__.pw_coeffs().set_num_extra(gkvec_par__.gvec_count_fft(), comm_col, n__, N__);
493 
494  for (int j = 0; j < phi__.pw_coeffs().spl_num_col().local_size(); j++) {
495  if (fft_coarse_.pu() == CPU) {
496  /* phi(G) -> phi(r) */
497  fft_coarse_.transform<1>(gkvec_par__, phi__.pw_coeffs().extra().at<CPU>(0, j));
498  #pragma omp parallel for schedule(static)
499  for (int ir = 0; ir < fft_coarse_.local_size(); ir++) {
500  /* save phi(r) */
501  buf_rg_[ir] = fft_coarse_.buffer(ir);
502  /* multiply by step function */
503  fft_coarse_.buffer(ir) *= theta_[ir];
504  }
505  /* phi(r) * Theta(r) -> ophi(G) */
506  fft_coarse_.transform<-1>(gkvec_par__, ophi__.pw_coeffs().extra().at<CPU>(0, j));
507  #pragma omp parallel for schedule(static)
508  for (int ir = 0; ir < fft_coarse_.local_size(); ir++) {
509  /* multiply be effective potential, which itself was multiplied by the step function in constructor */
510  fft_coarse_.buffer(ir) = buf_rg_[ir] * veff_vec_(ir, 0);
511  }
512  /* phi(r) * Theta(r) * V(r) -> ophi(G) */
513  fft_coarse_.transform<-1>(gkvec_par__, hphi__.pw_coeffs().extra().at<CPU>(0, j));
514  }
515  #ifdef __GPU
516  if (fft_coarse_.pu() == GPU) {
517  /* phi(G) -> phi(r) */
518  fft_coarse_.transform<1>(gkvec_par__, phi__.pw_coeffs().extra().at<CPU>(0, j));
519  /* save phi(r) */
520  acc::copy(buf_rg_.at<GPU>(), fft_coarse_.buffer().at<GPU>(), fft_coarse_.local_size());
521  /* multiply by step function */
522  scale_matrix_rows_gpu(fft_coarse_.local_size(), 1, fft_coarse_.buffer().at<GPU>(), theta_.at<GPU>());
523  /* phi(r) * Theta(r) -> ophi(G) */
524  fft_coarse_.transform<-1>(gkvec_par__, ophi__.pw_coeffs().extra().at<CPU>(0, j));
525  /* multiply by effective potential */
526  scale_matrix_rows_gpu(fft_coarse_.local_size(), 1, buf_rg_.at<GPU>(), veff_vec_.at<GPU>());
527  /* copy phi(r) * Theta(r) * V(r) to GPU buffer */
528  acc::copy(fft_coarse_.buffer().at<GPU>(), buf_rg_.at<GPU>(), fft_coarse_.local_size());
529  /* phi(r) * Theta(r) * V(r) -> ophi(G) */
530  fft_coarse_.transform<-1>(gkvec_par__, hphi__.pw_coeffs().extra().at<CPU>(0, j));
531  }
532  #endif
533 
534  /* add kinetic energy */
535  for (int x: {0, 1, 2}) {
536  for (int igloc = 0; igloc < gkvec_par__.gvec_count_fft(); igloc++) {
537  /* global index of G-vector */
538  int ig = gkvec_par__.gvec_offset_fft() + igloc;
539  /* \hat P phi = phi(G+k) * (G+k), \hat P is momentum operator */
540  buf_pw[igloc] = phi__.pw_coeffs().extra()(igloc, j) * gkvec_par__.gvec().gkvec_cart(ig)[x];
541  }
542  /* transform Cartesian component of wave-function gradient to real space */
543  fft_coarse_.transform<1>(gkvec_par__, &buf_pw[0]);
544  switch (fft_coarse_.pu()) {
545  case CPU: {
546  #pragma omp parallel for schedule(static)
547  for (int ir = 0; ir < fft_coarse_.local_size(); ir++) {
548  /* multiply be step function */
549  fft_coarse_.buffer(ir) *= theta_[ir];
550  }
551  break;
552  }
553  case GPU: {
554  #ifdef __GPU
555  /* multiply by step function */
556  scale_matrix_rows_gpu(fft_coarse_.local_size(), 1, fft_coarse_.buffer().at<GPU>(), theta_.at<GPU>());
557  #endif
558  break;
559  }
560  }
561  /* transform back to PW domain */
562  fft_coarse_.transform<-1>(gkvec_par__, &buf_pw[0]);
563  for (int igloc = 0; igloc < gkvec_par__.gvec_count_fft(); igloc++) {
564  int ig = gkvec_par__.gvec_offset_fft() + igloc;
565  hphi__.pw_coeffs().extra()(igloc, j) += 0.5 * buf_pw[igloc] * gkvec_par__.gvec().gkvec_cart(ig)[x];
566  }
567  }
568  }
569 
570  hphi__.pw_coeffs().remap_backward(gkvec_par__.gvec_fft_slab(), comm_col, n__, N__);
571  ophi__.pw_coeffs().remap_backward(gkvec_par__.gvec_fft_slab(), comm_col, n__, N__);
572 
573  fft_coarse_.dismiss();
574 
575  #ifdef __GPU
576  if (fft_coarse_.pu() == GPU) {
577  hphi__.pw_coeffs().copy_to_device(N__, n__);
578  ophi__.pw_coeffs().copy_to_device(N__, n__);
579  }
580  #endif
581  if (param_->control().print_checksum_) {
582  auto cs1 = hphi__.checksum_pw(N__, n__);
583  auto cs2 = ophi__.checksum_pw(N__, n__);
584  if (phi__.comm().rank() == 0) {
585  DUMP("checksum(hphi_pw): %18.10f %18.10f", cs1.real(), cs1.imag());
586  DUMP("checksum(ophi_pw): %18.10f %18.10f", cs2.real(), cs2.imag());
587  }
588  }

658  /// Apply magnetic field to the wave-functions.
659  /** In case of collinear magnetism only Bz is applied to <tt>phi</tt> and stored in the first component of

661  * component of <tt>bphi</tt>. The second componet of <tt>bphi</tt> is used to store -Bz|phi>. */
662  void apply_b(Gvec_partition const& gkvec_par__,
663  int N__,
664  int n__,
665  wave_functions& phi__,
666  std::vector<wave_functions>& bphi__)
667  {
668  PROFILE("sirius::Local_operator::apply_b");
669 
670  auto& comm_col = gkvec_par__.gvec().comm_ortho_fft();
671 
672  fft_coarse_.prepare(gkvec_par__);
673 
674  //#ifdef __GPU
675  //if (fft_coarse_.pu() == GPU) {
676  // phi__.pw_coeffs().copy_to_host(N__, n__);
677  //}
678  //#endif
679 
680  /* components of H|psi> to which H is applied */
681  std::vector<int> iv(1, 0);
682  if (bphi__.size() == 3) {
683  iv.push_back(2);
684  }
685 
686  phi__.pw_coeffs().remap_forward(gkvec_par__.gvec_fft_slab(), comm_col, n__, N__);
687  for (int i: iv) {
688  bphi__[i].pw_coeffs().set_num_extra(gkvec_par__.gvec_count_fft(), comm_col, n__, N__);
689  }
690 
691  for (int j = 0; j < phi__.pw_coeffs().spl_num_col().local_size(); j++) {
692  if (fft_coarse_.pu() == CPU) {
693  /* phi(G) -> phi(r) */
694  fft_coarse_.transform<1>(gkvec_par__, phi__.pw_coeffs().extra().at<CPU>(0, j));
695  /* save phi(r) */
696  if (bphi__.size() == 3) {

700  for (int ir = 0; ir < fft_coarse_.local_size(); ir++) {
701  /* multiply by Bz */

703  }

705  fft_coarse_.transform<-1>(gkvec_par__, bphi__[0].pw_coeffs().extra().at<CPU>(0, j));
706  /* non-collinear case */
707  if (bphi__.size() == 3) {
708  #pragma omp parallel for schedule(static)
709  for (int ir = 0; ir < fft_coarse_.local_size(); ir++) {
710  /* multiply by Bx-iBy */

712  }
713  /* phi(r) * (Bx(r)-iBy(r)) -> bphi[2](G) */
714  fft_coarse_.transform<-1>(gkvec_par__, bphi__[2].pw_coeffs().extra().at<CPU>(0, j));
715  }
716  } else {
717  #ifdef __GPU
718  /* phi(G) -> phi(r) */
719  fft_coarse_.transform<1>(gkvec_par__, phi__.pw_coeffs().extra().at<CPU>(0, j));
720  /* multiply by Bz */
721  scale_matrix_rows_gpu(fft_coarse_.local_size(), 1, fft_coarse_.buffer().at<GPU>(), veff_vec_.at<GPU>(0, 1));
722  /* phi(r) * Bz(r) -> bphi[0](G) */
723  fft_coarse_.transform<-1>(gkvec_par__, bphi__[0].pw_coeffs().extra().at<CPU>(0, j));
724  #else
725  TERMINATE_NO_GPU
726  #endif
727  }
728  }
729 
730  for (int i: iv) {
731  bphi__[i].pw_coeffs().remap_backward(gkvec_par__.gvec_fft_slab(), comm_col, n__, N__);
732  //#ifdef __GPU
733  //if (fft_coarse_.pu() == GPU) {
734  // bphi__[i].pw_coeffs().copy_to_device(N__, n__);
735  //}
736  //#endif
737  }
738 
739  fft_coarse_.dismiss();
740  }
