25 inline void K_point::generate_fv_states()
26 {
27  PROFILE("sirius::K_point::generate_fv_states");
28 
29  if (!ctx_.full_potential()) {
30  return;
31  }
32 
33  #ifdef __GPU
34  if (ctx_.processing_unit() == GPU) {
35  fv_eigen_vectors_slab().pw_coeffs().allocate_on_device();
36  fv_eigen_vectors_slab().pw_coeffs().copy_to_device(0, ctx_.num_fv_states());
37  }
38  #endif
39 
40  mdarray<double_complex, 2> alm(num_gkvec_loc(), unit_cell_.max_mt_aw_basis_size(), memory_t::host_pinned);
41  mdarray<double_complex, 2> tmp(unit_cell_.max_mt_aw_basis_size(), ctx_.num_fv_states());
42 
43  #ifdef __GPU
44  if (ctx_.processing_unit() == GPU) {
45  alm.allocate(memory_t::device);
46  tmp.allocate(memory_t::device);
47  }
48  #endif
49 
50  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
51  auto location = fv_eigen_vectors_slab().spl_num_atoms().location(ia);
52  /* number of alm coefficients for atom */
53  int mt_aw_size = unit_cell_.atom(ia).mt_aw_basis_size();
54  int mt_lo_size = unit_cell_.atom(ia).mt_lo_basis_size();
55  /* generate matching coefficients for all G-vectors */
56  alm_coeffs_loc_->generate(ia, alm);
57 
58  double_complex* tmp_ptr_gpu = (ctx_.processing_unit() == GPU) ? tmp.at<GPU>() : nullptr;
59  mdarray<double_complex, 2> tmp1(tmp.at<CPU>(), tmp_ptr_gpu, mt_aw_size, ctx_.num_fv_states());
60 
61  /* compute F(lm, i) = A(lm, G)^{T} * evec(G, i) for a single atom */
62  if (ctx_.processing_unit() == CPU) {
63  linalg<CPU>::gemm(1, 0, mt_aw_size, ctx_.num_fv_states(), num_gkvec_loc(),
64  alm.at<CPU>(), alm.ld(),
65  fv_eigen_vectors_slab().pw_coeffs().prime().at<CPU>(), fv_eigen_vectors_slab().pw_coeffs().prime().ld(),
66  tmp1.at<CPU>(), tmp1.ld());
67  }
68  #ifdef __GPU
69  if (ctx_.processing_unit() == GPU) {
70  alm.copy_to_device(mt_aw_size * num_gkvec_loc());
71  linalg<GPU>::gemm(1, 0, mt_aw_size, ctx_.num_fv_states(), num_gkvec_loc(),
72  alm.at<GPU>(), alm.ld(),
73  fv_eigen_vectors_slab().pw_coeffs().prime().at<GPU>(), fv_eigen_vectors_slab().pw_coeffs().prime().ld(),
74  tmp1.at<GPU>(), tmp1.ld());
75  tmp1.copy_to_host();
76  }
77  #endif
78 
79  comm_.reduce(tmp1.at<CPU>(), static_cast<int>(tmp1.size()), location.rank);
80 
81  #ifdef __PRINT_OBJECT_CHECKSUM
82  auto z1 = tmp1.checksum();
83  DUMP("checksum(tmp1): %18.10f %18.10f", std::real(z1), std::imag(z1));
84  #endif
85 
86  if (location.rank == comm_.rank()) {
87  int offset1 = fv_states().offset_mt_coeffs(location.local_index);
88  int offset2 = fv_eigen_vectors_slab().offset_mt_coeffs(location.local_index);
89  for (int i = 0; i < ctx_.num_fv_states(); i++) {
90  /* aw block */
91  std::memcpy(fv_states().mt_coeffs().prime().at<CPU>(offset1, i),
92  tmp1.at<CPU>(0, i),
93  mt_aw_size * sizeof(double_complex));
94  /* lo block */
95  if (mt_lo_size) {
96  std::memcpy(fv_states().mt_coeffs().prime().at<CPU>(offset1 + mt_aw_size, i),
97  fv_eigen_vectors_slab().mt_coeffs().prime().at<CPU>(offset2, i),
98  mt_lo_size * sizeof(double_complex));
99  }
100  }
101  }
102  }
103 
104  #pragma omp parallel for
105  for (int i = 0; i < ctx_.num_fv_states(); i++) {
106  /* G+k block */
107  std::memcpy(fv_states().pw_coeffs().prime().at<CPU>(0, i),
108  fv_eigen_vectors_slab().pw_coeffs().prime().at<CPU>(0, i),
109  num_gkvec_loc() * sizeof(double_complex));
110  }
111 
112  #ifdef __GPU
113  if (ctx_.processing_unit() == GPU) {
114  fv_eigen_vectors_slab().pw_coeffs().deallocate_on_device();
115  }
116  #endif
117 }
