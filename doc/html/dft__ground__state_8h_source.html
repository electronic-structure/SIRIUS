94  /// Return nucleus energy in the electrostatic field.
95  /** Compute energy of nucleus in the electrostatic potential generated by the total (electrons + nuclei)
96  * charge density. Diverging self-interaction term z*z/|r=0| is excluded. */
97  double energy_enuc() const
98  {
99  double enuc{0};
100  if (ctx_.full_potential()) {
101  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
102  int ia = unit_cell_.spl_num_atoms(ialoc);
103  int zn = unit_cell_.atom(ia).zn();
104  enuc -= 0.5 * zn * potential_.vh_el(ia) * y00;
105  }
106  ctx_.comm().allreduce(&enuc, 1);
107  }
108  return enuc;

180  /// Total energy of the electronic subsystem.
181  /** From the definition of the density functional we have:
182  *
183  * \f[
184  * E[\rho] = T[\rho] + E^{H}[\rho] + E^{XC}[\rho] + E^{ext}[\rho]
185  * \f]
186  * where \f$ T[\rho] \f$ is the kinetic energy, \f$ E^{H}[\rho] \f$ - electrostatic energy of
187  * electron-electron density interaction, \f$ E^{XC}[\rho] \f$ - exchange-correlation energy
188  * and \f$ E^{ext}[\rho] \f$ - energy in the external field of nuclei.
189  *
190  * Electrostatic and external field energies are grouped in the following way:
191  * \f[
192  * \frac{1}{2} \int \int \frac{\rho({\bf r})\rho({\bf r'}) d{\bf r} d{\bf r'}}{|{\bf r} - {\bf r'}|} +
193  * \int \rho({\bf r}) V^{nuc}({\bf r}) d{\bf r} = \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} +
194  * \frac{1}{2} \int \rho({\bf r}) V^{nuc}({\bf r}) d{\bf r}
195  * \f]
196  * Here \f$ V^{H}({\bf r}) \f$ is the total (electron + nuclei) electrostatic potential returned by the
197  * poisson solver. Next we transform the remaining term:
198  * \f[
199  * \frac{1}{2} \int \rho({\bf r}) V^{nuc}({\bf r}) d{\bf r} =
200  * \frac{1}{2} \int \int \frac{\rho({\bf r})\rho^{nuc}({\bf r'}) d{\bf r} d{\bf r'}}{|{\bf r} - {\bf r'}|} =
201  * \frac{1}{2} \int V^{H,el}({\bf r}) \rho^{nuc}({\bf r}) d{\bf r}
202  * \f]
203  */
204  double total_energy()
205  {
206  double tot_en{0};
207 
208  switch (ctx_.esm_type()) {
209  case electronic_structure_method_t::full_potential_lapwlo:
210  case electronic_structure_method_t::full_potential_pwlo: {
211  tot_en = (energy_kin() + energy_exc() + 0.5 * energy_vha() + energy_enuc());
212  break;
213  }
214 
215  case electronic_structure_method_t::pseudopotential: {
216  tot_en = (kset_.valence_eval_sum() - energy_veff() + energy_vloc() - potential_.PAW_one_elec_energy()) +
217  0.5 * energy_vha() + energy_exc() + potential_.PAW_total_energy() + ewald_energy_;
218  break;
219  }
220  }
221 
222  return tot_en;
223  }

313 inline double DFT_ground_state::ewald_energy()
314 {
315  PROFILE("sirius::DFT_ground_state::ewald_energy");
316 
317  double alpha = 1.5;
318 
319  double ewald_g = 0;
320 
321  #pragma omp parallel
322  {
323  double ewald_g_pt = 0;
324 
325  #pragma omp for
326  for (int igloc = 0; igloc < ctx_.gvec_count(); igloc++) {
327  int ig = ctx_.gvec_offset() + igloc;
328 
329  double g2 = std::pow(ctx_.gvec().shell_len(ctx_.gvec().shell(ig)), 2);
330 
331  double_complex rho(0, 0);
332 
333  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
334  rho += ctx_.gvec_phase_factor(ig, ia) * static_cast<double>(unit_cell_.atom(ia).zn());
335  }
336 
337  if (ig) {
338  ewald_g_pt += std::pow(std::abs(rho), 2) * std::exp(-g2 / 4 / alpha) / g2;
339  } else {
340  ewald_g_pt -= std::pow(unit_cell_.num_electrons(), 2) / alpha / 4; // constant term in QE comments
341  }
342 
343  if (ctx_.gvec().reduced() && ig) {
344  rho = 0;
345  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
346  rho += std::conj(ctx_.gvec_phase_factor(ig, ia)) * static_cast<double>(unit_cell_.atom(ia).zn());
347  }
348 
349  ewald_g_pt += std::pow(std::abs(rho), 2) * std::exp(-g2 / 4 / alpha) / g2;
350  }
351  }
352 
353  #pragma omp critical
354  ewald_g += ewald_g_pt;
355  }
356  ctx_.comm().allreduce(&ewald_g, 1);
357  ewald_g *= (twopi / unit_cell_.omega());
358 
359  /* remove self-interaction */
360  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
361  ewald_g -= std::sqrt(alpha / pi) * std::pow(unit_cell_.atom(ia).zn(), 2);
362  }
363 
364  double ewald_r = 0;
365  #pragma omp parallel
366  {
367  double ewald_r_pt = 0;
368 
369  #pragma omp for
370  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
371  for (int i = 1; i < unit_cell_.num_nearest_neighbours(ia); i++) {
372  int ja = unit_cell_.nearest_neighbour(i, ia).atom_id;
373  double d = unit_cell_.nearest_neighbour(i, ia).distance;
374  ewald_r_pt += 0.5 * unit_cell_.atom(ia).zn() * unit_cell_.atom(ja).zn() *
375  gsl_sf_erfc(std::sqrt(alpha) * d) / d;
376  }
377  }
378 
379  #pragma omp critical
380  ewald_r += ewald_r_pt;
381  }
382 
383  return (ewald_g + ewald_r);
384 }

704 /** \page DFT Spin-polarized DFT
705  * \section section1 Preliminary notes
706  *
707  * \note Here and below sybol \f$ {\boldsymbol \sigma} \f$ is reserved for the vector of Pauli matrices. Spin components
708  * are labeled with \f$ \alpha \f$ or \f$ \beta\f$.
709  *
710  * Wave-function of spin-1/2 particle is a two-component spinor:
711  * \f[
712  * {\bf \varphi}({\bf r})=\left( \begin{array}{c} \varphi_1({\bf r}) \\ \varphi_2({\bf r}) \end{array} \right)
713  * \f]
714  * Operator of spin:
715  * \f[
716  * {\bf \hat S}=\frac{\hbar}{2}{\bf \sigma},
717  * \f]
718  * Pauli matrices:
719  * \f[
720  * \sigma_x=\left( \begin{array}{cc}
721  * 0 & 1 \\
722  * 1 & 0 \\ \end{array} \right) \,
723  * \sigma_y=\left( \begin{array}{cc}
724  * 0 & -i \\
725  * i & 0 \\ \end{array} \right) \,
726  * \sigma_z=\left( \begin{array}{cc}
727  * 1 & 0 \\
728  * 0 & -1 \\ \end{array} \right)
729  * \f]
730  *
731  * \section section2 Density and magnetization
732  * Density is defined as:
733  * \f[
734  * \rho({\bf r}) = \sum_{j}^{occ} \Psi_{j}^{*}({\bf r}){\bf I} \Psi_{j}({\bf r}) =
735  * \sum_{j}^{occ} \psi_{j}^{\uparrow *} \psi_{j}^{\uparrow} + \psi_{j}^{\downarrow *} \psi_{j}^{\downarrow}
736  * \f]
737  * Magnetization is defined as:
738  * \f[
739  * {\bf m}({\bf r}) = \sum_{j}^{occ} \Psi_{j}^{*}({\bf r}) {\boldsymbol \sigma} \Psi_{j}({\bf r})
740  * \f]
741  * \f[
742  * m_x({\bf r}) = \sum_{j}^{occ} \psi_{j}^{\uparrow *} \psi_{j}^{\downarrow} + \psi_{j}^{\downarrow *} \psi_{j}^{\uparrow}
743  * \f]
744  * \f[
745  * m_y({\bf r}) = \sum_{j}^{occ} -i \psi_{j}^{\uparrow *} \psi_{j}^{\downarrow} + i \psi_{j}^{\downarrow *} \psi_{j}^{\uparrow}
746  * \f]
747  * \f[
748  * m_z({\bf r}) = \sum_{j}^{occ} \psi_{j}^{\uparrow *} \psi_{j}^{\uparrow} - \psi_{j}^{\downarrow *} \psi_{j}^{\downarrow}
749  * \f]
750  * Density matrix is defined as:
751  * \f[
752  * {\boldsymbol \rho}({\bf r}) = \frac{1}{2} \Big( {\bf I}\rho({\bf r}) + {\boldsymbol \sigma} {\bf m}({\bf r})\Big) =
753  * \frac{1}{2} \sum_{j}^{occ} \left( \begin{array}{cc} \psi_{j}^{\uparrow *} \psi_{j}^{\uparrow} &
754  * \psi_{j}^{\downarrow *} \psi_{j}^{\uparrow} \\
755  * \psi_{j}^{\uparrow *} \psi_{j}^{\downarrow} &
756  * \psi_{j}^{\downarrow *} \psi_{j}^{\downarrow} \end{array} \right)
757  * \f]
758  * Pay attention to the order of spin indices in the \f$ 2 \times 2 \f$ density matrix:
759  * \f[
760  * \rho_{\alpha \beta}({\bf r}) = \frac{1}{2} \sum_{j}^{occ} \psi_{j}^{\beta *}({\bf r})\psi_{j}^{\alpha}({\bf r})
761  * \f]
762  */

765 \page stress Stress tensor
766 \section section1 Preliminary notes
767 Derivative of the G-vector in Cartesian coordinates over the lattice vector components:
768 \f[
769  \frac{\partial G_{\beta}}{\partial a_{\mu\nu}} + ({\bf a}^{-1})_{\nu \beta} G_{\mu} = 0
770 \f]
771 Mathematica proof script:
772 \verbatim
773 A = Table[Subscript[a, i, j], {i, 1, 3}, {j, 1, 3}];
774 invA = Inverse[A];
775 B = 2*Pi*Transpose[Inverse[A]];
776 G = Table[Subscript[g, i], {i, 1, 3}];
777 gvec = B.G;
778 Do[
779  Print[FullSimplify[
780  D[gvec[[beta]], Subscript[a, mu, nu]] + invA[[nu]][[beta]]*gvec[[mu]]]],
781 {beta, 1, 3}, {mu, 1, 3}, {nu, 1,3}]
782 \endverbatim
783 Another relation:
784 \f[
785  \frac{\partial}{\partial a_{\mu \nu}} \frac{1}{\sqrt{\Omega}} + \frac{1}{2} \frac{1}{\sqrt{\Omega}} ({\bf a}^{-1})_{\nu \mu} = 0
786 \f]
787 Mathematica proof script:
788 \verbatim
789 A = Table[Subscript[a, i, j], {i, 1, 3}, {j, 1, 3}];
790 invA = Inverse[A];
791 Do[
792  Print[FullSimplify[
793  D[1/Sqrt[Det[A]], Subscript[a, mu, nu]] + (1/2)*(1/Sqrt[Det[A]]) * invA[[nu]][[mu]]
794  ]
795  ],
796 {mu, 1, 3}, {nu, 1, 3}]
797 \endverbatim
798 
799 Derivative of the G-vector real spherical harmonics over the lattice vector components:
800 \f[
801  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial a_{\mu \nu}} =
802  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \theta} \frac{\partial \theta} {\partial a_{\mu \nu}} +
803  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \phi} \frac{\partial \phi} {\partial a_{\mu \nu}}
804 \f]
805 Derivatives of the \f$ R_{\ell m} \f$ with respect to the \f$ \theta,\, \phi\f$ angles can be tabulated up to a given \f$ \ell_{max} \f$.
806 The derivatives of angles are computed as following:
807 \f[
808  \frac{\partial \theta} {\partial a_{\mu \nu}} = \sum_{\beta=1}^{3} \frac{\partial \theta}{\partial G_{\beta}} \frac{\partial G_{\beta}} {\partial a_{\mu \nu}}
809 \f]
810 \f[
811  \frac{\partial \phi} {\partial a_{\mu \nu}} = \sum_{\beta=1}^{3} \frac{\partial \phi}{\partial G_{\beta}} \frac{\partial G_{\beta}} {\partial a_{\mu \nu}}
812 \f]
813 where
814 \f[
815  \frac{\partial \theta}{\partial G_{x}} = \frac{\cos(\phi) \cos(\theta)}{G} \\
816  \frac{\partial \theta}{\partial G_{y}} = \frac{\cos(\theta) \sin(\phi)}{G} \\
817  \frac{\partial \theta}{\partial G_{z}} = -\frac{\sin(\theta)}{G}
818 \f]
819 and
820 \f[
821  \frac{\partial \phi}{\partial G_{x}} = -\frac{\sin(\phi)}{\sin(\theta) G} \\
822  \frac{\partial \phi}{\partial G_{y}} = \frac{\cos(\phi)}{\sin(\theta) G} \\
823  \frac{\partial \phi}{\partial G_{z}} = 0
824 \f]
825 The derivative of \f$ phi \f$ has discontinuities at \f$ \theta = 0, \theta=\pi \f$. This, however, is not a problem, because
826 multiplication by the the derivative of \f$ R_{\ell m} \f$ removes it. The following functions have to be hardcoded:
827 \f[
828  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \theta} \\
829  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \phi} \frac{1}{\sin(\theta)}
830 \f]
831 
832 Derivatives of the spherical Bessel functions are computed in the same fashion:
833 \f[
834  \frac{\partial j_{\ell}(Gx)}{\partial a_{\mu \nu}} =
835  \frac{\partial j_{\ell}(Gx)}{\partial G} \frac{\partial G} {\partial a_{\mu \nu}} =
836  \frac{\partial j_{\ell}(Gx)}{\partial G} \sum_{\beta=1}^{3}\frac{\partial G}{\partial G_{\beta}} \frac{\partial G_{\beta}} {\partial a_{\mu \nu}}
837 \f]
838 The derivatives of \f$ G \f$ are:
839 \f[
840  \frac{\partial G}{\partial G_{x}} = \sin(\theta)\cos(\phi) \\
841  \frac{\partial G}{\partial G_{y}} = \sin(\theta)\sin(\phi) \\
842  \frac{\partial G}{\partial G_{z}} = \cos(\theta)
843 \f]
844 
845 Let's write the full expression for the derivative of beta-projector matrix elements with respect to lattice vector
846 components:
847 \f[
848  \frac{\partial \langle {\bf G+k}|\beta_{\ell m} \rangle} {\partial a_{\mu \nu}} =
849  \frac{\partial} {\partial a_{\mu \nu}} \frac{4\pi}{\sqrt{\Omega}}(-i)^{\ell} R_{\ell m}(\theta_{G+k}, \phi_{G+k}) \int \beta_{\ell}(r) j_{\ell}(Gr) r^2 dr =\\
850  \frac{4\pi}{\sqrt{\Omega}} (-i)^{\ell} \Bigg[ \int \beta_{\ell}(r) j_{\ell}(Gr) r^2 dr
851  \Big( \frac{\partial R_{\ell m}(\theta, \phi)}{\partial a_{\mu \nu}} - \frac{1}{2} R_{\ell m}(\theta, \phi) ({\bf a}^{-1})_{\nu \mu} \Big) +
852  R_{\ell m}(\theta, \phi) \int \beta_{\ell}(r) \frac{\partial j_{\ell}(Gr)}{\partial a_{\mu \nu}} r^2 dr \Bigg]
853 \f]
