33 /// MPI grid interface
34 /** The following terminology is used. Suppose we have a 4x5 grid of MPI ranks. We say it's a two-\em dimensional
35  * grid with the first dimension of the size 4 and the second dimensoion of the size 5. The \em actual number of
36  * grid dimensions is two, however we may also consider the grid as being a D-dimensional (D >= 2) with implicit
37  * dimension sizes equal to one, e.g. 4x5 := 4x5x1x1x1... The communication happens along single or multiple
38  * \em directions along the grid dimensions. We specify directions wth bits, eg. directions=00000101 reads as
39  * "communication along 1-st and 3-rd dimensions".
40  * \image html mpi_grid_comm.png "Communication along dimension d0 (between ranks of d0)."
41  * In the provided example the corresponding communicator is MPI_grid::communicator(1 << d0), where d0 is the integer
42  * index of dimension.
43  */
44 class MPI_grid

61  * First communicator is the trivial "self" communicator; the last communicator handles the entire grid. */
62  std::vector<Communicator> communicators_;

71  inline int valid_directions(int directions__) const
72  {
73  return (directions__ & ((1 << dimensions_.size()) - 1));
74  }

76  /// Initialize the grid.
77  void initialize()
78  {
79  PROFILE("sddk::MPI_grid::initialize");
80 
81  if (dimensions_.size() == 0) {
82  TERMINATE("no dimensions provided for the MPI grid");
83  }
84 
85  int sz{1};
86  for (size_t i = 0; i < dimensions_.size(); i++) {
87  sz *= dimensions_[i];
88  }
89 
90  if (parent_communicator_.size() != sz) {
91  std::stringstream s;
92  s << "Number of MPI ranks doesn't match the size of the grid." << std::endl << " grid dimensions :";
93  for (auto& e : dimensions_) {
94  s << " " << e;
95  }
96  s << std::endl << " available number of MPI ranks : " << parent_communicator_.size();
97 
98  TERMINATE(s);
99  }
100 
101  /* communicator of the entire grid */
102  std::vector<int> periods(dimensions_.size(), 0);
103  CALL_MPI(MPI_Cart_create, (parent_communicator_.mpi_comm(), (int)dimensions_.size(), &dimensions_[0],
104  &periods[0], 0, &base_grid_communicator_.mpi_comm()));
105 
106  /* total number of communicators inside the grid */
107  int num_comm = 1 << dimensions_.size();
108 
109  communicators_ = std::vector<Communicator>(num_comm); //.resize(num_comm);
110 
111  coordinates_ = std::vector<int>(dimensions_.size(), -1);
112 
113  communicator_size_ = std::vector<int>(num_comm, 0);
114 
115  communicator_root_ = std::vector<bool>(num_comm, false);
116 
117  /* get coordinates */
118  CALL_MPI(MPI_Cart_get, (base_grid_communicator_.mpi_comm(), (int)dimensions_.size(), &dimensions_[0],
119  &periods[0], &coordinates_[0]));
120 
121  /* get all possible communicators */
122  for (int i = 1; i < num_comm; i++) {
123  bool is_root = true;
124  int comm_size = 1;
125  std::vector<int> flg(dimensions_.size(), 0);
126 
127  /* each bit represents a directions */
128  for (int j = 0; j < (int)dimensions_.size(); j++) {
129  if (i & (1 << j)) {
130  flg[j] = 1;
131  is_root = is_root && (coordinates_[j] == 0);
132  comm_size *= dimensions_[j];
133  }
134  }
135 
136  communicator_root_[i] = is_root;
137 
138  communicator_size_[i] = comm_size;
139 
140  /* subcommunicators */
141  CALL_MPI(MPI_Cart_sub, (base_grid_communicator_.mpi_comm(), &flg[0], &communicators_[i].mpi_comm()));
142  }
143 
144  /* explicitly set the size of "self" communicator */
145  communicator_size_[0] = 1;
146 
147  /* explicitly set the root of "self" communicator */
148  communicator_root_[0] = true;
149 
150  /* expicitly set the "self" communicator */
151  communicators_[0].mpi_comm() = MPI_COMM_SELF;
152 
153  /* double check the size of communicators */
154  for (int i = 1; i < num_comm; i++) {
155  if (communicators_[i].size() != communicator_size_[i]) {
156  TERMINATE("communicator sizes don't match");
157  }
158  }
159 
160  for (int i = 0; i < (int)dimensions_.size(); i++) {
161  if (communicator(1 << i).rank() != coordinate(i)) {
162  TERMINATE("ranks don't match");
163  }
164  }
165 
166  if (communicator().cart_rank(coordinates_) != parent_communicator_.rank()) {
167  TERMINATE("cartesian and communicator ranks don't match");
168  }
169  }

248 /// A bundle of MPI communicators.
249 /** Each MPI rank is assigned to one of the MPI subgroups. */
250 class Communicator_bundle
251 {
252  private:
253  /// MPI subgroup communicator
254  Communicator comm_;
255 
256  /// Total number of communicators.
257  int size_;
258 
259  /// ID of a communication subgroup.
260  int id_;
261 
262  public:
263  Communicator_bundle()
264  : size_(-1)
265  , id_(-1)
266  {
267  }
268 
269  Communicator_bundle(Communicator const& base_comm__, int num_elements__)
270  {
271  /* number of communicators can't be larger than the number of MPI ranks */
272  size_ = std::min(base_comm__.size(), num_elements__);
273  int rank = base_comm__.rank();
274  /* assign ID in cyclic fasion */
275  id_ = rank % size_;
276  /* split parent communicator */
277  CALL_MPI(MPI_Comm_split, (base_comm__.mpi_comm(), id_, rank, &comm_.mpi_comm()));
278  }
