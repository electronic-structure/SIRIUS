327 mdarray<double, 3> Real_space_prj::generate_beta_radial_integrals(mdarray<Spline<double>, 2>& beta_rf__, int m__)
328 {
329  PROFILE("sirius::Real_space_prj::generate_beta_radial_integrals");
330 
331  mdarray<double, 3> beta_radial_integrals(unit_cell_.max_mt_radial_basis_size(),
332  unit_cell_.num_atom_types(),
333  gvec_.num_shells());
334 
335  splindex<block> spl_gsh(gvec_.num_shells(), comm_.size(), comm_.rank());
336  #pragma omp parallel
337  {
338  mdarray<Spline<double>, 2> jl(unit_cell_.lmax() + 1, unit_cell_.num_atom_types());
339  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++)
340  {
341  for (int l = 0; l <= unit_cell_.lmax(); l++) jl(l, iat) = Spline<double>(beta_rf__(0, iat).radial_grid());
342  }
343 
344  #pragma omp for
345  for (int igsh_loc = 0; igsh_loc < (int)spl_gsh.local_size(); igsh_loc++)
346  {
347  int igsh = (int)spl_gsh[igsh_loc];
348 
349  /* get spherical Bessel functions */
350  double G = gvec_.shell_len(igsh);
351  std::vector<double> v(unit_cell_.lmax() + 1);
352  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++)
353  {
354  for (int ir = 0; ir < beta_rf__(0, iat).num_points(); ir++)
355  {
356  double x = beta_rf__(0, iat).x(ir) * G;
357  gsl_sf_bessel_jl_array(unit_cell_.lmax(), x, &v[0]);
358  for (int l = 0; l <= unit_cell_.lmax(); l++) jl(l, iat)[ir] = v[l];
359  }
360  for (int l = 0; l <= unit_cell_.lmax(); l++) jl(l, iat).interpolate();
361  }
362 
363  /* compute radial integrals */
364  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++)
365  {
366  auto& atom_type = unit_cell_.atom_type(iat);
367  for (int idxrf = 0; idxrf < atom_type.mt_radial_basis_size(); idxrf++)
368  {
369  int l = atom_type.indexr(idxrf).l;
370  int nr = beta_rf__(idxrf, iat).num_points();
371  beta_radial_integrals(idxrf, iat, igsh) = inner(jl(l, iat), beta_rf__(idxrf, iat), m__, nr);
372  }
373  }
374  }
375  }
376 
377  int ld = unit_cell_.max_mt_radial_basis_size() * unit_cell_.num_atom_types();
378  comm_.allgather(beta_radial_integrals.at<CPU>(), static_cast<int>(ld * spl_gsh.global_offset()),
379  static_cast<int>(ld * spl_gsh.local_size()));
