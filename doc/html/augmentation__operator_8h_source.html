46  void generate_pw_coeffs(double omega__, Gvec const& gvec__, Radial_integrals const& radial_integrals__)
47  {
48  PROFILE("sirius::Augmentation_operator::generate_pw_coeffs");
49 
50  //auto qri = get_radial_integrals(gvec__);
51 
52  double fourpi_omega = fourpi / omega__;
53 
54  /* maximum l of beta-projectors */
55  int lmax_beta = atom_type_.indexr().lmax();
56  int lmmax = Utils::lmmax(2 * lmax_beta);
57 
58  std::vector<int> l_by_lm = Utils::l_by_lm(2 * lmax_beta);
59 
60  std::vector<double_complex> zilm(lmmax);
61  for (int l = 0, lm = 0; l <= 2 * lmax_beta; l++) {
62  for (int m = -l; m <= l; m++, lm++) {
63  zilm[lm] = std::pow(double_complex(0, 1), l);
64  }
65  }
66 
67  /* Gaunt coefficients of three real spherical harmonics */
68  Gaunt_coefficients<double> gaunt_coefs(lmax_beta, 2 * lmax_beta, lmax_beta, SHT::gaunt_rlm);
69 
70  /* split G-vectors between ranks */
71  int gvec_count = gvec__.gvec_count(comm_.rank());
72  int gvec_offset = gvec__.gvec_offset(comm_.rank());
73 
74  /* array of real spherical harmonics for each G-vector */
75  mdarray<double, 2> gvec_rlm(Utils::lmmax(2 * lmax_beta), gvec_count);
76  for (int igloc = 0; igloc < gvec_count; igloc++) {
77  int ig = gvec_offset + igloc;
78  auto rtp = SHT::spherical_coordinates(gvec__.gvec_cart(ig));
79  SHT::spherical_harmonics(2 * lmax_beta, rtp[1], rtp[2], &gvec_rlm(0, igloc));
80  }
81 
82  /* number of beta-projectors */
83  int nbf = atom_type_.mt_basis_size();
84 
85  q_mtrx_ = mdarray<double, 2>(nbf, nbf);
86  q_mtrx_.zero();
87 
88  /* array of plane-wave coefficients */
89  q_pw_ = mdarray<double, 2>(nbf * (nbf + 1) / 2, 2 * gvec_count, memory_t::host_pinned, "q_pw_");
90  #pragma omp parallel for
91  for (int igloc = 0; igloc < gvec_count; igloc++) {
92  int ig = gvec_offset + igloc;
93  double g = gvec__.gvec_len(ig);
94 
95  std::vector<double_complex> v(lmmax);
96 
97  for (int xi2 = 0; xi2 < nbf; xi2++) {
98  int lm2 = atom_type_.indexb(xi2).lm;
99  int idxrf2 = atom_type_.indexb(xi2).idxrf;
100 
101  for (int xi1 = 0; xi1 <= xi2; xi1++) {
102  int lm1 = atom_type_.indexb(xi1).lm;
103  int idxrf1 = atom_type_.indexb(xi1).idxrf;
104 
105  /* packed orbital index */
106  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
107  /* packed radial-function index */
108  int idxrf12 = idxrf2 * (idxrf2 + 1) / 2 + idxrf1;
109 
110  for (int lm3 = 0; lm3 < lmmax; lm3++) {
111  v[lm3] = std::conj(zilm[lm3]) * gvec_rlm(lm3, igloc) * radial_integrals__.aug_radial_integral(idxrf12, l_by_lm[lm3], atom_type_.id(), g);
112  }
113 
114  double_complex z = fourpi_omega * gaunt_coefs.sum_L3_gaunt(lm2, lm1, &v[0]);
115  q_pw_(idx12, 2 * igloc) = z.real();
116  q_pw_(idx12, 2 * igloc + 1) = z.imag();
117  }
118  }
119  }
120 
121  sym_weight_ = mdarray<double, 1>(nbf * (nbf + 1) / 2, memory_t::host_pinned, "sym_weight_");
122  for (int xi2 = 0; xi2 < nbf; xi2++) {
123  for (int xi1 = 0; xi1 <= xi2; xi1++) {
124  /* packed orbital index */
125  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
126  sym_weight_(idx12) = (xi1 == xi2) ? 1 : 2;
127  }
128  }
129 
130  if (comm_.rank() == 0) {
131  for (int xi2 = 0; xi2 < nbf; xi2++) {
132  for (int xi1 = 0; xi1 <= xi2; xi1++) {
133  /* packed orbital index */
134  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
135  q_mtrx_(xi1, xi2) = q_mtrx_(xi2, xi1) = omega__ * q_pw_(idx12, 0);
136  }
137  }
138  }
139  /* broadcast from rank#0 */
140  comm_.bcast(&q_mtrx_(0, 0), nbf * nbf , 0);
141 
142  #ifdef __PRINT_OBJECT_CHECKSUM
143  double cs = q_pw_.checksum();
144  comm_.allreduce(&cs, 1);
145  DUMP("checksum(q_pw) : %18.10f", cs);
146  #endif
147  }
