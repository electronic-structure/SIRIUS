35 /// Parse unit cell input section.
36 /** The following part of the input file is parsed:
37  * \code{.json}
38  * "unit_cell" : {
39  * "lattice_vectors" : [
40  * [a1_x, a1_y, a1_z],
41  * [a2_x, a2_y, a2_z],
42  * [a3_x, a3_y, a3_z]
43  * ],
44  *
45  * "lattice_vectors_scale" : scale,
46  *
47  * "atom_types" : [label_A, label_B, ...],
48  *
49  * "atom_files" : {
50  * label_A : file_A,
51  * label_B : file_B,
52  * ...
53  * },
54  *
55  * "atoms" : {
56  * label_A: [
57  * coordinates_A_1,
58  * coordinates_A_2,
59  * ...
60  * ],
61  * label_B : [
62  * coordinates_B_1,
63  * coordinates_B_2,
64  * ...
65  * ]
66  * }
67  * }
68  * \endcode
69  */
70 struct Unit_cell_input_section
71 {
72  vector3d<double> a0_;
73  vector3d<double> a1_;
74  vector3d<double> a2_;
75 
76  std::vector<std::string> labels_;
77  std::map<std::string, std::string> atom_files_;
78  std::vector< std::vector< std::vector<double> > > coordinates_;
79 
80  bool exist_{false};
81 
82  void read(json const& parser)
83  {
84  if (parser.count("unit_cell")) {
85  exist_ = true;
86 
87  auto section = parser["unit_cell"];
88  auto a0 = section["lattice_vectors"][0].get<std::vector<double>>();
89  auto a1 = section["lattice_vectors"][1].get<std::vector<double>>();
90  auto a2 = section["lattice_vectors"][2].get<std::vector<double>>();
91 
92  if (a0.size() != 3 || a1.size() != 3 || a2.size() != 3) {
93  TERMINATE("wrong lattice vectors");
94  }
95 
96  double scale = section.value("lattice_vectors_scale", 1.0);
97 
98  for (int x: {0, 1, 2}) {
99  a0_[x] = a0[x] * scale;
100  a1_[x] = a1[x] * scale;
101  a2_[x] = a2[x] * scale;
102  }
103 
104  matrix3d<double> lv;
105  for (int x: {0, 1, 2}) {
106  lv(x, 0) = a0_[x];
107  lv(x, 1) = a1_[x];
108  lv(x, 2) = a2_[x];
109  }
110  auto ilv = inverse(lv);
111 
112  labels_.clear();
113  coordinates_.clear();
114 
115  std::string units = section.value("atom_coordinate_units", "lattice");
116 
117  for (auto& label: section["atom_types"]) {
118  if (std::find(std::begin(labels_), std::end(labels_), label) != std::end(labels_)) {
119  TERMINATE("duplicate atom type label");
120  }
121  labels_.push_back(label);
122  }
123 
124  if (section.count("atom_files")) {
125  for (auto& label: labels_) {
126  atom_files_[label] = section["atom_files"].value(label, "");
127  }
128  }
129 
130  for (int iat = 0; iat < (int)labels_.size(); iat++) {
131  coordinates_.push_back(std::vector< std::vector<double> >());
132  for (size_t ia = 0; ia < section["atoms"][labels_[iat]].size(); ia++) {
133  auto v = section["atoms"][labels_[iat]][ia].get<std::vector<double>>();
134 
135  if (!(v.size() == 3 || v.size() == 6)) {
136  TERMINATE("wrong coordinates size");
137  }
138  if (v.size() == 3) {
139  v.resize(6, 0.0);
140  }
141 
142  vector3d<double> v1(v[0], v[1], v[2]);
143  if (units == "A") {
144  for (int x: {0, 1, 2}) {
145  v1[x] /= bohr_radius;
146  }
147  }
148  if (units == "au" || units == "A") {
149  v1 = ilv * v1;
150  auto rv1 = reduce_coordinates(v1);
151  for (int x: {0, 1, 2}) {
152  v[x] = rv1.first[x];
153  }
154  }
155 
156  coordinates_[iat].push_back(v);
157  }
158  }
159  }
160  }
161 };

238 /// Parse control input section.
239 /** The following part of the input file is parsed:
240  * \code{.json}
241  * "control" : {
242  * "mpi_grid_dims" : (1- 2- or 3-dimensional vector<int>) MPI grid layout
243  * "cyclic_block_size" : (int) PBLAS / ScaLAPACK block size
244  * "reduce_gvec" : (bool) use reduced G-vector set (reduce_gvec = true) or full set (reduce_gvec = false)
245  * "std_evp_solver_type" : (string) type of eigen-solver for the standard eigen-problem
246  * "gen_evp_solver_type" : (string) type of eigen-solver for the generalized eigen-problem
247  * "electronic_structure_method" : (string) electronic structure method
248  * "processing_unit" : (string) primary processing unit
249  * "fft_mode" : (string) serial or parallel FFT
250  * }
251  * \endcode
252  */
253 struct Control_input_section

255  std::vector<int> mpi_grid_dims_;
256  int cyclic_block_size_{-1};
257  bool reduce_gvec_{true};
258  std::string std_evp_solver_name_{""};
259  std::string gen_evp_solver_name_{""};
260  std::string fft_mode_{"serial"};
261  std::string processing_unit_{""};
262  double rmt_max_{2.2};
263  double spglib_tolerance_{1e-4};
264  /// Level of verbosity.
265  /** The following convention in proposed:
266  * - 0: silent mode (no output is printed) \n
267  * - 1: basic output (low level of output) \n
268  * - 2: extended output (medium level of output) \n
269  * - 3: extensive output (hi level of output) */
270  int verbosity_{0};
