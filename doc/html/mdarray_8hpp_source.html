188  mdarray_mem_mgr(size_t const size__, memory_t mode__)
189  : size_(size__),

191  {
192  if ((mode_ & memory_t::host) != memory_t::none || (mode_ & memory_t::host_pinned) != memory_t::none) {
193  mdarray_mem_count::allocated() += size_ * sizeof(T);

201  {

203  mdarray_mem_count::allocated() -= size_ * sizeof(T);

257  void init_dimensions(std::initializer_list<mdarray_index_descriptor> const args)
258  {
259  assert(args.size() == N);
260 
261  int i{0};
262  for (auto d : args) {
263  dims_[i++] = d;
264  }
265 
266  offsets_[0] = -dims_[0].begin();
267  size_t ld = 1;
268  for (int i = 1; i < N; i++) {
269  ld *= dims_[i - 1].size();
270  offsets_[i] = ld;
271  offsets_[0] -= ld * dims_[i].begin();
272  }
273  }

418  /// Allocate memory for array.
419  void allocate(memory_t memory__) const
420  {
421  if ((memory__ & memory_t::host) != memory_t::none && (memory__ & memory_t::host_pinned) != memory_t::none) {
422  printf("error at line %i of file %s: host memory can only be of one type\n", __LINE__, __FILE__);
423  exit(0);
424  }
425 
426  #ifndef __GPU
427  if ((memory__ & memory_t::host_pinned) != memory_t::none) {
428  memory__ = memory_t::host;
429  }
430  #endif
431 
432  size_t sz = size();
433 
434  /* host allocation */
435  if ((memory__ & memory_t::host) != memory_t::none || (memory__ & memory_t::host_pinned) != memory_t::none) {
436  if ((memory__ & memory_t::host) != memory_t::none) {
437  raw_ptr_ = static_cast<T*>(malloc(sz * sizeof(T)));
438  unique_ptr_ = std::unique_ptr<T[], mdarray_mem_mgr<T>>(raw_ptr_, mdarray_mem_mgr<T>(sz, memory_t::host));
439  }
440 
441  if ((memory__ & memory_t::host_pinned) != memory_t::none) {
442  #ifdef __GPU
443  raw_ptr_ = static_cast<T*>(cuda_malloc_host(sz * sizeof(T)));
444  unique_ptr_ = std::unique_ptr<T[], mdarray_mem_mgr<T>>(raw_ptr_, mdarray_mem_mgr<T>(sz, memory_t::host_pinned));
445  #endif
446  }
447 
448  /* call constructor on non-trivial data */
449  if (raw_ptr_ != nullptr && !std::is_pod<T>::value) {
450  for (size_t i = 0; i < sz; i++) {
451  new (raw_ptr_ + i) T();
452  }
453  }
454  }
455 
456  /* device allocation */
457  if ((memory__ & memory_t::device) != memory_t::none) {
458  #ifdef __GPU
459  raw_ptr_device_ = static_cast<T*>(cuda_malloc(sz * sizeof(T)));
460  unique_ptr_device_ = std::unique_ptr<T[], mdarray_mem_mgr<T>>(raw_ptr_device_, mdarray_mem_mgr<T>(sz, memory_t::device));
461  #endif
462  }
463  }
