92 inline void Band::solve_sv(K_point* kp,
93  Periodic_function<double>* effective_magnetic_field[3]) const
94 {
95  PROFILE("sirius::Band::solve_sv");
96 
97  if (!ctx_.need_sv()) {
98  kp->bypass_sv();
99  return;
100  }
101 
102  if (kp->num_ranks() > 1 && !std_evp_solver().parallel()) {
103  TERMINATE("eigen-value solver is not parallel");
104  }
105 
106  std::vector<double> band_energies(ctx_.num_bands());
107 
108  /* product of the second-variational Hamiltonian and a first-variational wave-function */
109  std::vector<wave_functions> hpsi;
110  for (int i = 0; i < ctx_.num_mag_comp(); i++) {
111  hpsi.push_back(std::move(wave_functions(ctx_.processing_unit(),
112  kp->gkvec(),
113  unit_cell_.num_atoms(),
114  [this](int ia)
115  {
116  return unit_cell_.atom(ia).mt_basis_size();
117  },
118  ctx_.num_fv_states())));
119  }
120 
121  /* compute product of magnetic field and wave-function */
122  if (ctx_.num_spins() == 2) {
123  apply_magnetic_field(kp->fv_states(), kp->gkvec(), effective_magnetic_field, hpsi);
124  }
125  else {
126  hpsi[0].pw_coeffs().prime().zero();
127  hpsi[0].mt_coeffs().prime().zero();
128  }
129 
130  //== if (ctx_.uj_correction())
131  //== {
132  //== apply_uj_correction<uu>(kp->fv_states_col(), hpsi);
133  //== if (ctx_.num_mag_dims() != 0) apply_uj_correction<dd>(kp->fv_states_col(), hpsi);
134  //== if (ctx_.num_mag_dims() == 3)
135  //== {
136  //== apply_uj_correction<ud>(kp->fv_states_col(), hpsi);
137  //== if (ctx_.std_evp_solver()->parallel()) apply_uj_correction<du>(kp->fv_states_col(), hpsi);
138  //== }
139  //== }
140 
141  //== if (ctx_.so_correction()) apply_so_correction(kp->fv_states_col(), hpsi);
142 
143  int nfv = ctx_.num_fv_states();
144  int bs = ctx_.cyclic_block_size();
145 
146  #ifdef __GPU
147  if (ctx_.processing_unit() == GPU) {
148  kp->fv_states().allocate_on_device();
149  kp->fv_states().copy_to_device(0, nfv);
150  for (int i = 0; i < ctx_.num_mag_comp(); i++) {
151  hpsi[i].allocate_on_device();
152  hpsi[i].copy_to_device(0, nfv);
153  }
154  }
155  #endif
156 
157  #ifdef __PRINT_OBJECT_CHECKSUM
158  auto z1 = kp->fv_states().checksum(0, nfv);
159  DUMP("checksum(fv_states): %18.10f %18.10f", std::real(z1), std::imag(z1));
160  for (int i = 0; i < ctx_.num_mag_comp(); i++) {
161  z1 = hpsi[i].checksum(0, nfv);
162  DUMP("checksum(hpsi[i]): %18.10f %18.10f", std::real(z1), std::imag(z1));
163  }
164  #endif
165 
166  if (ctx_.num_mag_dims() != 3) {
167  dmatrix<double_complex> h(nfv, nfv, ctx_.blacs_grid(), bs, bs);
168  #ifdef __GPU
169  if (kp->num_ranks() == 1 && ctx_.processing_unit() == GPU) {
170  h.allocate(memory_t::device);
171  }
172  #endif
173  /* perform one or two consecutive diagonalizations */
174  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
175 
176  /* compute <wf_i | h * wf_j> */
177  inner(kp->fv_states(), 0, nfv, hpsi[ispn], 0, nfv, h, 0, 0);
178 
179  for (int i = 0; i < nfv; i++) {
180  h.add(i, i, kp->fv_eigen_value(i));
181  }
182  #ifdef __PRINT_OBJECT_CHECKSUM
183  auto z1 = h.checksum();
184  DUMP("checksum(h): %18.10f %18.10f", std::real(z1), std::imag(z1));
185  #endif
186  sddk::timer t1("sirius::Band::solve_sv|stdevp");
187  std_evp_solver().solve(nfv, h.at<CPU>(), h.ld(), &band_energies[ispn * nfv],
188  kp->sv_eigen_vectors(ispn).at<CPU>(), kp->sv_eigen_vectors(ispn).ld());
189  }
190  } else {
191  int nb = ctx_.num_bands();
192  dmatrix<double_complex> h(nb, nb, ctx_.blacs_grid(), bs, bs);
193  #ifdef __GPU
194  if (kp->num_ranks() == 1 && ctx_.processing_unit() == GPU) {
195  h.allocate(memory_t::device);
196  }
197  #endif
198  /* compute <wf_i | h * wf_j> for up-up block */
199  inner(kp->fv_states(), 0, nfv, hpsi[0], 0, nfv, h, 0, 0);
200  /* compute <wf_i | h * wf_j> for dn-dn block */
201  inner(kp->fv_states(), 0, nfv, hpsi[1], 0, nfv, h, nfv, nfv);
202  /* compute <wf_i | h * wf_j> for up-dn block */
203  inner(kp->fv_states(), 0, nfv, hpsi[2], 0, nfv, h, 0, nfv);
204 
205  if (kp->comm().size() == 1) {
206  for (int i = 0; i < nfv; i++) {
207  for (int j = 0; j < nfv; j++) {
208  h(nfv + j, i) = std::conj(h(i, nfv + j));
209  }
210  }
211  } else {
212  linalg<CPU>::tranc(nfv, nfv, h, 0, nfv, h, nfv, 0);
213  }
214 
215  for (int i = 0; i < nfv; i++) {
216  h.add(i, i, kp->fv_eigen_value(i));
217  h.add(i + nfv, i + nfv, kp->fv_eigen_value(i));
218  }
219  #ifdef __PRINT_OBJECT_CHECKSUM
220  auto z1 = h.checksum();
221  DUMP("checksum(h): %18.10f %18.10f", std::real(z1), std::imag(z1));
222  #endif
223  sddk::timer t1("sirius::Band::solve_sv|stdevp");
224  std_evp_solver().solve(nb, h.at<CPU>(), h.ld(), &band_energies[0],
225  kp->sv_eigen_vectors(0).at<CPU>(), kp->sv_eigen_vectors(0).ld());
226  }
227 
228  #ifdef __GPU
229  if (ctx_.processing_unit() == GPU) {
230  kp->fv_states().deallocate_on_device();
231  for (int i = 0; i < ctx_.num_mag_comp(); i++) {
232  hpsi[i].deallocate_on_device();
233  }
234  }
235  #endif
236 
237  kp->set_band_energies(&band_energies[0]);
238 }

240 inline void Band::solve_for_kset(K_point_set& kset__,
241  Potential& potential__,
242  bool precompute__) const
243 {
244  PROFILE("sirius::Band::solve_for_kset");
245 
246  if (precompute__ && ctx_.full_potential()) {
247  potential__.generate_pw_coefs();
248  potential__.update_atomic_potential();
249  unit_cell_.generate_radial_functions();
250  unit_cell_.generate_radial_integrals();
251  }
252 
253  if (ctx_.full_potential()) {
254  local_op_->prepare(ctx_.gvec_coarse(), ctx_.num_mag_dims(), potential__.effective_potential(),
255  potential__.effective_magnetic_field(), ctx_.step_function());
256  } else {
257  local_op_->prepare(ctx_.gvec_coarse(), ctx_.num_mag_dims(), potential__.effective_potential(),
258  potential__.effective_magnetic_field());
259  }
260 
261  /* solve secular equation and generate wave functions */
262  for (int ikloc = 0; ikloc < kset__.spl_num_kpoints().local_size(); ikloc++) {
263  int ik = kset__.spl_num_kpoints(ikloc);
264 
265  if (use_second_variation && ctx_.full_potential()) {
266  /* solve non-magnetic Hamiltonian (so-called first variation) */
267  diag_fv_full_potential(kset__.k_point(ik), potential__);
268  /* generate first-variational states */
269  kset__.k_point(ik)->generate_fv_states();
270  /* solve magnetic Hamiltonian */
271  solve_sv(kset__.k_point(ik), potential__.effective_magnetic_field());
272  /* generate spinor wave-functions */
273  kset__.k_point(ik)->generate_spinor_wave_functions();
274  } else {
275  solve_fd(kset__.k_point(ik), potential__.effective_potential(), potential__.effective_magnetic_field());
276  }
277  }
278 
279  local_op_->dismiss();
280 
281  /* synchronize eigen-values */
282  kset__.sync_band_energies();
283 
284  if (ctx_.control().verbosity_ > 0 && ctx_.comm().rank() == 0) {
285  printf("Lowest band energies\n");
286  for (int ik = 0; ik < kset__.num_kpoints(); ik++) {
287  printf("ik : %2i, ", ik);
288  if (ctx_.num_mag_dims() != 1) {
289  for (int j = 0; j < std::min(ctx_.control().num_bands_to_print_, ctx_.num_bands()); j++) {
290  printf("%12.6f", kset__.k_point(ik)->band_energy(j));
291  }
292  } else {
293  for (int j = 0; j < std::min(ctx_.control().num_bands_to_print_, ctx_.num_fv_states()); j++) {
294  printf("%12.6f", kset__.k_point(ik)->band_energy(j));
295  }
296  printf("\n ");
297  for (int j = 0; j < std::min(ctx_.control().num_bands_to_print_, ctx_.num_fv_states()); j++) {
298  printf("%12.6f", kset__.k_point(ik)->band_energy(ctx_.num_fv_states() + j));
299  }
300  }
301  printf("\n");
302  }
