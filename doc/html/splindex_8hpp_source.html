280  inline typename splindex_base<T>::location_t location(T idxglob__) const
281  {
282  assert(idxglob__ < this->global_index_size_);
283 
284  /* number of full blocks */
285  T num_blocks = idxglob__ / block_size_;
286

288  T idxloc = (num_blocks / this->num_ranks_) * block_size_ + idxglob__ % block_size_;

297  inline T local_size(int rank__) const
298  {
299  assert(rank__ >= 0 && rank__ < this->num_ranks_);
300 
301  /* number of full blocks */
302  T num_blocks = this->global_index_size_ / block_size_;
303 
304  T n = (num_blocks / this->num_ranks_) * block_size_;
305 
306  int rank_offs = static_cast<int>(num_blocks % this->num_ranks_);
307 
308  if (rank__ < rank_offs) {
309  n += block_size_;
310  } else if (rank__ == rank_offs) {
311  n += this->global_index_size_ % block_size_;
312  }
313  return n;
