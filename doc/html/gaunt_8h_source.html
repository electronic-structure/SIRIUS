52 /// Compact storage of non-zero Gaunt coefficients \f$ \langle \ell_1 m_1 | \ell_3 m_3 | \ell_2 m_2 \rangle \f$.
53 /** Very important! The following notation is adopted and used everywhere: lm1 and lm2 represent 'bra' and 'ket'
54  * spherical harmonics of the Gaunt integral and lm3 represent the inner spherical harmonic.
55  */
56 template <typename T>

87  /// Class constructor.
88  Gaunt_coefficients(int lmax1__,
89  int lmax3__,
90  int lmax2__,
91  T (*get__)(int l1, int l2, int l3, int m1, int m2, int m3))
92  : lmax1_(lmax1__),
93  lmax3_(lmax3__),
94  lmax2_(lmax2__),
95  get_(get__)
96  {
97  lmmax1_ = Utils::lmmax(lmax1_);
98  lmmax3_ = Utils::lmmax(lmax3_);
99  lmmax2_ = Utils::lmmax(lmax2_);
100 
101  gaunt_packed_L1_L2_.resize(lmmax3_);
102  gaunt_L1_L2<T> g12;
103 
104  gaunt_packed_L3_ = mdarray<std::vector<gaunt_L3<T> >, 2>(lmmax1_, lmmax2_);
105  gaunt_L3<T> g3;
106 
107  for (int l1 = 0; l1 <= lmax1_; l1++) {
108  for (int m1 = -l1; m1 <= l1; m1++) {
109  int lm1 = Utils::lm_by_l_m(l1, m1);
110  for (int l2 = 0; l2 <= lmax2_; l2++) {
111  for (int m2 = -l2; m2 <= l2; m2++) {
112  int lm2 = Utils::lm_by_l_m(l2, m2);
113  for (int l3 = 0; l3 <= lmax3_; l3++) {
114  for (int m3 = -l3; m3 <= l3; m3++) {
115  int lm3 = Utils::lm_by_l_m(l3, m3);
116 
117  T gc = get_(l1, l3, l2, m1, m3, m2);
118  if (std::abs(gc) > 1e-12) {
119  g12.lm1 = lm1;
120  g12.lm2 = lm2;
121  g12.coef = gc;
122  gaunt_packed_L1_L2_[lm3].push_back(g12);
123 
124  g3.lm3 = lm3;
125  g3.coef = gc;
126  gaunt_packed_L3_(lm1, lm2).push_back(g3);
127  }
128  }
129  }
130  }
131  }
132  }
133  }
134  }

178  /// Return a sum over L3 (lm3) index of Gaunt coefficients and a complex vector.
179  /** The following operation is performed:
180  * \f[
181  * \sum_{\ell_3 m_3} \langle \ell_1 m_1 | \ell_3 m_3 | \ell_2 m_2 \rangle v_{\ell_3 m_3}
182  * \f]
183  * Result is assumed to be complex.
184  */
185  inline double_complex sum_L3_gaunt(int lm1, int lm2, double_complex const* v) const
186  {
187  double_complex zsum(0, 0);
188  for (int k = 0; k < (int)gaunt_packed_L3_(lm1, lm2).size(); k++) {
189  zsum += gaunt_packed_L3_(lm1, lm2)[k].coef * v[gaunt_packed_L3_(lm1, lm2)[k].lm3];
190  }
191  return zsum;
192  }
193 
194  /// Return a sum over L3 (lm3) index of Gaunt coefficients and a real vector.
195  /** The following operation is performed:
196  * \f[
197  * \sum_{\ell_3 m_3} \langle \ell_1 m_1 | \ell_3 m_3 | \ell_2 m_2 \rangle v_{\ell_3 m_3}
198  * \f]
199  * Result is assumed to be of the same type as Gaunt coefficients.
200  */
201  inline T sum_L3_gaunt(int lm1, int lm2, double const* v) const
202  {
203  T sum = 0;
204  for (int k = 0; k < (int)gaunt_packed_L3_(lm1, lm2).size(); k++) {
205  sum += gaunt_packed_L3_(lm1, lm2)[k].coef * v[gaunt_packed_L3_(lm1, lm2)[k].lm3];
206  }
207  return sum;
208  }
