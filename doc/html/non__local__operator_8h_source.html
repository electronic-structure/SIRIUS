58  Non_local_operator(Beta_projectors& beta__, device_t pu__) : beta_(beta__), pu_(pu__)
59  {
60  PROFILE("sirius::Non_local_operator::Non_local_operator");
61 
62  auto& uc = beta_.unit_cell();
63  packed_mtrx_offset_ = mdarray<int, 1>(uc.num_atoms());
64  packed_mtrx_size_ = 0;

66  {
67  int nbf = uc.atom(ia).mt_basis_size();
68  packed_mtrx_offset_(ia) = packed_mtrx_size_;
69  packed_mtrx_size_ += nbf * nbf;
70  }

72  #ifdef __GPU

99 template<>
100 inline void Non_local_operator<double_complex>::apply(int chunk__,
101  int ispn__,
102  wave_functions& op_phi__,
103  int idx0__,
104  int n__)
105 {
106  PROFILE("sirius::Non_local_operator::apply");
107 
108  if (is_null_) return;
109 
110  assert(op_phi__.pw_coeffs().num_rows_loc() == beta_.num_gkvec_loc());
111 
112  auto beta_phi = beta_.beta_phi<double_complex>(chunk__, n__);
113  auto& beta_gk = beta_.beta_gk();
114  int num_gkvec_loc = beta_.num_gkvec_loc();
115  int nbeta = beta_.beta_chunk(chunk__).num_beta_;
116 
117  if (static_cast<size_t>(nbeta * n__) > work_.size())
118  {
119  work_ = mdarray<double_complex, 1>(nbeta * n__);
120  #ifdef __GPU
121  if (pu_ == GPU) work_.allocate(memory_t::device);
122  #endif
123  }
124 
125  if (pu_ == CPU)
126  {
127  #pragma omp parallel for
128  for (int i = 0; i < beta_.beta_chunk(chunk__).num_atoms_; i++)
129  {
130  /* number of beta functions for a given atom */
131  int nbf = beta_.beta_chunk(chunk__).desc_(0, i);
132  int offs = beta_.beta_chunk(chunk__).desc_(1, i);
133  int ia = beta_.beta_chunk(chunk__).desc_(3, i);
134 
135  /* compute O * <beta|phi> */
136  linalg<CPU>::gemm(0, 0, nbf, n__, nbf,
137  op_.at<CPU>(packed_mtrx_offset_(ia), ispn__), nbf,
138  beta_phi.at<CPU>(offs, 0), nbeta,
139  work_.at<CPU>(offs), nbeta);
140  }
141 
142  /* compute <G+k|beta> * O * <beta|phi> and add to op_phi */
143  linalg<CPU>::gemm(0, 0, num_gkvec_loc, n__, nbeta, double_complex(1, 0),
144  beta_gk.at<CPU>(), num_gkvec_loc, work_.at<CPU>(), nbeta, double_complex(1, 0),
145  op_phi__.pw_coeffs().prime().at<CPU>(0, idx0__), op_phi__.pw_coeffs().prime().ld());
146  }
147  #ifdef __GPU
148  if (pu_ == GPU)
149  {
150  #pragma omp parallel for
151  for (int i = 0; i < beta_.beta_chunk(chunk__).num_atoms_; i++)
152  {
153  /* number of beta functions for a given atom */
154  int nbf = beta_.beta_chunk(chunk__).desc_(0, i);
155  int offs = beta_.beta_chunk(chunk__).desc_(1, i);
156  int ia = beta_.beta_chunk(chunk__).desc_(3, i);
157 
158  /* compute O * <beta|phi> */
159  linalg<GPU>::gemm(0, 0, nbf, n__, nbf,
160  op_.at<GPU>(packed_mtrx_offset_(ia), ispn__), nbf,
161  beta_phi.at<GPU>(offs, 0), nbeta,
162  work_.at<GPU>(offs), nbeta,
163  omp_get_thread_num());
164 
165  }
166  cuda_device_synchronize();
167  double_complex alpha(1, 0);
168 
169  /* compute <G+k|beta> * O * <beta|phi> and add to op_phi */
170  linalg<GPU>::gemm(0, 0, num_gkvec_loc, n__, nbeta, &alpha,
171  beta_gk.at<GPU>(), beta_gk.ld(), work_.at<GPU>(), nbeta, &alpha,
172  op_phi__.pw_coeffs().prime().at<GPU>(0, idx0__), op_phi__.pw_coeffs().prime().ld());
173 
174  cuda_device_synchronize();
175  }
176  #endif
177 }

259 template <typename T>
260 class D_operator: public Non_local_operator<T>
261 {
262  public:
263 
264  D_operator(Simulation_context const& ctx__, Beta_projectors& beta__) : Non_local_operator<T>(beta__, ctx__.processing_unit())
265  {
266  this->op_ = mdarray<T, 2>(this->packed_mtrx_size_, ctx__.num_mag_dims() + 1);
267  this->op_.zero();
268 
269  auto& uc = this->beta_.unit_cell();
270 
271  for (int j = 0; j < ctx__.num_mag_dims() + 1; j++) {
272  for (int ia = 0; ia < uc.num_atoms(); ia++) {
273  int nbf = uc.atom(ia).mt_basis_size();
274  for (int xi2 = 0; xi2 < nbf; xi2++) {
275  for (int xi1 = 0; xi1 < nbf; xi1++) {
276  this->op_(this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, j) = uc.atom(ia).d_mtrx(xi1, xi2, j);
277  }
278  }
279  }
280  }
281  if (ctx__.num_mag_dims()) {
282  for (int ia = 0; ia < uc.num_atoms(); ia++) {
283  int nbf = uc.atom(ia).mt_basis_size();
284  for (int xi2 = 0; xi2 < nbf; xi2++) {
285  for (int xi1 = 0; xi1 < nbf; xi1++) {
286  auto v0 = this->op_(this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, 0);
287  auto v1 = this->op_(this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, 1);
288  this->op_(this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, 0) = std::real(v0 + v1);
289  this->op_(this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, 1) = std::real(v0 - v1);
290  }
291  }
292  }
293  }
294  #ifdef __GPU
295  if (this->pu_ == GPU) {
296  this->op_.allocate(memory_t::device);
297  this->op_.copy_to_device();
298  }
299  #endif
300  }

308  Q_operator(Simulation_context const& ctx__, Beta_projectors& beta__) : Non_local_operator<T>(beta__, ctx__.processing_unit())
309  {

311  this->op_ = mdarray<T, 2>(this->packed_mtrx_size_, 1);

319  continue;
320  }
321  int nbf = uc.atom(ia).mt_basis_size();
322  for (int xi2 = 0; xi2 < nbf; xi2++)
323  {
324  for (int xi1 = 0; xi1 < nbf; xi1++)
325  {
326  if (ctx__.unit_cell().atom_type(iat).pp_desc().augment) {
327  this->op_(this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, 0) = ctx__.augmentation_op(iat).q_mtrx(xi1, xi2);
