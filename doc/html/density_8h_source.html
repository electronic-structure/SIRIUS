60 /// Generate charge density and magnetization from occupied spinor wave-functions.
61 /** Let's start from the definition of the complex density matrix:
62  * \f[
63  * \rho_{\sigma' \sigma}({\bf r}) =
64  * \sum_{j{\bf k}} n_{j{\bf k}} \Psi_{j{\bf k}}^{\sigma*}({\bf r}) \Psi_{j{\bf k}}^{\sigma'}({\bf r}) =
65  * \frac{1}{2} \left( \begin{array}{cc} \rho({\bf r})+m_z({\bf r}) &
66  * m_x({\bf r})-im_y({\bf r}) \\ m_x({\bf r})+im_y({\bf r}) & \rho({\bf r})-m_z({\bf r}) \end{array} \right)
67  * \f]
68  * We notice that the diagonal components of the density matrix are actually real and the off-diagonal components are
69  * expressed trough two independent functions \f$ m_x({\bf r}) \f$ and \f$ m_y({\bf r}) \f$. Having this in mind we
70  * will work with a slightly different object, namely a real density matrix, defined as a 1-, 2- or 4-dimensional
71  * (depending on the number of magnetic components) vector with the following elements:
72  * - \f$ [ \rho({\bf r}) ] \f$ in case of non-magnetic configuration
73  * - \f$ [ \rho_{\uparrow \uparrow}({\bf r}), \rho_{\downarrow \downarrow}({\bf r}) ] =
74  * [ \frac{\rho({\bf r})+m_z({\bf r})}{2}, \frac{\rho({\bf r})-m_z({\bf r})}{2} ] \f$ in case of collinear
75  * magnetic configuration
76  * - \f$ [ \rho_{\uparrow \uparrow}({\bf r}), \rho_{\downarrow \downarrow}({\bf r}),
77  * 2 \Re \rho_{\uparrow \downarrow}({\bf r}), -2 \Im \rho_{\uparrow \downarrow}({\bf r}) ] =
78  * [ \frac{\rho({\bf r})+m_z({\bf r})}{2}, \frac{\rho({\bf r})-m_z({\bf r})}{2},
79  * m_x({\bf r}), m_y({\bf r}) ] \f$ in the general case of non-collinear magnetic configuration
80  *
81  * At this point it is straightforward to compute the density and magnetization in the interstitial (see add_k_point_contribution_rg()).
82  * The muffin-tin part of the density and magnetization is obtained in a slighlty more complicated way. Recall the
83  * expansion of spinor wave-functions inside the muffin-tin \f$ \alpha \f$
84  * \f[
85  * \Psi_{j{\bf k}}^{\sigma}({\bf r}) = \sum_{\xi}^{N_{\xi}^{\alpha}} {S_{\xi}^{\sigma j {\bf k},\alpha}}
86  * f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)Y_{\ell_{\xi}m_{\xi}}(\hat {\bf r})
87  * \f]
88  * which we insert into expression for the complex density matrix:
89  * \f[
90  * \rho_{\sigma' \sigma}({\bf r}) = \sum_{j{\bf k}} n_{j{\bf k}} \sum_{\xi}^{N_{\xi}^{\alpha}}
91  * S_{\xi}^{\sigma j {\bf k},\alpha*} f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)
92  * Y_{\ell_{\xi}m_{\xi}}^{*}(\hat {\bf r}) \sum_{\xi'}^{N_{\xi'}^{\alpha}} S_{\xi'}^{\sigma' j{\bf k},\alpha}
93  * f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r)Y_{\ell_{\xi'}m_{\xi'}}(\hat {\bf r})
94  * \f]
95  * First, we eliminate a sum over bands and k-points by forming an auxiliary density tensor:
96  * \f[
97  * D_{\xi \sigma, \xi' \sigma'}^{\alpha} = \sum_{j{\bf k}} n_{j{\bf k}} S_{\xi}^{\sigma j {\bf k},\alpha*}
98  * S_{\xi'}^{\sigma' j {\bf k},\alpha}
99  * \f]
100  * The expression for complex density matrix simplifies to:
101  * \f[
102  * \rho_{\sigma' \sigma}({\bf r}) = \sum_{\xi \xi'} D_{\xi \sigma, \xi' \sigma'}^{\alpha}
103  * f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)Y_{\ell_{\xi}m_{\xi}}^{*}(\hat {\bf r})
104  * f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r)Y_{\ell_{\xi'}m_{\xi'}}(\hat {\bf r})
105  * \f]
106  * Now we can switch to the real density matrix and write its' expansion in real spherical harmonics. Let's take
107  * non-magnetic case as an example:
108  * \f[
109  * \rho({\bf r}) = \sum_{\xi \xi'} D_{\xi \xi'}^{\alpha}
110  * f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)Y_{\ell_{\xi}m_{\xi}}^{*}(\hat {\bf r})
111  * f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r)Y_{\ell_{\xi'}m_{\xi'}}(\hat {\bf r}) =
112  * \sum_{\ell_3 m_3} \rho_{\ell_3 m_3}^{\alpha}(r) R_{\ell_3 m_3}(\hat {\bf r})
113  * \f]
114  * where
115  * \f[
116  * \rho_{\ell_3 m_3}^{\alpha}(r) = \sum_{\xi \xi'} D_{\xi \xi'}^{\alpha} f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)
117  * f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r) \langle Y_{\ell_{\xi}m_{\xi}} | R_{\ell_3 m_3} | Y_{\ell_{\xi'}m_{\xi'}} \rangle
118  * \f]
119  * We are almost done. Now it is time to switch to the full index notation \f$ \xi \rightarrow \{ \ell \lambda m \} \f$
120  * and sum over \a m and \a m' indices:
121  * \f[
122  * \rho_{\ell_3 m_3}^{\alpha}(r) = \sum_{\ell \lambda, \ell' \lambda'} f_{\ell \lambda}^{\alpha}(r)
123  * f_{\ell' \lambda'}^{\alpha}(r) d_{\ell \lambda, \ell' \lambda', \ell_3 m_3}^{\alpha}
124  * \f]
125  * where
126  * \f[
127  * d_{\ell \lambda, \ell' \lambda', \ell_3 m_3}^{\alpha} =
128  * \sum_{mm'} D_{\ell \lambda m, \ell' \lambda' m'}^{\alpha}
129  * \langle Y_{\ell m} | R_{\ell_3 m_3} | Y_{\ell' m'} \rangle
130  * \f]
131  * This is our final answer: radial components of density and magnetization are expressed as a linear combination of
132  * quadratic forms in radial functions.
133  *
134  * \note density and potential are allocated as global function because it's easier to load and save them. */

205  /// Symmetrize density matrix.
206  /** Initially, density matrix is obtained with summation over irreducible BZ:
207  * \f[
208  * \tilde n_{\ell \lambda m \sigma, \ell' \lambda' m' \sigma'}^{\alpha} =
209  * \sum_{j} \sum_{{\bf k}}^{IBZ} \langle Y_{\ell m} u_{\ell \lambda}^{\alpha}| \Psi_{j{\bf k}}^{\sigma} \rangle w_{\bf k} n_{j{\bf k}}
210  * \langle \Psi_{j{\bf k}}^{\sigma'} | u_{\ell' \lambda'}^{\alpha} Y_{\ell' m'} \rangle
211  * \f]
212  * In order to symmetrize it, the following operation is performed:
213  * \f[
214  * n_{\ell \lambda m \sigma, \ell' \lambda' m' \sigma'}^{\alpha} = \sum_{{\bf P}}
215  * \sum_{j} \sum_{\bf k}^{IBZ} \langle Y_{\ell m} u_{\ell \lambda}^{\alpha}| \Psi_{j{\bf P}{\bf k}}^{\sigma} \rangle w_{\bf k} n_{j{\bf k}}
216  * \langle \Psi_{j{\bf P}{\bf k}}^{\sigma'} | u_{\ell' \lambda'}^{\alpha} Y_{\ell' m'} \rangle
217  * \f]
218  * where \f$ {\bf P} \f$ is the space-group symmetry operation. The inner product between wave-function and
219  * local orbital is transformed as:
220  * \f[
221  * \langle \Psi_{j{\bf P}{\bf k}}^{\sigma} | u_{\ell \lambda}^{\alpha} Y_{\ell m} \rangle =
222  * \int \Psi_{j{\bf P}{\bf k}}^{\sigma *}({\bf r}) u_{\ell \lambda}^{\alpha}(r) Y_{\ell m}(\hat {\bf r}) dr =
223  * \int \Psi_{j{\bf k}}^{\sigma *}({\bf P}^{-1}{\bf r}) u_{\ell \lambda}^{\alpha}(r) Y_{\ell m}(\hat {\bf r}) dr =
224  * \int \Psi_{j{\bf k}}^{\sigma *}({\bf r}) u_{\ell \lambda}^{{\bf P}\alpha}(r) Y_{\ell m}({\bf P} \hat{\bf r}) dr
225  * \f]
226  * Under rotation the spherical harmonic is transformed as:
227  * \f[
228  * Y_{\ell m}({\bf P} \hat{\bf r}) = {\bf P}^{-1}Y_{\ell m}(\hat {\bf r}) = \sum_{m'} D_{m'm}^{\ell}({\bf P}^{-1}) Y_{\ell m'}(\hat {\bf r}) =
229  * \sum_{m'} D_{mm'}^{\ell}({\bf P}) Y_{\ell m'}(\hat {\bf r})
230  * \f]
231  * The inner-product integral is then rewritten as:
232  * \f[
233  * \langle \Psi_{j{\bf P}{\bf k}}^{\sigma} | u_{\ell \lambda}^{\alpha} Y_{\ell m} \rangle =
234  * \sum_{m'} D_{mm'}^{\ell}({\bf P}) \langle \Psi_{j{\bf k}}^{\sigma} | u_{\ell \lambda}^{{\bf P}\alpha} Y_{\ell m} \rangle
235  * \f]
236  * and the final expression for density matrix gets the following form:
237  * \f[
238  * n_{\ell \lambda m \sigma, \ell' \lambda' m' \sigma'}^{\alpha} = \sum_{{\bf P}}
239  * \sum_{j} \sum_{\bf k}^{IBZ} \sum_{m_1 m_2} D_{mm_1}^{\ell *}({\bf P}) D_{m'm_2}^{\ell'}({\bf P})
240  * \langle Y_{\ell m_1} u_{\ell \lambda}^{{\bf P} \alpha}|
241  * \Psi_{j{\bf k}}^{\sigma} \rangle w_{\bf k} n_{j{\bf k}} \langle \Psi_{j{\bf k}}^{\sigma'} |
242  * u_{\ell' \lambda'}^{{\bf P}\alpha} Y_{\ell' m_2} \rangle = \sum_{{\bf P}}
243  * \sum_{m_1 m_2} D_{mm_1}^{\ell *}({\bf P}) D_{m'm_2}^{\ell'}({\bf P})
244  * \tilde n_{\ell \lambda m_1 \sigma, \ell' \lambda' m_2 \sigma'}^{{\bf P}\alpha}
245  * \f]
246  */
247  void symmetrize_density_matrix();

258  void reduce_density_matrix(Atom_type const& atom_type__,
259  int ia__,
260  mdarray<double_complex, 4> const& zdens__,
261  Gaunt_coefficients<T> const& gaunt_coeffs__,
262  mdarray<double, 3>& mt_density_matrix__)
263  {
264  mt_density_matrix__.zero();
265 
266  #pragma omp parallel for default(shared)
267  for (int idxrf2 = 0; idxrf2 < atom_type__.mt_radial_basis_size(); idxrf2++) {
268  int l2 = atom_type__.indexr(idxrf2).l;
269  for (int idxrf1 = 0; idxrf1 <= idxrf2; idxrf1++) {
270  int offs = idxrf2 * (idxrf2 + 1) / 2 + idxrf1;
271  int l1 = atom_type__.indexr(idxrf1).l;
272 
273  int xi2 = atom_type__.indexb().index_by_idxrf(idxrf2);
274  for (int lm2 = Utils::lm_by_l_m(l2, -l2); lm2 <= Utils::lm_by_l_m(l2, l2); lm2++, xi2++) {
275  int xi1 = atom_type__.indexb().index_by_idxrf(idxrf1);
276  for (int lm1 = Utils::lm_by_l_m(l1, -l1); lm1 <= Utils::lm_by_l_m(l1, l1); lm1++, xi1++) {
277  for (int k = 0; k < gaunt_coeffs__.num_gaunt(lm1, lm2); k++) {
278  int lm3 = gaunt_coeffs__.gaunt(lm1, lm2, k).lm3;
279  T gc = gaunt_coeffs__.gaunt(lm1, lm2, k).coef;
280  switch (num_mag_dims) {
281  case 3: {
282  mt_density_matrix__(lm3, offs, 2) += 2.0 * std::real(zdens__(xi1, xi2, 2, ia__) * gc);
283  mt_density_matrix__(lm3, offs, 3) -= 2.0 * std::imag(zdens__(xi1, xi2, 2, ia__) * gc);
284  }
285  case 1: {
286  mt_density_matrix__(lm3, offs, 1) += std::real(zdens__(xi1, xi2, 1, ia__) * gc);
287  }
288  case 0: {
289  mt_density_matrix__(lm3, offs, 0) += std::real(zdens__(xi1, xi2, 0, ia__) * gc);
290  }
291  }
292  }

299  /// Add k-point contribution to the auxiliary density matrix.
300  /** In case of full-potential LAPW complex density matrix has the following expression:
301  * \f[
302  * d_{\xi \sigma, \xi' \sigma'}^{\alpha} = \sum_{j{\bf k}} n_{j{\bf k}}
303  * S_{\xi}^{\sigma j {\bf k},\alpha*} S_{\xi'}^{\sigma' j {\bf k},\alpha}
304  * \f]
305  *
306  * where \f$ S_{\xi}^{\sigma j {\bf k},\alpha} \f$ are the expansion coefficients of
307  * spinor wave functions inside muffin-tin spheres.
308  *
309  * In case of LDA+U the occupation matrix is also computed. It has the following expression:
310  * \f[
311  * n_{\ell,mm'}^{\sigma \sigma'} = \sum_{i {\bf k}}^{occ} \int_{0}^{R_{MT}} r^2 dr
312  * \Psi_{\ell m}^{i{\bf k}\sigma *}({\bf r}) \Psi_{\ell m'}^{i{\bf k}\sigma'}({\bf r})
313  * \f]
314  *
315  * In case of ultrasoft pseudopotential the following density matrix has to be computed for each atom:
316  * \f[
317  * d_{\xi \xi'}^{\alpha} = \langle \beta_{\xi}^{\alpha} | \hat N | \beta_{\xi'}^{\alpha} \rangle =
318  * \sum_{j {\bf k}} \langle \beta_{\xi}^{\alpha} | \Psi_{j{\bf k}} \rangle n_{j{\bf k}}
319  * \langle \Psi_{j{\bf k}} | \beta_{\xi'}^{\alpha} \rangle
320  * \f]
321  * Here \f$ \hat N = \sum_{j{\bf k}} | \Psi_{j{\bf k}} \rangle n_{j{\bf k}} \langle \Psi_{j{\bf k}} | \f$ is
322  * the occupancy operator written in spectral representation. */

613  /// Generate valence charge density and magnetization from the wave functions.
614  /** The interstitial density is generated on the coarse FFT grid and then transformed to the PW domain.
615  * After symmetrization and mixing and before the generation of the XC potential density is transformted to the
616  * real-space domain and checked for the number of electrons. */
617  inline void generate_valence(K_point_set& ks__);
618 
619  /// Add augmentation charge Q(r).
620  /** Restore valence density by adding the Q-operator constribution.
621  * The following term is added to the valence density, generated by the pseudo wave-functions:
622  * \f[
623  * \tilde \rho({\bf G}) = \sum_{\alpha} \sum_{\xi \xi'} d_{\xi \xi'}^{\alpha} Q_{\xi' \xi}^{\alpha}({\bf G})
624  * \f]
625  * Plane-wave coefficients of the Q-operator for a given atom \f$ \alpha \f$ can be obtained from the
626  * corresponding coefficients of the Q-operator for a given atom \a type A:
627  * \f[
628  * Q_{\xi' \xi}^{\alpha(A)}({\bf G}) = e^{-i{\bf G}\tau_{\alpha(A)}} Q_{\xi' \xi}^{A}({\bf G})
629  * \f]
630  * We use this property to split the sum over atoms into sum over atom types and inner sum over atoms of the
631  * same type:
632  * \f[
633  * \tilde \rho({\bf G}) = \sum_{A} \sum_{\xi \xi'} Q_{\xi' \xi}^{A}({\bf G}) \sum_{\alpha(A)}
634  * d_{\xi \xi'}^{\alpha(A)} e^{-i{\bf G}\tau_{\alpha(A)}} =
635  * \sum_{A} \sum_{\xi \xi'} Q_{\xi' \xi}^{A}({\bf G}) d_{\xi \xi'}^{A}({\bf G})
636  * \f]
637  * where
638  * \f[
639  * d_{\xi \xi'}^{A}({\bf G}) = \sum_{\alpha(A)} d_{\xi \xi'}^{\alpha(A)} e^{-i{\bf G}\tau_{\alpha(A)}}
640  * \f]
641  */
642  void augment(K_point_set& ks__)
643  {
644  PROFILE("sirius::Density::augment");
645 
646  /*check if we need to augment charge density and magnetization */
647  bool need_to_augment{false};
648  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
649  need_to_augment |= unit_cell_.atom_type(iat).pp_desc().augment;
650  }
651  if (!need_to_augment) {
652  return;
653  }
654 
655  /* collect density and magnetization into single array */
656  std::vector<Periodic_function<double>*> rho_vec(ctx_.num_mag_dims() + 1);
657  rho_vec[0] = rho_;
658  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
659  rho_vec[1 + j] = magnetization_[j];
660  }
661 
662  if (ctx_.control().print_checksum_) {
663  for (auto e: rho_vec) {
664  auto cs = e->checksum_pw();
665  DUMP("checksum(rho_vec_pw): %20.14f %20.14f", cs.real(), cs.imag());
666  }
667  }
668 
669  mdarray<double_complex, 2> rho_aug(ctx_.gvec_count(), ctx_.num_mag_dims() + 1, ctx_.dual_memory_t());
670 
671  switch (ctx_.processing_unit()) {
672  case CPU: {
673  generate_rho_aug<CPU>(rho_vec, rho_aug);
674  break;
675  }
676  case GPU: {
677  generate_rho_aug<GPU>(rho_vec, rho_aug);
678  break;
679  }
680  }
681 
682  for (int iv = 0; iv < ctx_.num_mag_dims() + 1; iv++) {
683  #pragma omp parallel for
684  for (int igloc = 0; igloc < ctx_.gvec_count(); igloc++) {
685  int ig = ctx_.gvec_offset() + igloc;
686  rho_vec[iv]->f_pw(ig) += rho_aug(igloc, iv);
687  }
688  }
689 
690  sddk::timer t5("sirius::Density::augment|mpi");
691  for (auto e: rho_vec) {
692  ctx_.comm().allgather(&e->f_pw(0), ctx_.gvec_offset(), ctx_.gvec_count());
693 
694  if (ctx_.control().print_checksum_) {
695  auto cs = e->checksum_pw();
696  DUMP("checksum(rho_vec_pw): %20.14f %20.14f", cs.real(), cs.imag());
697  }
698  }
699  t5.stop();
700  }
