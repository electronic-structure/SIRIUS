204  /// Setup the Hermitian subspace matrix.
205  /** Compute \f$ O_{ii'} = \langle \phi_i | \hat O | \phi_{i'} \rangle \f$ operator matrix
206  * for the subspace spanned by the wave-functions \f$ \phi_i \f$. The matrix is always returned
207  * in the CPU pointer because most of the standard math libraries start from the CPU. */
208  template <typename T>
209  inline void set_subspace_mtrx(int N__,
210  int n__,
211  wave_functions& phi__,
212  wave_functions& op_phi__,
213  dmatrix<T>& mtrx__,
214  dmatrix<T>& mtrx_old__) const
215  {
216  PROFILE("sirius::Band::set_subspace_mtrx");
217 
218  assert(n__ != 0);
219  if (mtrx_old__.size()) {
220  assert(&mtrx__.blacs_grid() == &mtrx_old__.blacs_grid());
221  }
222 
223  /* copy old N x N distributed matrix */
224  if (N__ > 0) {
225  splindex<block_cyclic> spl_row(N__, mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
226  splindex<block_cyclic> spl_col(N__, mtrx__.blacs_grid().num_ranks_col(), mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
227 
228  #pragma omp parallel for
229  for (int i = 0; i < spl_col.local_size(); i++) {
230  std::memcpy(&mtrx__(0, i), &mtrx_old__(0, i), spl_row.local_size() * sizeof(T));
231  }
232  }
233 
234  /* <{phi,phi_new}|Op|phi_new> */
235  inner(phi__, 0, N__ + n__, op_phi__, N__, n__, mtrx__, 0, N__);
236 
237  /* restore lower part */
238  if (N__ > 0) {
239  if (mtrx__.blacs_grid().comm().size() == 1) {
240  #pragma omp parallel for
241  for (int i = 0; i < N__; i++) {
242  for (int j = N__; j < N__ + n__; j++) {
243  mtrx__(j, i) = std::conj(mtrx__(i, j));
244  }
245  }
246  } else {
247  linalg<CPU>::tranc(n__, N__, mtrx__, 0, N__, mtrx__, N__, 0);
248  }
249  }
250 
251  if (ctx_.control().print_checksum_) {
252  splindex<block_cyclic> spl_row(N__ + n__, mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
253  splindex<block_cyclic> spl_col(N__ + n__, mtrx__.blacs_grid().num_ranks_col(), mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
254  double_complex cs(0, 0);
255  for (int i = 0; i < spl_col.local_size(); i++) {
256  for (int j = 0; j < spl_row.local_size(); j++) {
257  cs += mtrx__(j, i);
258  }
259  }
260  mtrx__.blacs_grid().comm().allreduce(&cs, 1);
261  DUMP("checksum(subspace_mtrx): %18.10f %18.10f", cs.real(), cs.imag());
262  }
263 
264  mtrx__.make_real_diag(N__ + n__);
265 
266  /* save new matrix */
267  if (mtrx_old__.size()) {
268  splindex<block_cyclic> spl_row(N__ + n__, mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
269  splindex<block_cyclic> spl_col(N__ + n__, mtrx__.blacs_grid().num_ranks_col(), mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
270 
271  #pragma omp parallel for
272  for (int i = 0; i < spl_col.local_size(); i++) {
273  std::memcpy(&mtrx_old__(0, i), &mtrx__(0, i), spl_row.local_size() * sizeof(T));
274  }
275  }
276  }

290  /// Diagonalize a pseudo-potential Hamiltonian.
291  template <typename T>
292  void diag_pseudo_potential(K_point* kp__,
293  Periodic_function<double>* effective_potential__,
294  Periodic_function<double>* effective_magnetic_field__[3]) const
295  {
296  PROFILE("sirius::Band::diag_pseudo_potential");
297 
298  local_op_->prepare(kp__->gkvec());
299  ctx_.fft_coarse().prepare(kp__->gkvec().partition());
300 
301  D_operator<T> d_op(ctx_, kp__->beta_projectors());
302  Q_operator<T> q_op(ctx_, kp__->beta_projectors());
303 
304  auto& itso = ctx_.iterative_solver_input_section();
305  if (itso.type_ == "exact") {
306  if (ctx_.num_mag_dims() != 3) {
307  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
308  diag_pseudo_potential_exact(kp__, ispn, d_op, q_op);
309  }
310  } else {
311  STOP();
312  }
313  } else if (itso.type_ == "davidson") {
314  if (ctx_.num_mag_dims() != 3) {
315  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
316  diag_pseudo_potential_davidson(kp__, ispn, d_op, q_op);
317  }
318  } else {
319  STOP();
320  }
321  } else if (itso.type_ == "rmm-diis") {
322  if (ctx_.num_mag_dims() != 3) {
323  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
324  diag_pseudo_potential_rmm_diis(kp__, ispn, d_op, q_op);
325  }
326  } else {
327  STOP();
328  }
329  } else if (itso.type_ == "chebyshev") {
330  P_operator<T> p_op(ctx_, kp__->beta_projectors(), kp__->p_mtrx());
331  if (ctx_.num_mag_dims() != 3) {
332  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
333  diag_pseudo_potential_chebyshev(kp__, ispn, d_op, q_op, p_op);
334 
335  }
336  } else {
337  STOP();
338  }
339  } else {
340  TERMINATE("unknown iterative solver type");
341  }
342 
343  ctx_.fft_coarse().dismiss();
344  }

458  /// Apply the muffin-tin part of the Hamiltonian to the apw basis functions of an atom.
459  /** The following matrix is computed:
460  * \f[
461  * b_{L_2 \nu_2}^{\alpha}({\bf G'}) = \sum_{L_1 \nu_1} \sum_{L_3}
462  * a_{L_1\nu_1}^{\alpha}({\bf G'})
463  * \langle u_{\ell_1\nu_1}^{\alpha} | h_{L3}^{\alpha} | u_{\ell_2\nu_2}^{\alpha}
464  * \rangle \langle Y_{L_1} | R_{L_3} | Y_{L_2} \rangle
465  * \f]
466  */
467  template <spin_block_t sblock>
468  void apply_hmt_to_apw(Atom const& atom__,
469  int num_gkvec__,
470  mdarray<double_complex, 2>& alm__,
471  mdarray<double_complex, 2>& halm__) const
472  {
473  auto& type = atom__.type();
474 
475  // TODO: this is k-independent and can in principle be precomputed together with radial integrals if memory is available
476  // TODO: for spin-collinear case hmt is Hermitian; compute upper triangular part and use zhemm
477  mdarray<double_complex, 2> hmt(type.mt_aw_basis_size(), type.mt_aw_basis_size());
478  /* compute the muffin-tin Hamiltonian */
479  for (int j2 = 0; j2 < type.mt_aw_basis_size(); j2++) {
480  int lm2 = type.indexb(j2).lm;
481  int idxrf2 = type.indexb(j2).idxrf;
482  for (int j1 = 0; j1 < type.mt_aw_basis_size(); j1++) {
483  int lm1 = type.indexb(j1).lm;
484  int idxrf1 = type.indexb(j1).idxrf;
485  hmt(j1, j2) = atom__.radial_integrals_sum_L3<sblock>(idxrf1, idxrf2, gaunt_coefs_->gaunt_vector(lm1, lm2));
486  }
487  }
488  linalg<CPU>::gemm(0, 1, num_gkvec__, type.mt_aw_basis_size(), type.mt_aw_basis_size(), alm__, hmt, halm__);
489  }

491  void apply_o1mt_to_apw(Atom const& atom__,
492  int num_gkvec__,
493  mdarray<double_complex, 2>& alm__,
494  mdarray<double_complex, 2>& oalm__) const
495  {
496  auto& type = atom__.type();
497 
498  for (int j = 0; j < type.mt_aw_basis_size(); j++) {
499  int l = type.indexb(j).l;
500  int lm = type.indexb(j).lm;
501  int idxrf = type.indexb(j).idxrf;
502  for (int order = 0; order < type.aw_order(l); order++) {
503  int j1 = type.indexb().index_by_lm_order(lm, order);
504  int idxrf1 = type.indexr().index_by_l_order(l, order);
505  for (int ig = 0; ig < num_gkvec__; ig++) {
506  oalm__(ig, j) += atom__.symmetry_class().o1_radial_integral(idxrf, idxrf1) * alm__(ig, j1);
507  }
508  }
509  }

530  /// Setup the Hamiltonian and overlap matrices in APW+lo basis
531  /** The Hamiltonian matrix has the following expression:
532  * \f[
533  * H_{\mu' \mu}=\langle \varphi_{\mu' } | \hat H | \varphi_{\mu } \rangle =
534  * \left( \begin{array}{cc}
535  * H_{\bf G'G} & H_{{\bf G'}j} \\
536  * H_{j'{\bf G}} & H_{j'j}
537  * \end{array} \right)
538  * \f]
539  * APW-APW block:
540  * \f{eqnarray*}{
541  * H_{{\bf G'} {\bf G}}^{\bf k} &=& \sum_{\alpha} \sum_{L'\nu', L\nu} a_{L'\nu'}^{\alpha *}({\bf G'+k})
542  * \langle u_{\ell' \nu'}^{\alpha}Y_{\ell' m'}|\hat h^{\alpha} | u_{\ell \nu}^{\alpha}Y_{\ell m} \rangle
543  * a_{L\nu}^{\alpha}({\bf G+k}) + \frac{1}{2}{\bf G'} {\bf G} \cdot \Theta({\bf G - G'}) + \tilde V_{eff}({\bf G - G'}) \\
544  * &=& \sum_{\alpha} \sum_{\xi' } a_{\xi'}^{\alpha *}({\bf G'+k})
545  * b_{\xi'}^{\alpha}({\bf G+k}) + \frac{1}{2}{\bf G'} {\bf G} \cdot \Theta({\bf G - G'}) + \tilde V_{eff}({\bf G - G'})
546  * \f}
547  * APW-lo block:
548  * \f[
549  * H_{{\bf G'} j}^{\bf k} = \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'+k})
550  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}|\hat h^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle
551  * \f]
552  * lo-APW block:
553  * \f[
554  * H_{j' {\bf G}}^{\bf k} = \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
555  * |\hat h^{\alpha_{j'}} | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G+k})
556  * \f]
557  * lo-lo block:
558  * \f[
559  * H_{j' j}^{\bf k} = \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
560  * |\hat h^{\alpha_{j}} | \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle \delta_{\alpha_j \alpha_{j'}}
561  * \f]
562  *
563  * The overlap matrix has the following expression:
564  * \f[
565  * O_{\mu' \mu} = \langle \varphi_{\mu'} | \varphi_{\mu} \rangle
566  * \f]
567  * APW-APW block:
568  * \f[
569  * O_{{\bf G'} {\bf G}}^{\bf k} = \sum_{\alpha} \sum_{L\nu} a_{L\nu}^{\alpha *}({\bf G'+k})
570  * a_{L\nu}^{\alpha}({\bf G+k}) + \Theta({\bf G-G'})
571  * \f]
572  *
573  * APW-lo block:
574  * \f[
575  * O_{{\bf G'} j}^{\bf k} = \sum_{\nu'} a_{\ell_j m_j \nu'}^{\alpha_j *}({\bf G'+k})
576  * \langle u_{\ell_j \nu'}^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j} \rangle
577  * \f]
578  *
579  * lo-APW block:
580  * \f[
581  * O_{j' {\bf G}}^{\bf k} =
582  * \sum_{\nu'} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} | u_{\ell_{j'} \nu'}^{\alpha_{j'}} \rangle
583  * a_{\ell_{j'} m_{j'} \nu'}^{\alpha_{j'}}({\bf G+k})
584  * \f]
585  *
586  * lo-lo block:
587  * \f[
588  * O_{j' j}^{\bf k} = \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} |
589  * \phi_{\ell_{j}}^{\zeta_{j} \alpha_{j}} \rangle \delta_{\alpha_{j'} \alpha_j}
590  * \delta_{\ell_{j'} \ell_j} \delta_{m_{j'} m_j}
591  * \f]
592  */

599  /// Apply LAPW Hamiltonain and overlap to the trial wave-functions.
600  /** Check the documentation of Band::set_fv_h_o() for the expressions of Hamiltonian and overlap
601  * matrices and \ref basis for the definition of the LAPW+lo basis.
602  *
603  * For the set of wave-functions expanded in LAPW+lo basis (k-point index is dropped for simplicity)
604  * \f[
605  * \psi_{i} = \sum_{\mu} \phi_{\mu} C_{\mu i}
606  * \f]
607  * where \f$ \mu = \{ {\bf G}, j \} \f$ is a combined index of LAPW and local orbitals we want to contrusct
608  * a subspace Hamiltonian and overlap matrices:
609  * \f[
610  * H_{i' i} = \langle \psi_{i'} | \hat H | \psi_i \rangle =
611  * \sum_{\mu' \mu} C_{\mu' i'}^{*} \langle \phi_{\mu'} | \hat H | \phi_{\mu} \rangle C_{\mu i} =
612  * \sum_{\mu'} C_{\mu' i'}^{*} h_{\mu' i}(\psi)
613  * \f]
614  * \f[
615  * O_{i' i} = \langle \psi_{i'} | \psi_i \rangle =
616  * \sum_{\mu' \mu} C_{\mu' i'}^{*} \langle \phi_{\mu'} | \phi_{\mu} \rangle C_{\mu i} =
617  * \sum_{\mu'} C_{\mu' i'}^{*} o_{\mu' i}(\psi)
618  * \f]
619  * where
620  * \f[
621  * h_{\mu' i}(\psi) = \sum_{\mu} \langle \phi_{\mu'} | \hat H | \phi_{\mu} \rangle C_{\mu i}
622  * \f]
623  * and
624  * \f[
625  * o_{\mu' i}(\psi) = \sum_{\mu} \langle \phi_{\mu'} | \phi_{\mu} \rangle C_{\mu i}
626  * \f]
627  * For the APW block of \f$ h_{\mu' i}(\psi) \f$ and \f$ o_{\mu' i}(\psi) \f$ we have:
628  * \f[
629  * h_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} +
630  * \sum_{j} \langle \phi_{\bf G'} | \hat H | \phi_{j} \rangle C_{j i}
631  * \f]
632  * \f[
633  * o_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \phi_{\bf G} \rangle C_{{\bf G} i} +
634  * \sum_{j} \langle \phi_{\bf G'} | \phi_{j} \rangle C_{j i}
635  * \f]
636  * and for the lo block:
637  * \f[
638  * h_{j' i}(\psi) = \sum_{{\bf G}} \langle \phi_{j'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} +
639  * \sum_{j} \langle \phi_{j'} | \hat H | \phi_{j} \rangle C_{j i}
640  * \f]
641  * \f[
642  * o_{j' i}(\psi) = \sum_{{\bf G}} \langle \phi_{j'} | \phi_{\bf G} \rangle C_{{\bf G} i} +
643  * \sum_{j} \langle \phi_{j'} | \phi_{j} \rangle C_{j i}
644  * \f]
645  *
646  * APW-APW contribution, muffin-tin part:
647  * \f[
648  * h_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} =
649  * \sum_{{\bf G}} \sum_{\alpha} \sum_{\xi'} a_{\xi'}^{\alpha *}({\bf G'}) b_{\xi'}^{\alpha}({\bf G})
650  * C_{{\bf G} i}
651  * \f]
652  * \f[
653  * o_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \phi_{\bf G} \rangle C_{{\bf G} i} =
654  * \sum_{{\bf G}} \sum_{\alpha} \sum_{\xi'} a_{\xi'}^{\alpha *}({\bf G'}) a_{\xi'}^{\alpha}({\bf G})
655  * C_{{\bf G} i}
656  * \f]
657  * APW-APW contribution, interstitial effective potential part:
658  * \f[
659  * h_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} V({\bf r}) \psi_{i}({\bf r}) d{\bf r}
660  * \f]
661  * This is done by transforming \f$ \psi_i({\bf G}) \f$ to the real space, multiplying by effectvive potential
662  * and step function and transforming the result back to the \f$ {\bf G} \f$ domain.
663  *
664  * APW-APW contribution, interstitial kinetic energy part:
665  * \f[
666  * h_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} \Big( -\frac{1}{2} \nabla \Big)
667  * \Big( \nabla \psi_{i}({\bf r}) \Big) d{\bf r}
668  * \f]
669  * and the gradient of the wave-function is computed with FFT as:
670  * \f[
671  * \Big( \nabla \psi_{i}({\bf r}) \Big) = \sum_{\bf G} i{\bf G}e^{i{\bf G}{\bf r}}\psi_i({\bf G})
672  * \f]
673  *
674  * APW-APW contribution, interstitial overlap:
675  * \f[
676  * o_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} \psi_{i}({\bf r}) d{\bf r}
677  * \f]
678  *
679  * APW-lo contribution:
680  * \f[
681  * h_{{\bf G'} i}(\psi) = \sum_{j} \langle \phi_{\bf G'} | \hat H | \phi_{j} \rangle C_{j i} =
682  * \sum_{j} C_{j i} \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'})
683  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}|\hat h^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle =
684  * \sum_{j} C_{j i} \sum_{\xi'} a_{\xi'}^{\alpha_j *}({\bf G'}) h_{\xi' \xi_j}^{\alpha_j}
685  * \f]
686  * \f[
687  * o_{{\bf G'} i}(\psi) = \sum_{j} \langle \phi_{\bf G'} | \phi_{j} \rangle C_{j i} =
688  * \sum_{j} C_{j i} \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'})
689  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}| \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle =
690  * \sum_{j} C_{j i} \sum_{\nu'} a_{\ell_j m_j \nu'}^{\alpha_j *}({\bf G'}) o_{\nu' \zeta_j \ell_j}^{\alpha_j}
691  * \f]
692  * lo-APW contribution:
693  * \f[
694  * h_{j' i}(\psi) = \sum_{\bf G} \langle \phi_{j'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} =
695  * \sum_{\bf G} C_{{\bf G} i} \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
696  * |\hat h^{\alpha_{j'}} | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G}) =
697  * \sum_{\bf G} C_{{\bf G} i} \sum_{\xi} h_{\xi_{j'} \xi}^{\alpha_{j'}} a_{\xi}^{\alpha_{j'}}({\bf G})
698  * \f]
699  * \f[
700  * o_{j' i}(\psi) = \sum_{\bf G} \langle \phi_{j'} | \phi_{\bf G} \rangle C_{{\bf G} i} =
701  * \sum_{\bf G} C_{{\bf G} i} \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
702  * | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G}) =
703  * \sum_{\bf G} C_{{\bf G} i} \sum_{\nu} o_{\zeta_{j'} \nu \ell_{j'}}^{\alpha_{j'}} a_{\ell_{j'} m_{j'} \nu}^{\alpha_{j'}}({\bf G})
704  * \f]
705  * lo-lo contribution:
706  * \f[
707  * h_{j' i}(\psi) = \sum_{j} \langle \phi_{j'} | \hat H | \phi_{j} \rangle C_{j i} = \sum_{j} C_{j i} h_{\xi_{j'} \xi_j}^{\alpha_j}
708  * \delta_{\alpha_j \alpha_{j'}}
709  * \f]
710  * \f[

712  * o_{\zeta_{j'} \zeta_{j} \ell_j}^{\alpha_j}
713  * \delta_{\alpha_j \alpha_{j'}} \delta_{\ell_j \ell_{j'}} \delta_{m_j m_{j'}}
714  * \f]
715  */
