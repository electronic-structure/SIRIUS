180  /// Write a multidimensional array.
181  template <typename T>
182  void write(const std::string& name, T const* data, const std::vector<int>& dims)
183  {
184  /* open group */
185  HDF5_group group(file_id_, path_);
186 
187  /* make dataspace */
188  HDF5_dataspace dataspace(dims);
189 
190  /* create new dataset */
191  HDF5_dataset dataset(group, dataspace, name, type_wrapper<T>::hdf5_type_id());
192 
193  /* write data */
194  if (H5Dwrite(dataset.id(), type_wrapper<T>::hdf5_type_id(), dataspace.id(), H5S_ALL, H5P_DEFAULT, data) < 0) {
195  TERMINATE("error in H5Dwrite()");
196  }
197  }

220  /// Constructor to create the HDF5 tree.
221  HDF5_tree(const std::string& file_name__, bool truncate)
222  : file_name_(file_name__),
223  file_id_(-1),
224  root_node_(true)
225  {
226  if (H5open() < 0) TERMINATE("error in H5open()");
227 
228  if (false) H5Eset_auto(H5E_DEFAULT, NULL, NULL);
229 
230  if (!truncate && Utils::file_exists(file_name_))
231  {
232  /* try to open existing file */
233  file_id_ = H5Fopen(file_name_.c_str(), H5F_ACC_RDWR, H5P_DEFAULT);
234 
235  if (file_id_ < 0) TERMINATE("H5Fopen() failed");
236  }
237  else
238  {
239  /* create a new file */
240  file_id_ = H5Fcreate(file_name_.c_str(), H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT);
241  if (file_id_ < 0) TERMINATE("error in H5Fcreate()");
242  }
243 
244  path_ = "/";
245  }
